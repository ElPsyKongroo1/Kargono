653 void ScriptModuleBuilder::CreateScriptModule()
654 {
655 // Release active script module so it is available to be written to...
656 KG_INFO("Closing active script module...");
657 ScriptService::CloseActiveScriptModule();
658
659 // Load in ScriptRegistry if not already loaded
660 if (Assets::AssetService::GetScriptRegistry().size() == 0)
661 {
662 KG_WARN("Loading script registry from disk since in-memory registry is empty");
663 Assets::AssetService::DeserializeScriptRegistry();
664 }
665
666 // Load in project components if not already loaded
667 if (Assets::AssetService::GetProjectComponentRegistry().size() == 0)
668 {
669 KG_WARN("Loading script registry from disk since in-memory registry is empty");
670 Assets::AssetService::DeserializeProjectComponentRegistry();
671 }
672
673 // Load in ai states if not already loaded
674 if (Assets::AssetService::GetAIStateRegistry().size() == 0)
675 {
676 KG_WARN("Loading script registry from disk since in-memory registry is empty");
677 Assets::AssetService::DeserializeAIStateRegistry();
678 }
679
680 KG_INFO("Creating Script Module CPP Files...");
681 CreateModuleHeaderFile();
682 bool generateCPPSuccess = CreateModuleCPPFile();
683 if (!generateCPPSuccess)
684 {
685 KG_WARN("Failure to generate C++ scripts from kgscripts");
686 ScriptService::LoadActiveScriptModule();
687 Assets::AssetService::DeserializeScriptRegistry();
688 return;
689 }
690
691 KG_INFO("Clearing previous compilation logs...");
692 Utility::FileSystem::DeleteSelectedFile("Log/BuildScriptLibraryDebug.log");
693 KG_INFO("Compiling debug script module...");
694#if defined(KG_PLATFORM_WINDOWS)
695 bool buildSuccessful = CompileModuleCodeMSVC(true);
696#elif defined(KG_PLATFORM_LINUX)
697 bool buildSuccessful = CompileModuleCodeGCC(true);
698#endif
699 if (!buildSuccessful)
700 {
701 KG_WARN("Failure to compile script module");
702 ScriptService::LoadActiveScriptModule();
703 Assets::AssetService::DeserializeScriptRegistry();
704 return;
705 }
706 KG_INFO("Clearing previous compilation logs...");
707 Utility::FileSystem::DeleteSelectedFile("Log/BuildScriptLibrary.log");
708 KG_INFO("Compiling release script module...");
709#if defined(KG_PLATFORM_WINDOWS)
710 buildSuccessful = CompileModuleCodeMSVC(false);
711#elif defined(KG_PLATFORM_LINUX)
712 buildSuccessful = CompileModuleCodeGCC(false);
713#endif
714 if (!buildSuccessful)
715 {
716 KG_WARN("Failed to compile release script module");
717 ScriptService::LoadActiveScriptModule();
718 Assets::AssetService::DeserializeScriptRegistry();
719 return;
720 }
721
722 KG_INFO("Opening New Scripting Module...");
723 ScriptService::LoadActiveScriptModule();
724
725 // Revalidate in-memory script cache
726 Assets::AssetService::DeserializeScriptRegistry();
727 for (auto& [handle, scriptRef] : Assets::AssetService::GetScriptCache())
728 {
729 ScriptService::LoadScriptFunction(scriptRef, scriptRef->m_FuncType);
730 }
731 KG_INFO("Successfully build and loaded new script module");
