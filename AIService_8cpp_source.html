42 void AIService::OnUpdate(Timestep timeStep)
43 {
44 // Ensure a valid scene is active
45 Ref<Scenes::Scene> activeScene = Scenes::SceneService::GetActiveScene();
46 KG_ASSERT(activeScene, "Invalid scene reference when calling AIService's OnUpdate()");
47
48 // Run on update for all active AI with AIComponents including the global, then the current state
49 for (entt::entity enttID : Scenes::SceneService::GetActiveScene()->GetAllEntitiesWith<ECS::AIStateComponent>())
50 {
51 ECS::Entity entity = activeScene->GetEntityByEnttID(enttID);
52 KG_ASSERT(entity, "Invalid entity obtained. Could not run OnUpdate on provided entity.");
53 ECS::AIStateComponent& aiComponent = entity.GetComponent<ECS::AIStateComponent>();
54
55 // Call Global State OnUpdate
56 if (aiComponent.GlobalStateReference && aiComponent.GlobalStateReference->OnUpdate)
57 {
58 Utility::CallWrappedVoidEntityFloat(aiComponent.GlobalStateReference->OnUpdate->m_Function, entity.GetUUID(), timeStep);
59 }
60
61 // Call Current State OnUpdate
62 if (aiComponent.CurrentStateReference && aiComponent.CurrentStateReference->OnUpdate)
63 {
64 Utility::CallWrappedVoidEntityFloat(aiComponent.CurrentStateReference->OnUpdate->m_Function, entity.GetUUID(), timeStep);
65 }
66 }
67
68 // Check/Handle AIContext's delayed messages queue
69 HandleDelayedMessages();

290 void AIService::SendAIMessage(uint32_t messageType, UUID senderEntity, UUID receiverEntity, float delayTime)
291 {
292 // Initialize message
293 KG_ASSERT(s_AIContext);
294
295 AIMessage newMessage{ messageType, senderEntity, receiverEntity, Utility::Time::GetTime() + delayTime };
296
297 // Check if message should be handled immediately or placed into delay queue
298 if (delayTime > 0.001f)
299 {
300 // Insert message into delay queue. If unsuccessful, increment delay time and retry until successful
301 s_AIContext->MessageQueue.push({std::move(newMessage)});
302 }
303 else
304 {
305 // Handle message
306 HandleAIMessage(std::move(newMessage));
307 }
308 }
309 void AIService::HandleAIMessage(const AIMessage& messageToHandle)
310 {
311 // Ensure a valid scene is active and a valid entity is provided
312 Ref<Scenes::Scene> activeScene = Scenes::SceneService::GetActiveScene();
313 KG_ASSERT(activeScene, "Invalid scene reference inside AIService");
314 ECS::Entity receiverEntity = activeScene->GetEntityByUUID(messageToHandle.ReceiverEntity);
315 KG_ASSERT(receiverEntity, "Invalid entity obtained inside AIService");
316
317
318 // Get ai component whose OnMessage() function needs to be called
319 ECS::AIStateComponent& receiverAIComponent = receiverEntity.GetComponent<ECS::AIStateComponent>();
320
321 // Call OnMessage for recipient's global state
322 if (receiverAIComponent.GlobalStateReference && receiverAIComponent.GlobalStateReference->OnMessage)
323 {
324 Utility::CallWrappedVoidUInt32EntityEntityFloat(receiverAIComponent.GlobalStateReference->OnMessage->m_Function, messageToHandle.MessageType, messageToHandle.SenderEntity, messageToHandle.ReceiverEntity, messageToHandle.DispatchTime);
325 }
326
327 // Call OnMessage for recipient's current state
328 if (receiverAIComponent.CurrentStateReference && receiverAIComponent.CurrentStateReference->OnMessage)
329 {
330 Utility::CallWrappedVoidUInt32EntityEntityFloat(receiverAIComponent.CurrentStateReference->OnMessage->m_Function, messageToHandle.MessageType, messageToHandle.SenderEntity, messageToHandle.ReceiverEntity, messageToHandle.DispatchTime);
331 }

333 void AIService::HandleDelayedMessages()
334 {
335 float currentTime = Utility::Time::GetTime();
336 auto& messageQueue = s_AIContext->MessageQueue;
337
338 // Loop through messageQueue and process messages if they are due for dispatch
339 while (!messageQueue.empty())
340 {
341 // Get the current top message
342 const AIMessage& currentMessage = messageQueue.top();
343
344 // Check if current message is due for dispatch
345 if (currentMessage.DispatchTime <= currentTime)
346 {
347 // Remove message from queue
348 messageQueue.pop();
349
350 // Handle message
351 HandleAIMessage(std::move(currentMessage));
352
353 }
354
355 // Exit if current message (and remainder of queue) is still waiting...
356 else
357 {
358 break;
359 }
360 }
361 }
