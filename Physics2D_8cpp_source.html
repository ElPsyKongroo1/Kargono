61 b2BodyUserData& bodyUser = body->GetUserData();

63 rb2d.RuntimeBody = body;

123 void Physics2DService::OnUpdate(Timestep ts)
124 {
125 const int32_t velocityIterations = 6;
126 const int32_t positionIterations = 2;
127 s_ActivePhysicsWorld->m_PhysicsWorld->Step(ts, velocityIterations, positionIterations);
128
129 // Retrieve transform from Box2D
130 auto view = s_ActivePhysicsWorld->m_Scene->GetAllEntitiesWith<ECS::Rigidbody2DComponent>();
131 for (auto enttID : view)
132 {
133 ECS::Entity entity = s_ActivePhysicsWorld->m_Scene->GetEntityByEnttID(enttID);
134 auto& transform = entity.GetComponent<ECS::TransformComponent>();
135 auto& rb2d = entity.GetComponent<ECS::Rigidbody2DComponent>();
136
137 b2Body* body = (b2Body*)rb2d.RuntimeBody;
138 const auto& position = body->GetPosition();
139 transform.Translation.x = position.x;
140 transform.Translation.y = position.y;
141 transform.Rotation.z = body->GetAngle();
142 // TODO FOR DEBUGGING
143 KG_ASSERT(!std::isnan(position.x) && !std::isnan(position.y) && !std::isnan(body->GetAngle()));

147 RaycastResult Physics2DService::Raycast(Math::vec2 startPoint, Math::vec2 endPoint)
148 {
149 RayCastCallback newCallback;
150
151 if (std::isnan(startPoint.x) || std::isnan(startPoint.y) || std::isnan(endPoint.y) || std::isnan(endPoint.y))
152 {
153 KG_WARN("A not a number float was found as input to a 2D raycast call!");
154 return RaycastResult(false, Assets::EmptyHandle);
155 }
156
157 s_ActivePhysicsWorld->m_PhysicsWorld->RayCast(&newCallback, b2Vec2(startPoint.x, startPoint.y), b2Vec2(endPoint.x, endPoint.y));
158 if (newCallback.m_Fixture)
159 {
160 return RaycastResult(true,
161 newCallback.m_Fixture->GetBody()->GetUserData().UUID,
162 { newCallback.m_NormalVector.x, newCallback.m_NormalVector.y },
163 { newCallback.m_ContactPoint.x, newCallback.m_ContactPoint.y });
164 }
165 else
166 {
167 return RaycastResult(false, Assets::EmptyHandle);
168 }
