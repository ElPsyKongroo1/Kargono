20 Ref<Scene> SceneService::CreateSceneCopy(Ref<Scene> other)
21 {
22 Ref<Scene> newScene = CreateRef<Scene>();
23 newScene->m_PhysicsSpecification = other->m_PhysicsSpecification;
24
25 auto& srcSceneRegistry = other->m_EntityRegistry.m_EnTTRegistry;
26 auto& dstSceneRegistry = newScene->m_EntityRegistry.m_EnTTRegistry;
27 std::unordered_map<UUID, entt::entity> enttMap;
28
29 // Create entities in new scene
30 auto idView = srcSceneRegistry.view<ECS::IDComponent>();
31 for (auto e : idView)
32 {
33 UUID uuid = srcSceneRegistry.get<ECS::IDComponent>(e).ID;
34 const auto& name = srcSceneRegistry.get<ECS::TagComponent>(e).Tag;
35 ECS::Entity newEntity = newScene->CreateEntityWithUUID(uuid, name);
36 enttMap[uuid] = (entt::entity)newEntity;
37 }
38
39 // Copy components (except IDComponent and TagComponent)
40 Utility::CopyComponent(ECS::AllComponents{}, dstSceneRegistry, srcSceneRegistry, enttMap);
41
42 // TODO: There is probably a faster way to get all the entities inside the scene without iterating through all entities
43 // TODO: I am just not aware of a view function using raw storage references in entt
44 // Transfer custom components into new scene
45 // Handle all project components
46 for (auto& [entityHandle, enttID] : other->m_EntityRegistry.m_EntityMap)
47 {
48 // Copy over data
49 ECS::Entity existingEntity{ other->GetEntityByUUID(entityHandle) };
50 ECS::Entity newEntity{ newScene->GetEntityByUUID(entityHandle) };
51 for (auto& [handle, asset] : Assets::AssetService::GetProjectComponentRegistry())
52 {
53 Ref<ECS::ProjectComponent> projectComponent = Assets::AssetService::GetProjectComponent(handle);
54 KG_ASSERT(projectComponent);
55
56 if (existingEntity.HasProjectComponentData(handle))
57 {
58 // Add project component into registry
59 newEntity.AddProjectComponentData(handle);
60
61 // Get source and destination data buffers
62 uint8_t* sourceDataPtr = (uint8_t*)existingEntity.GetProjectComponentData(handle);
63 uint8_t* destinationDataPtr = (uint8_t*)newEntity.GetProjectComponentData(handle);
64 std::memcpy(destinationDataPtr, sourceDataPtr, projectComponent->m_BufferSize);
65 }
66 }
67 }
68
69 return newScene;
70
71 }
