81 {
82 Ref<Scene> newScene = CreateRef<Scene>();

84
85 auto& srcSceneRegistry = other->m_Registry;

162 m_PhysicsWorld = CreateScope<Physics::Physics2DWorld>(this, m_PhysicsSpecification.Gravity);
163
164 // Invoke OnCreate
165 auto classInstanceView = GetAllEntitiesWith<ClassInstanceComponent>();
166 for (auto e : classInstanceView)

168 Scenes::Entity entity = { e, this };

170 Ref<EntityClass> entityClass = Assets::AssetManager::GetEntityClass(classInstanceComp.ClassHandle);

179 // Insert entities with ClassInstanceComponents into m_ScriptClassToEntityList map
180 auto view = GetAllEntitiesWith<ClassInstanceComponent>();

185 KG_ASSERT(classInstanceComp.ClassReference);

188 m_ScriptClassToEntityList.insert({ classInstanceComp.ClassReference->GetName(), {}});
