734  void ViewportPanel::OnOverlayRender()
735  {
736  if (s_EditorApp->m_SceneState == SceneState::Play)
737  {
738  Scenes::Entity camera = Scenes::Scene::GetActiveScene()->GetPrimaryCameraEntity();

744  Rendering::RenderingService::BeginScene(m_EditorCamera);

746 
747  if (s_EditorApp->m_ShowPhysicsColliders)
748  {
749  // Circle Colliders
750  {
751  auto view = Scenes::Scene::GetActiveScene()->GetAllEntitiesWith<Scenes::TransformComponent, Scenes::CircleCollider2DComponent>();
752  for (auto entity : view)
753  {
754  auto [tc, cc2d] = view.get<Scenes::TransformComponent, Scenes::CircleCollider2DComponent>(entity);
755 
756  Math::vec3 translation = tc.Translation + Math::vec3(cc2d.Offset.x, cc2d.Offset.y, 0.001f);
757 
758  Math::vec3 scale = tc.Scale * Math::vec3(cc2d.Radius * 2.0f);
759

761  * glm::scale(Math::mat4(1.0f), scale);

767  // Box Colliders
