732  void ViewportPanel::OnOverlayRender()
733  {
734  if (s_EditorApp->m_SceneState == SceneState::Play)
735  {
736  Entity camera = Scene::GetActiveScene()->GetPrimaryCameraEntity();
737  if (!camera) { return; }
738  Renderer::BeginScene(camera.GetComponent<CameraComponent>().Camera, glm::inverse(camera.GetComponent<TransformComponent>().GetTransform()));
739  }
740  else
741  {
742  Renderer::BeginScene(m_EditorCamera);
743  }
744 
745  if (s_EditorApp->m_ShowPhysicsColliders)
746  {
747  // Circle Colliders
748  {
749  auto view = Scene::GetActiveScene()->GetAllEntitiesWith<TransformComponent, CircleCollider2DComponent>();
750  for (auto entity : view)
751  {
752  auto [tc, cc2d] = view.get<TransformComponent, CircleCollider2DComponent>(entity);
753 
754  Math::vec3 translation = tc.Translation + Math::vec3(cc2d.Offset.x, cc2d.Offset.y, 0.001f);
755 
756  Math::vec3 scale = tc.Scale * Math::vec3(cc2d.Radius * 2.0f);
757 
758  Math::mat4 transform = glm::translate(Math::mat4(1.0f), translation)
759  * glm::scale(Math::mat4(1.0f), scale);
760 
761  s_CircleInputSpec.TransformMatrix = transform;
762  Renderer::SubmitDataToRenderer(s_CircleInputSpec);
763  }
764  }
765  // Box Colliders
766  {
767  auto view = Scene::GetActiveScene()->GetAllEntitiesWith<TransformComponent, BoxCollider2DComponent>();
768  for (auto entity : view)
769  {
770  auto [tc, bc2d] = view.get<TransformComponent, BoxCollider2DComponent>(entity);
771 
772  Math::vec3 translation = tc.Translation + Math::vec3(bc2d.Offset.x, bc2d.Offset.y, 0.001f);
773  Math::vec3 scale = tc.Scale * Math::vec3(bc2d.Size * 2.0f, 1.0f);
774

776  * glm::rotate(Math::mat4(1.0f), tc.Rotation.z, Math::vec3(0.0f, 0.0f, 1.0f))
