141 void RuntimeUIService::PushRenderData(const Math::mat4& cameraViewMatrix, uint32_t viewportWidth, uint32_t viewportHeight)
142 {
143 if (!s_ActiveUI)
144 {
145 return;
146 }
147
148 Rendering::RendererAPI::ClearDepthBuffer();
149 // Iterate through all characters
150 Math::mat4 orthographicProjection = glm::ortho((float)0, static_cast<float>(viewportWidth),
151 (float)0, static_cast<float>(viewportHeight), (float)-1, (float)1);
152 Math::mat4 outputMatrix = orthographicProjection;
153
154 Rendering::RenderingService::BeginScene(outputMatrix);
155
156 // Submit all windows
157 for (auto window : s_ActiveUI->m_DisplayedWindows)
158 {
159 Math::vec3 scale = Math::vec3(viewportWidth * window->Size.x, viewportHeight * window->Size.y, 1.0f);
160 Math::vec3 initialTranslation = Math::vec3((viewportWidth * window->ScreenPosition.x), (viewportHeight * window->ScreenPosition.y), window->ScreenPosition.z);
161 Math::vec3 translation = Math::vec3( initialTranslation.x + (scale.x / 2), initialTranslation.y + (scale.y / 2), initialTranslation.z);
162
163 s_BackgroundInputSpec.TransformMatrix = glm::translate(Math::mat4(1.0f), translation)
164 * glm::scale(Math::mat4(1.0f), scale);
165 Rendering::Shader::SetDataAtInputLocation<Math::vec4>(window->BackgroundColor, "a_Color", s_BackgroundInputSpec.Buffer, s_BackgroundInputSpec.Shader);
166
167 Rendering::RenderingService::SubmitDataToRenderer(s_BackgroundInputSpec);
168
169 initialTranslation.z += 0.001f;
170 for (auto& widget : window->Widgets)
171 {
172 widget->PushRenderData(initialTranslation, scale, static_cast<float>(viewportWidth));
173 }
174 }
175
176 Rendering::RenderingService::EndScene();

450 {
451 for (auto& window : s_ActiveUI->m_Windows)
452 {
453 for (auto& currentWidget : window.Widgets)
454 {
455 Ref<Widget> currentBestChoice;
456 uint32_t currentChoiceLocation;
457 float currentBestDistance;
458 uint32_t iteration;
459
460
461 // Calculate Right
462 currentBestChoice = nullptr;
463 currentChoiceLocation = 0;
464 currentBestDistance = std::numeric_limits<float>::max();
465 iteration = 0;
466
467 for (auto& potentialChoice : window.Widgets)
468 {
469 if (potentialChoice == currentWidget || !potentialChoice->Selectable)
470 {
471 iteration++;
472 continue;
473 }
474
475 float currentDistance = glm::abs(glm::distance(potentialChoice->WindowPosition, currentWidget->WindowPosition));
476
477 // Constraints ----------------
478 float currentWidgetRightExtent = currentWidget->WindowPosition.x + (currentWidget->Size.x / 2);
479 float potentialWidgetLeftExtent = potentialChoice->WindowPosition.x - (potentialChoice->Size.x / 2);
480 if (currentWidgetRightExtent >= potentialWidgetLeftExtent)
481 {
482 iteration++;
483 continue;
484 }
485
486 // Preferences ----------------
487 if (currentBestChoice == nullptr)
488 {
489 currentBestChoice = potentialChoice;
490 currentChoiceLocation = iteration;
491 currentBestDistance = currentDistance;
492 iteration++;
493 continue;
494 }
495
496
497 // Prefer Lowest Distance Widget
498 if (currentDistance < currentBestDistance)
499 {
500 currentBestChoice = potentialChoice;
501 currentChoiceLocation = iteration;
502 currentBestDistance = currentDistance;
503 iteration++;
504 continue;
505 }
506
507 iteration++;
508 }
509 if (currentBestChoice)
510 {
511 currentWidget->DirectionPointer.Right = currentChoiceLocation;
