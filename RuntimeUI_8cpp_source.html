30 void RuntimeUIService::Init()
31 {

33 s_ActiveUIHandle = Assets::EmptyHandle;
34 s_DefaultFont = FontService::InstantiateEditorFont("Resources/fonts/arial.ttf");

46 shapeComp->Indices = CreateRef<std::vector<uint32_t>>(Rendering::Shape::s_Quad.GetIndices());
47
48 s_BackgroundInputSpec.Shader = localShader;
49 s_BackgroundInputSpec.Buffer = localBuffer;
50 s_BackgroundInputSpec.ShapeComponent = shapeComp;
51 }
52 KG_VERIFY(true, "Runtime UI Engine Init");

147 void RuntimeUIService::PushRenderData(const Math::mat4& cameraViewMatrix, uint32_t viewportWidth, uint32_t viewportHeight)
148 {
149 if (!s_ActiveUI)
150 {
151 return;
152 }
153
154 Rendering::RendererAPI::ClearDepthBuffer();
155 // Iterate through all characters
156 Math::mat4 orthographicProjection = glm::ortho((float)0, static_cast<float>(viewportWidth),
157 (float)0, static_cast<float>(viewportHeight), (float)-1, (float)1);
158 Math::mat4 outputMatrix = orthographicProjection;
159
160 /*Math::vec2 windowLocation = Math::vec2(0.3f, 0.2f);
161 Math::vec2 windowSize = Math::vec2(0.2f, 0.2f);*/
162
163 Rendering::RenderingService::BeginScene(outputMatrix);
164
165 // Submit all windows
166 for (auto window : s_ActiveUI->m_DisplayedWindows)
167 {
168 Math::vec3 scale = Math::vec3(viewportWidth * window->Size.x, viewportHeight * window->Size.y, 1.0f);
169 Math::vec3 initialTranslation = Math::vec3((viewportWidth * window->ScreenPosition.x), (viewportHeight * window->ScreenPosition.y), window->ScreenPosition.z);
170 Math::vec3 translation = Math::vec3( initialTranslation.x + (scale.x / 2), initialTranslation.y + (scale.y / 2), initialTranslation.z);
171
172 s_BackgroundInputSpec.TransformMatrix = glm::translate(Math::mat4(1.0f), translation)
173 * glm::scale(Math::mat4(1.0f), scale);
174 Rendering::Shader::SetDataAtInputLocation<Math::vec4>(window->BackgroundColor, "a_Color", s_BackgroundInputSpec.Buffer, s_BackgroundInputSpec.Shader);
175
176 Rendering::RenderingService::SubmitDataToRenderer(s_BackgroundInputSpec);
177
178 initialTranslation.z += 0.001f;
179 for (auto& widget : window->Widgets)
180 {
181 widget->PushRenderData(initialTranslation, scale, static_cast<float>(viewportWidth));
182 }
183 }
184
185 Rendering::RenderingService::EndScene();
186

487 void RuntimeUIService::CalculateDirections()
488 {
489 for (auto& window : s_ActiveUI->m_Windows)
490 {
491 for (auto& currentWidget : window.Widgets)
492 {
493 Ref<Widget> currentBestChoice;
494 uint32_t currentChoiceLocation;
495 float currentBestDistance;
496 uint32_t iteration;
497
498
499 // Calculate Right
500 currentBestChoice = nullptr;
501 currentChoiceLocation = 0;
502 currentBestDistance = std::numeric_limits<float>::max();
503 iteration = 0;
504
505 for (auto& potentialChoice : window.Widgets)
506 {
507 if (potentialChoice == currentWidget || !potentialChoice->Selectable)
508 {
509 iteration++;
510 continue;
511 }
512
513 float currentDistance = glm::abs(glm::distance(potentialChoice->WindowPosition, currentWidget->WindowPosition));
514
515 // Constraints ----------------
516 float currentWidgetRightExtent = currentWidget->WindowPosition.x + (currentWidget->Size.x / 2);
517 float potentialWidgetLeftExtent = potentialChoice->WindowPosition.x - (potentialChoice->Size.x / 2);
518 if (currentWidgetRightExtent >= potentialWidgetLeftExtent)
519 {
520 iteration++;
521 continue;
522 }
523
524 // Preferences ----------------
525 if (currentBestChoice == nullptr)
526 {
527 currentBestChoice = potentialChoice;
528 currentChoiceLocation = iteration;
529 currentBestDistance = currentDistance;
530 iteration++;
531 continue;
532 }
533
534
535 // Prefer Lowest Distance Widget
536 if (currentDistance < currentBestDistance)
537 {
538 currentBestChoice = potentialChoice;
539 currentChoiceLocation = iteration;
540 currentBestDistance = currentDistance;
541 iteration++;
542 continue;
543 }
544
545 iteration++;
546 }
547 if (currentBestChoice)
548 {
549 currentWidget->DirectionPointer.Right = currentChoiceLocation;
550 }
551 else
552 {
553 currentWidget->DirectionPointer.Right = -1;
554 }
555
556 // Calculate Left
557 currentBestChoice = nullptr;
