513 static bool AddAIMessageType(const std::string& newMessageType)
514 {
515 // Check if message already exists
516 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
517 {
518 // Return false if message already exists
519 if (aiMessage == newMessageType)
520 {
521 return false;
522 }
523 }
524
525 // Add AIMessageType to project if none is found
526 s_ActiveProject->AllMessageTypes.push_back(newMessageType);
527 return true;
528 }

530 static bool EditAIMessageType(const std::string& oldMessageType, const std::string& newMessageType)
531 {
532 constexpr size_t k_InvalidPosition{ std::numeric_limits<size_t>().max() };
533
534 // Find the indicated message
535 size_t iteration{ 0 };
536 size_t foundMessagePosition{ std::numeric_limits<size_t>().max() };
537 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
538 {
539 // If we find the message type, note its location
540 if (aiMessage == oldMessageType)
541 {
542 foundMessagePosition = iteration;
543 }
544 iteration++;
545 }
546
547 // Could not find ai message to replace
548 if (foundMessagePosition == k_InvalidPosition)
549 {
550 KG_WARN("Could not find ai message type to replace when editing type");
551 return false;
552 }
553
554 // Replace text at specified position
555 s_ActiveProject->AllMessageTypes.at(foundMessagePosition) = newMessageType;
556 return true;
557 }
558
559 static bool DeleteAIMessageType(const std::string& aiMessageType)
560 {
561 constexpr size_t k_InvalidPosition{ std::numeric_limits<size_t>().max() };
562
563 // Find the indicated message
564 size_t iteration{ 0 };
565 size_t foundMessagePosition{ std::numeric_limits<size_t>().max() };
566 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
567 {
568 // If we find the message type, note its location
569 if (aiMessage == aiMessageType)
570 {
571 foundMessagePosition = iteration;
572 }
573 iteration++;
574 }
575
576 // Could not find ai message to replace
577 if (foundMessagePosition == k_InvalidPosition)
578 {
579 KG_WARN("Could not find ai message type to replace when editing type");
580 return false;
581 }
582
583 // Erase text at position
584 s_ActiveProject->AllMessageTypes.erase(s_ActiveProject->AllMessageTypes.begin() + foundMessagePosition);
585 return true;
586 }
