530  static bool EditAIMessageType(const std::string& oldMessageType, const std::string& newMessageType)
531  {
532  constexpr size_t k_InvalidPosition{ std::numeric_limits<size_t>().max() };
533 
534  // Find the indicated message
535  size_t iteration{ 0 };
536  size_t foundMessagePosition{ std::numeric_limits<size_t>().max() };
537  for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
538  {
539  // If we find the message type, note its location
540  if (aiMessage == oldMessageType)
541  {
542  foundMessagePosition = iteration;
543  }
544  iteration++;
545  }
546 
547  // Could not find ai message to replace
548  if (foundMessagePosition == k_InvalidPosition)
549  {
550  KG_WARN("Could not find ai message type to replace when editing type");
551  return false;
552  }
553 
554  // Replace text at specified position
555  s_ActiveProject->AllMessageTypes.at(foundMessagePosition) = newMessageType;
556  return true;
