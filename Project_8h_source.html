502 static bool AddAIMessageType(const std::string& newMessageType)
503 {
504 // Check if message already exists
505 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
506 {
507 // Return false if message already exists
508 if (aiMessage == newMessageType)
509 {
510 return false;
511 }
512 }
513
514 // Add AIMessageType to project if none is found
515 s_ActiveProject->AllMessageTypes.push_back(newMessageType);
516 return true;
517 }

519 static bool EditAIMessageType(const std::string& oldMessageType, const std::string& newMessageType)
520 {
521 constexpr size_t k_InvalidPosition{ std::numeric_limits<size_t>().max() };
522
523 // Find the indicated message
524 size_t iteration{ 0 };
525 size_t foundMessagePosition{ std::numeric_limits<size_t>().max() };
526 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
527 {
528 // If we find the message type, note its location
529 if (aiMessage == oldMessageType)
530 {
531 foundMessagePosition = iteration;
532 }
533 iteration++;
534 }
535
536 // Could not find ai message to replace
537 if (foundMessagePosition == k_InvalidPosition)
538 {
539 KG_WARN("Could not find ai message type to replace when editing type");
540 return false;
541 }
542
543 // Replace text at specified position
544 s_ActiveProject->AllMessageTypes.at(foundMessagePosition) = newMessageType;
545 return true;
546 }
547
548 static bool DeleteAIMessageType(const std::string& aiMessageType)
549 {
550 constexpr size_t k_InvalidPosition{ std::numeric_limits<size_t>().max() };
551
552 // Find the indicated message
553 size_t iteration{ 0 };
554 size_t foundMessagePosition{ std::numeric_limits<size_t>().max() };
555 for (std::string& aiMessage : s_ActiveProject->AllMessageTypes)
556 {
557 // If we find the message type, note its location
558 if (aiMessage == aiMessageType)
559 {
560 foundMessagePosition = iteration;
561 }
562 iteration++;
563 }
564
565 // Could not find ai message to replace
566 if (foundMessagePosition == k_InvalidPosition)
567 {
568 KG_WARN("Could not find ai message type to replace when editing type");
569 return false;
570 }
571
572 // Erase text at position
573 s_ActiveProject->AllMessageTypes.erase(s_ActiveProject->AllMessageTypes.begin() + foundMessagePosition);
574 return true;
575 }
