132 KG_ASSERT(s_ActiveProject, "Failed to build runtime executable since no active project exists");

136 // Ensure all relevant directories have been created

138 Utility::FileSystem::CreateNewDirectory(projectPath / "Editor/Log");
139
140 // Check if the project path exists
141 if (!Utility::FileSystem::PathExists(projectPath) || !Utility::FileSystem::IsRegularFile(projectPath / "Makefile"))
142 {
143 KG_WARN("Failed to build executable. Could not locate/invalid Makefile");
144 return false;
145 }
146
147 std::stringstream outputStream {};
148
149 // Group commands together
150 outputStream << "{ ";
151
152 // Change directory to the project path
153 outputStream << "cd \"" << projectPath.string() << "\" && ";
154
155 // Set environment variables or build options
156 if (createServer)
157 {
158 outputStream << "export ExternalCompilerOptions=KG_EXPORT_SERVER && ";
159 }
160 else
161 {
162 outputStream << "export ExternalCompilerOptions=KG_EXPORT_RUNTIME && ";
163 }
164
165 // Specify make command
166 outputStream << "make ";
167
168 // Specify target executable
169 if (createServer)
170 {
171 outputStream << "Server ";
172 }
173 else
174 {
175 outputStream << "Runtime ";
176 }
177
178 // Specify the output directory and configuration
179 outputStream << "TARGETDIR=\"" << intermediatesPath.string() << "\" ";
180 outputStream << "OBJDIR=\"" << intermediatesPath.string() << "\" ";
181 outputStream << "CONFIG=Dist; ";
182
183 // Group commands together
184 outputStream << " } ";
185
186 // Redirect output to log files
187 if (createServer)
188 {
189 outputStream << ">> ./Log/BuildServerExecutable.log 2>&1 ";
190 }
191 else
192 {
193 outputStream << ">> ./Log/BuildRuntimeExecutable.log 2>&1 ";
194 }
195 // Execute the build command
196 bool success = system(outputStream.str().c_str()) == 0;
197
198 if (!success)
199 {
200 KG_WARN("Command line operation returned error code");
201 return false;
202 }
203
204 // Move Executable into main directory
205 if (createServer)

217 return false;
218 }
