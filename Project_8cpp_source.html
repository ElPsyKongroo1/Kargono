190 bool ProjectService::BuildRuntimeExecutable(const std::filesystem::path& exportDirectory, bool createServer)
191 {
192 KG_ASSERT(s_ActiveProject, "Failed to build runtime executable since no active project exists");
193 std::filesystem::path solutionPath { std::filesystem::current_path().parent_path() / "Kargono.sln" };
194 std::filesystem::path intermediatesPath { exportDirectory / "Temp/" };
195
196 // Check if the sourceFile path exists and is a regular file
197 if (!Utility::FileSystem::PathExists(solutionPath) || !Utility::FileSystem::IsRegularFile(solutionPath))
198 {
199 KG_WARN("Failed to build executable. Could not locate/invalid Kargono.sln file");
200 return false;
201 }
202
203 std::stringstream outputStream {};
204 outputStream << "("; // Parentheses to group all function calls together
205 // Access visual studio toolset console
206 outputStream << "\"C:\\Program Files\\Microsoft Visual Studio\\2022\\Community\\VC\\Auxiliary\\Build\\vcvars64.bat\"";
207
208 if (createServer)
209 {
210 outputStream << " && set ExternalCompilerOptions=KG_EXPORT_SERVER ";
211 }
212 else
213 {
214 outputStream << " && set ExternalCompilerOptions=KG_EXPORT_RUNTIME ";
215 }
216
217 // Start Next Command
218 outputStream << " && "; // Combine commands
219
220 // Cl command for compiling binary code
221 outputStream << "msbuild "; // Add Command
222
223 outputStream << "\"" << solutionPath.string() << "\" "; // Provide path to solution
224 if (createServer)
225 {
226 outputStream << "-t:Engine\\Engine,Applications\\Server "; // Specify Runtime project to be built
227 }
228 else
229 {
230 outputStream << "-t:Engine\\Engine,Applications\\Runtime "; // Specify Runtime project to be built
231 }
232
233 outputStream << "-p:OutDir=" << "\"" << intermediatesPath.string() << "/\" "; // Provide path to place executable
234
235 // Specify the intermediate output directory
236 outputStream << "-p:IntermediateOutputPath=" << "\"" << intermediatesPath.string() << "\" "; // Provide intermediate directory path
237
238 outputStream << "-p:Configuration=Dist "; // Run Dist configuration
239
240 // Specify the preprocessor define(s)
241 if (createServer)
242 {
243 outputStream << "-p:KG_EXPORT_SERVER=1" << " "; // Add KG_EXPORT define
244 }
245 else
246 {
247 outputStream << "-p:KG_EXPORT_RUNTIME=1" << " "; // Add KG_EXPORT define
248 }
249
250 outputStream << ")"; // Parentheses to group all function calls together
251
252 // Sends all three calls (open dev console, compiler, and linker) error/info to log file
253 if (createServer)
254 {
255 outputStream << " >> Log\\BuildServerExecutable.log 2>&1 ";
256 }
257 else
258 {
259 outputStream << " >> Log\\BuildRuntimeExecutable.log 2>&1 ";
260 }
261
262
263 // Call Command
264 bool success = system(outputStream.str().c_str()) != 1;
265
266 if (!success)
267 {
268 KG_WARN("Command line operation returned error code 1");
269 return false;
270 }
271
272 // Move Executable into main directory
273 if (createServer)
274 {
275 success = Utility::FileSystem::CopySingleFile(intermediatesPath / "Server.exe", exportDirectory / (s_ActiveProject->Name + "Server.exe"));
276 }
277 else
278 {
279 success = Utility::FileSystem::CopySingleFile(intermediatesPath / "Runtime.exe", exportDirectory / (s_ActiveProject->Name + ".exe"));
280 }
281
282 if (!success)
283 {
284 KG_WARN("Failed to copy executable file into export directory");
285 return false;
286 }
287
288 // Remove Intermediates
289 Utility::FileSystem::DeleteSelectedDirectory(intermediatesPath);
290
291 return true;
