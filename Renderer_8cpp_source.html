254  {

267  buffer->Shader->GetVertexArray()->GetVertexBuffers().at(0)->SetData(buffer->VertexBuffer.Data, dataSize);
268

271  {
272  uniformFunction(buffer);
273  }
274

277  {
278  drawFunction(buffer);
279  }
280

282  {
283  postDrawFunction(buffer);

286  }
