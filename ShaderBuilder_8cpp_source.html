541 std::tuple<ShaderSource, InputBufferLayout, UniformBufferList> ShaderBuilder::BuildShader(const ShaderSpecification& shaderSpec)
542 {
543 uint16_t countBuffer = 0;
544
545 //=================
546 // Reset Static Resources
547 //=================
548
549 // Clear static Output
550 s_OutputStream.str("");
551 s_InputBufferLayout.Clear();
552 s_UniformBufferLayout.Clear();
553
554 // Clear All Lambda Maps
555 s_VertexStructs.clear();
556 s_VertexInput.clear();
557 s_VertexToFragmentSmooth.clear();
558 s_VertexToFragmentFlat.clear();
559 s_VertexUniforms.clear();
560 s_VertexMain.clear();
561 s_FragmentStructs.clear();
562 s_FragmentFunctions.clear();
563 s_FragmentOutput.clear();
564 s_FragmentUniforms.clear();
565 s_FragmentMain.clear();
566
567 //=================
568 // Set up Lambda Maps based on shaderSpec
569 //=================
570
571 // Core
572 AddSimpleVertexOutput();
573 AddSimpleFragmentOutput();
574
575 if (shaderSpec.ColorInput == ColorInputType::FlatColor || shaderSpec.ColorInput == ColorInputType::VertexColor)
576 {
577 AddColorOutput();
578 }
579 if (shaderSpec.AddProjectionMatrix) { AddProjectionMatrix(); }
580 if (shaderSpec.AddEntityID) { AddEntityID(); }
581 if (shaderSpec.AddCircleShape) { AddCircleShape(); }
582 if (shaderSpec.TextureInput == TextureInputType::ColorTexture) { AddTextureOutput(); }
583 if (shaderSpec.TextureInput == TextureInputType::TextTexture) { AddTextTextureOutput(); }
584
585 //=================
586 // Start Building Shader
587 //=================
588
589 // ==== Vertex Shader ====
