153  inline Panels::BrowserFileType DetermineFileType(const std::filesystem::path& entry)
154  {
155  if (std::filesystem::is_directory(entry)) { return Panels::BrowserFileType::Directory; }
156  if (!entry.has_extension()) { return Panels::BrowserFileType::None; }
157  std::filesystem::path currentFileExtension = entry.extension();
158 
159  // Check for raw texture extension
160  for (const std::string& textureExtension : Assets::AssetService::GetTexture2DValidImportExtensions())
161  {
162  if (textureExtension == currentFileExtension)
163  {
164  return Panels::BrowserFileType::RawTexture;
165  }
166  }
167 
168  // Check for raw audio extension
169  for (const std::string& audioExtension : Assets::AssetService::GetAudioBufferValidImportExtensions())
170  {
171  if (audioExtension == currentFileExtension)
172  {
173  return Panels::BrowserFileType::RawAudio;
174  }
175  }
176  // Check for raw font extension
177  for (const std::string& fontExtension : Assets::AssetService::GetFontValidImportExtensions())
178  {
179  if (fontExtension == currentFileExtension)
180  {
181  return Panels::BrowserFileType::RawFont;
182  }
183  }
184  if (currentFileExtension == ".kgaistate") { return Panels::BrowserFileType::AIState; }
185  if (currentFileExtension == ".kgaudio") { return Panels::BrowserFileType::Audio; }
186  if (currentFileExtension == ".kgfont") { return Panels::BrowserFileType::Font; }
187  if (currentFileExtension == ".kgcomponent") { return Panels::BrowserFileType::ProjectComponent; }
188  if (currentFileExtension == ".kgreg") { return Panels::BrowserFileType::Registry; }
189  if (currentFileExtension == ".kgui") { return Panels::BrowserFileType::UserInterface; }
190  if (currentFileExtension == ".kgscript") { return Panels::BrowserFileType::Script; }
191  if (currentFileExtension == ".kgstate") { return Panels::BrowserFileType::GameState; }
192  if (currentFileExtension == ".kgscene") { return Panels::BrowserFileType::Scene; }
193  if (currentFileExtension == ".kgtexture") { return Panels::BrowserFileType::Texture; }
194  if (currentFileExtension == ".kginput") { return Panels::BrowserFileType::InputMap; }
195  if (currentFileExtension == ".kgbinary" || currentFileExtension == ".kgshadervert" ||
196  currentFileExtension == ".kgshaderfrag")
197  {
198  return Panels::BrowserFileType::Binary;
199  }
200  return Panels::BrowserFileType::GenericFile;
201  }
