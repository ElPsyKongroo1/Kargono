78 inline std::string GenerateFunctionSignature(WrappedFuncType funcType, const std::string& name)
79 {
80 WrappedVarType returnType = Utility::WrappedFuncTypeToReturnType(funcType);
81 std::vector<WrappedVarType> parameterTypes = Utility::WrappedFuncTypeToParameterTypes(funcType);
82 // Write out return value and function name
83 std::stringstream outputStream {};
84 outputStream << Utility::WrappedVarTypeToCPPString(returnType) << " " << name<< "(";
85
86 // Write out parameters into function signature
87 char letterIteration{ 'a' };
88 for (uint32_t iteration{ 0 }; static_cast<size_t>(iteration) < parameterTypes.size(); iteration++)
89 {
90 outputStream << Utility::WrappedVarTypeToCPPString(parameterTypes.at(iteration)) << " " << letterIteration;
91 if (iteration != parameterTypes.size() - 1)
92 {
93 outputStream << ',';
94 }
95 letterIteration++;
96 }
97 outputStream << ")";
98 return outputStream.str();
99 }
