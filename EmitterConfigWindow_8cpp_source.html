402 bool EmitterConfigWindow::OnAssetEvent(Events::Event* event)
403 {
404 Events::ManageAsset* manageAsset = (Events::ManageAsset*)event;
405
406 // Handle emitter config deletion
407 if (manageAsset->GetAssetType() == Assets::AssetType::EmitterConfig &&
408 manageAsset->GetAction() == Events::ManageAssetAction::PreDelete)
409 {
410 // Check if editor needs modification
411 if (manageAsset->GetAssetID() != m_EditorEmitterConfigHandle)
412 {
413 return false;
414 }
415
416 // Handle deletion of asset
417 ResetWindowResources();
418 return true;
419 }
420
421 // Handle emitter config name change
422 if (manageAsset->GetAssetType() == Assets::AssetType::EmitterConfig &&
423 manageAsset->GetAction() == Events::ManageAssetAction::UpdateAssetInfo)
424 {
425 // Check if editor needs modification
426 if (manageAsset->GetAssetID() != m_EditorEmitterConfigHandle)
427 {
428 return false;
429 }
430
431 // Update header name with new asset name
432 m_MainHeader.m_Label = Assets::AssetService::GetEmitterConfigFileLocation(manageAsset->GetAssetID()).filename().string();
433 return true;
434 }
435
436 return false;

439 void EmitterConfigWindow::OpenAssetInEditor(std::filesystem::path& assetLocation)
440 {
441 // Ensure provided path is within the active asset directory
442 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
443 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
444 {
445 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
446 return;
447 }
448
449 // Look for asset in registry using the file location
450 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
451 Assets::AssetHandle assetHandle = Assets::AssetService::GetEmitterConfigHandleFromFileLocation(relativePath);
452
453 // Validate resulting handle
454 if (!assetHandle)
455 {
456 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
457 return;
458 }
459
460 // Early out if asset is already open
461 if (m_EditorEmitterConfigHandle == assetHandle)
462 {
463 // Open the emitter config window in the editor
464 EngineService::SubmitToMainThread([]()
465 {
466 s_EditorApp->SetActiveEditorWindow(ActiveEditorUIWindow::EmitterConfigWindow);
467 s_EditorApp->m_EmitterConfigEditorWindow->LoadEditorEmitterIntoParticleService();
468 });
469 return;
470 }
471
472 // Check if panel is already occupied by an asset
473 if (!m_EditorEmitterConfig)
474 {
475 // Open emitter config in editor
476 OnOpenEmitterConfig(assetHandle);
477
478 // Open the emitter config window in the editor
479 EngineService::SubmitToMainThread([]()
480 {
481 s_EditorApp->SetActiveEditorWindow(ActiveEditorUIWindow::EmitterConfigWindow);
482 });
483 }
484 else
485 {
486 // Add warning to close active AI state before opening a new AIState
487 s_MainWindow->OpenWarningMessage("An emitter config is already active inside the editor. Please close the current emitter config before opening a new one.");
488 }
489 }
