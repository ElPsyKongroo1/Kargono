81 {

83 const auto& audioRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Audio/Intermediates/AudioRegistry.kgreg";

102 if (asset.Data.Type == Assets::AssetType::Audio)

104 Assets::AudioMetaData* metadata = static_cast<Assets::AudioMetaData*>(asset.Data.SpecificFileData.get());

106 out << YAML::Key << "SampleRate" << YAML::Value << metadata->SampleRate;
107 out << YAML::Key << "TotalPcmFrameCount" << YAML::Value << metadata->TotalPcmFrameCount;

113 out << YAML::EndMap; // Asset Map

124 AssetHandle AssetManager::ImportNewAudioFromFile(const std::filesystem::path& filePath)
125 {
126 // Create Checksum
127 const std::string currentCheckSum = Utility::FileSystem::ChecksumFromFile(filePath);
128
129 if (currentCheckSum.empty())
130 {
131 KG_ERROR("Failed to generate checksum from file!");
132 return {};
133 }
134
135 for (const auto& [handle, asset] : s_AudioRegistry)
136 {
137 if (asset.Data.CheckSum == currentCheckSum)
138 {
139 KG_INFO("Attempt to instantiate duplicate audio asset");
140 return handle;
141 }
142 }
143 if (filePath.extension().string() != ".wav")
144 {
145 KG_ERROR("Invalid file extension. Can only process .wav filetypes.");
146 return {};
147 }
148
149 // Create New Asset/Handle
150 AssetHandle newHandle{};
151
152 Assets::Asset newAsset{};
153 newAsset.Handle = newHandle;
154
155 // Create Intermediate
156 CreateAudioIntermediateFromFile(filePath, newAsset);
157 newAsset.Data.CheckSum = currentCheckSum;
158
159 // Register New Asset and Create Texture
160 s_AudioRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
161 SerializeAudioRegistry(); // Update Registry File on Disk
162
163 s_Audio.insert({ newHandle, InstantiateAudioIntoMemory(newAsset) });
164
165 return newHandle;

222 void AssetManager::CreateAudioIntermediateFromFile(const std::filesystem::path& filePath, Assets::Asset& newAsset)
223 {
224 // Create Buffers
225 uint32_t channels = 0;
226 uint32_t sampleRate = 0;
227 uint64_t totalPcmFrameCount = 0;
228 uint64_t totalSize = 0;
229 Buffer pcmData{};
230 drwav_int16* pSampleData = drwav_open_file_and_read_pcm_frames_s16(filePath.string().c_str(), &channels, &sampleRate, &totalPcmFrameCount, nullptr);
231 if (!pSampleData)
232 {
233 KG_ERROR("Failed to load audio file");
234 drwav_free(pSampleData, nullptr);
235 return;
236 }
237 totalSize = totalPcmFrameCount * channels * 2;
238 if ((totalSize) > drwav_uint64(std::numeric_limits<size_t>::max()))
239 {
240 KG_ERROR("Too much data in file for 32bit addressed vector");
241 drwav_free(pSampleData, nullptr);
242 return;
243 }
244 pcmData.Allocate(size_t(totalSize));
245 std::memcpy(pcmData.Data, pSampleData, pcmData.Size /*two bytes in s16*/);
246 drwav_free(pSampleData, nullptr);
247
248 // Save Binary Intermediate into File
249 std::string intermediatePath = "Audio/Intermediates/" + (std::string)newAsset.Handle + ".kgaudio";
250 std::filesystem::path intermediateFullPath = Projects::ProjectService::GetActiveAssetDirectory() / intermediatePath;
251 Utility::FileSystem::WriteFileBinary(intermediateFullPath, pcmData);
252
253 // Check that save was successful
254 if (!pcmData)
255 {
256 KG_ERROR("Failed to load data from file in audio importer!");
257 return;
258 }
259
260 // Load data into In-Memory Metadata object
261 newAsset.Data.Type = Assets::AssetType::Audio;
262 newAsset.Data.IntermediateLocation = intermediatePath;
263 Ref<Assets::AudioMetaData> metadata = CreateRef<Assets::AudioMetaData>();
264 metadata->Channels = channels;
265 metadata->SampleRate = sampleRate;;
266 metadata->TotalPcmFrameCount = totalPcmFrameCount;
267 metadata->TotalSize = totalSize;
268 metadata->InitialFileLocation = Utility::FileSystem::GetRelativePath(Projects::ProjectService::GetActiveAssetDirectory(), filePath);
269 newAsset.Data.SpecificFileData = metadata;
270 pcmData.Release();
271 }
