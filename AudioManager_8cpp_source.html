222  void AssetManager::CreateAudioIntermediateFromFile(const std::filesystem::path& filePath, Assets::Asset& newAsset)
223  {
224  // Create Buffers
225  uint32_t channels = 0;
226  uint32_t sampleRate = 0;
227  uint64_t totalPcmFrameCount = 0;
228  uint64_t totalSize = 0;
229  Buffer pcmData{};
230  drwav_int16* pSampleData = drwav_open_file_and_read_pcm_frames_s16(filePath.string().c_str(), &channels, &sampleRate, &totalPcmFrameCount, nullptr);
231  if (!pSampleData)
232  {
233  KG_ERROR("Failed to load audio file");
234  drwav_free(pSampleData, nullptr);
235  return;
236  }
237  totalSize = totalPcmFrameCount * channels * 2;
238  if ((totalSize) > drwav_uint64(std::numeric_limits<size_t>::max()))
239  {
240  KG_ERROR("Too much data in file for 32bit addressed vector");
241  drwav_free(pSampleData, nullptr);
242  return;
243  }
244  pcmData.Allocate(size_t(totalSize));
245  std::memcpy(pcmData.Data, pSampleData, pcmData.Size /*two bytes in s16*/);
246  drwav_free(pSampleData, nullptr);
247 
248  // Save Binary Intermediate into File
249  std::string intermediatePath = "Audio/Intermediates/" + (std::string)newAsset.Handle + ".kgaudio";
250  std::filesystem::path intermediateFullPath = Projects::Project::GetAssetDirectory() / intermediatePath;
251  Utility::FileSystem::WriteFileBinary(intermediateFullPath, pcmData);
252 
253  // Check that save was successful
254  if (!pcmData)
255  {
256  KG_ERROR("Failed to load data from file in audio importer!");
257  return;
258  }
259 
260  // Load data into In-Memory Metadata object
261  newAsset.Data.Type = Assets::AssetType::Audio;
262  newAsset.Data.IntermediateLocation = intermediatePath;
263  Ref<Assets::AudioMetaData> metadata = CreateRef<Assets::AudioMetaData>();
264  metadata->Channels = channels;
265  metadata->SampleRate = sampleRate;;
266  metadata->TotalPcmFrameCount = totalPcmFrameCount;
267  metadata->TotalSize = totalSize;
268  metadata->InitialFileLocation = Utility::FileSystem::GetRelativePath(Projects::Project::GetAssetDirectory(), filePath);
269  newAsset.Data.SpecificFileData = metadata;
270  pcmData.Release();
271  }
