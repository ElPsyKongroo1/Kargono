13 static void SerializeWrappedVarType(YAML::Emitter& out, WrappedVarType type, const std::string& name, void* dataSource)
14 {
15 switch (type)
16 {
17 case WrappedVarType::Integer32:
18 out << YAML::Key << name << YAML::Value << *(int32_t*)dataSource;
19 return;
20 case WrappedVarType::UInteger16:
21 out << YAML::Key << name << YAML::Value << *(uint16_t*)dataSource;
22 return;
23 case WrappedVarType::UInteger32:
24 out << YAML::Key << name << YAML::Value << *(uint32_t*)dataSource;
25 return;
26 case WrappedVarType::UInteger64:
27 out << YAML::Key << name << YAML::Value << *(uint64_t*)dataSource;
28 return;
29 case WrappedVarType::Vector3:
30 out << YAML::Key << name << YAML::Value << *(Math::vec3*)dataSource;
31 return;
32 case WrappedVarType::String:
33 out << YAML::Key << name << YAML::Value << *(std::string*)dataSource;
34 return;
35 case WrappedVarType::Bool:
36 out << YAML::Key << name << YAML::Value << *(bool*)dataSource;
37 return;
38 case WrappedVarType::Float:
39 out << YAML::Key << name << YAML::Value << *(float*)dataSource;
40 return;
41 case WrappedVarType::None:
42 case WrappedVarType::Void:
43 KG_ERROR("Invalid type provided while serializing data");
44 return;
45 }
46 KG_ERROR("Unknown Type of WrappedVariableType.");
47 return;
48 }

50 static void DeserializeWrappedVarType(YAML::Node& componentNode, WrappedVarType type, const std::string& name, void* destination)
51 {
52 switch (type)
53 {
54 case WrappedVarType::Integer32:
55 *(int32_t*)destination = componentNode[name].as<int32_t>();
56 return;
57 case WrappedVarType::UInteger16:
58 *(uint16_t*)destination = (uint16_t)componentNode[name].as<uint32_t>();
59 return;
60 case WrappedVarType::UInteger32:
61 *(uint32_t*)destination = componentNode[name].as<uint32_t>();
62 return;
63 case WrappedVarType::UInteger64:
64 *(uint64_t*)destination = componentNode[name].as<uint64_t>();
65 return;
66 case WrappedVarType::Vector3:
67 *(Math::vec3*)destination = componentNode[name].as<Math::vec3>();
68 return;
69 case WrappedVarType::String:
70 *(std::string*)destination = componentNode[name].as<std::string>();
71 return;
72 case WrappedVarType::Bool:
73 *(bool*)destination = componentNode[name].as<bool>();
74 return;
75 case WrappedVarType::Float:
76 *(float*)destination = componentNode[name].as<float>();
77 return;
78 case WrappedVarType::None:
79 case WrappedVarType::Void:
80 KG_ERROR("Invalid type provided while serializing data");
81 return;
82 }
83 KG_ERROR("Unknown Type of WrappedVariableType.");
84 return;

251 if (entity.HasComponent<ECS::CircleCollider2DComponent>())
252 {
253 out << YAML::Key << "CircleCollider2DComponent";
254 out << YAML::BeginMap; // Component Map
255 auto& cc2dComponent = entity.GetComponent<ECS::CircleCollider2DComponent>();
256 out << YAML::Key << "Offset" << YAML::Value << cc2dComponent.Offset;
257 out << YAML::Key << "Radius" << YAML::Value << cc2dComponent.Radius;

259 out << YAML::Key << "Friction" << YAML::Value << cc2dComponent.Friction;
260 out << YAML::Key << "Restitution" << YAML::Value << cc2dComponent.Restitution;
261 out << YAML::Key << "RestitutionThreshold" << YAML::Value << cc2dComponent.RestitutionThreshold;

263 out << YAML::EndMap; // Component Map
264 }
265
266 // Handle all project components
267 for (auto& [handle, asset] : Assets::AssetService::GetProjectComponentRegistry())
268 {
269 if (!entity.HasProjectComponentData(handle))
270 {
271 continue;
272 }
273 Ref<ECS::ProjectComponent> projectComponent = Assets::AssetService::GetProjectComponent(handle);
274 uint8_t* componentRef = (uint8_t*)entity.GetProjectComponentData(handle);

276 out << YAML::Key << projectComponent->m_Name + "Component";

280 SerializeWrappedVarType(out,

282 projectComponent->m_DataNames.at(iteration),
283 componentRef + projectComponent->m_DataLocations.at(iteration));
284 }
285 out << YAML::EndMap; // Component Map
286 }
287
288
289 out << YAML::EndMap; // Entity
290 return true;
291 }
