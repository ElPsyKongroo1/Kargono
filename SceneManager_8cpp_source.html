12 static bool SerializeEntity(YAML::Emitter& out, Scenes::Entity entity)
13 {
14 KG_ASSERT(entity.HasComponent<Scenes::IDComponent>(), "Entity does not have a component");
15 out << YAML::BeginMap; // Entity Map
16 out << YAML::Key << "Entity" << YAML::Value << static_cast<uint64_t>(entity.GetUUID());
17
18 if (entity.HasComponent<Scenes::TagComponent>())
19 {
20 out << YAML::Key << "TagComponent";
21 out << YAML::BeginMap; // Component Map
22 auto& tag = entity.GetComponent<Scenes::TagComponent>().Tag;
23 out << YAML::Key << "Tag" << YAML::Value << tag;
24 out << YAML::EndMap; // Component Map
25 }
26
27 if (entity.HasComponent<Scenes::ClassInstanceComponent>())
28 {
29 out << YAML::Key << "ClassInstanceComponent";
30 Scenes::ClassInstanceComponent& comp = entity.GetComponent<Scenes::ClassInstanceComponent>();
31 out << YAML::BeginMap; // Component Map
32 out << YAML::Key << "ClassHandle" << YAML::Value << static_cast<uint64_t>(comp.ClassHandle);
33 if (comp.ClassHandle != Assets::EmptyHandle)
34 {
35 Ref<Scenes::EntityClass> entityClass = Assets::AssetManager::GetEntityClass(comp.ClassHandle);
36 if (!entityClass)
37 {
38 KG_ERROR("Attempt to serialize ClassInstanceComponent without valid entityClass");
39 return false;
40 }
41 if (entityClass->GetFields().size() != comp.Fields.size())
42 {
43 KG_ERROR("Attempt to serialize ClassInstanceComponent where class and instance fields are unaligned");
44 return false;
45 }
46
47 out << YAML::Key << "InstanceFields" << YAML::Value << YAML::BeginSeq; // Begin Fields
48 uint32_t iteration{ 0 };
49 for (auto& fieldValue : comp.Fields)
50 {
51 const Scenes::ClassField& fieldType = entityClass->GetFields().at(iteration);
52 if (fieldType.Type != fieldValue->Type())
53 {
54 KG_ERROR("Attempt to serialize ClassInstanceComponent with incorrect types");
55 return false;
56 }
57 out << YAML::BeginMap; // Begin Field Map
58 out << YAML::Key << "Name" << YAML::Value << fieldType.Name;
59 out << YAML::Key << "Type" << YAML::Value << Utility::WrappedVarTypeToString(fieldType.Type);
60 SerializeWrappedVariableData(fieldValue, out);
61 out << YAML::EndMap; // End Field Map
62 iteration++;
63 }
64 out << YAML::EndSeq; // End Fields
65 }
66
67 out << YAML::EndMap; // Component Map
68 }
69
70 if (entity.HasComponent<Scenes::TransformComponent>())
71 {
72 out << YAML::Key << "TransformComponent";
73 out << YAML::BeginMap; // Component Map
74 auto& tc = entity.GetComponent<Scenes::TransformComponent>();
75 out << YAML::Key << "Translation" << YAML::Value << tc.Translation;
76 out << YAML::Key << "Rotation" << YAML::Value << tc.Rotation;
77 out << YAML::Key << "Scale" << YAML::Value << tc.Scale;
78
79 out << YAML::EndMap; // Component Map
80 }
81 if (entity.HasComponent<Scenes::CameraComponent>())
82 {
83 out << YAML::Key << "CameraComponent";
84 out << YAML::BeginMap; // Component Map
85
86 auto& cameraComponent = entity.GetComponent<Scenes::CameraComponent>();
87 auto& camera = cameraComponent.Camera;
88
89 out << YAML::Key << "Camera" << YAML::Value;
90 out << YAML::BeginMap;
91 out << YAML::Key << "ProjectionType" << YAML::Value << (int)camera.GetProjectionType();
92 out << YAML::Key << "PerspectiveFOV" << YAML::Value << camera.GetPerspectiveVerticalFOV();
93 out << YAML::Key << "PerspectiveNear" << YAML::Value << camera.GetPerspectiveNearClip();
94 out << YAML::Key << "PerspectiveFar" << YAML::Value << camera.GetPerspectiveFarClip();
95
96 out << YAML::Key << "OrthographicSize" << YAML::Value << camera.GetOrthographicSize();
97 out << YAML::Key << "OrthographicNear" << YAML::Value << camera.GetOrthographicNearClip();
98 out << YAML::Key << "OrthographicFar" << YAML::Value << camera.GetOrthographicFarClip();
99 out << YAML::EndMap;
100
101 out << YAML::Key << "Primary" << YAML::Value << cameraComponent.Primary;

103
104 out << YAML::EndMap; // Component Map
105 }

107 if (entity.HasComponent<Scenes::ShapeComponent>())
108 {
109 out << YAML::Key << "ShapeComponent";

113 if (shapeComponent.VertexColors)

116 for (const auto& color : *shapeComponent.VertexColors)
117 {
118 out << YAML::BeginMap;
119 out << YAML::Key << "Color" << YAML::Value << color;
120 out << YAML::EndMap;
121 }
122 out << YAML::EndSeq;
123 }
124 if (shapeComponent.Texture)
125 {
126 out << YAML::Key << "TextureHandle" << YAML::Value << static_cast<uint64_t>(shapeComponent.TextureHandle);
127 }
128 KG_ASSERT(sizeof(uint8_t) * 20 == sizeof(Rendering::ShaderSpecification), "Please Update Deserialization and Serialization. Incorrect size of input data in Scene Serializer!");
129 if (shapeComponent.Shader)
130 {
131 // Add Shader Handle
132 out << YAML::Key << "ShaderHandle" << YAML::Value << static_cast<uint64_t>(shapeComponent.ShaderHandle);
133 // Add Shader Specification
134 const Rendering::ShaderSpecification& shaderSpec = shapeComponent.Shader->GetSpecification();
135 out << YAML::Key << "ShaderSpecification" << YAML::Value;

200 void AssetManager::DeserializeSceneRegistry()
201 {
202 // Clear current registry and open registry in current project
203 s_SceneRegistry.clear();
204 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project to serialize from!");
205 const auto& sceneRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Scenes/SceneRegistry.kgreg";
206
207 if (!std::filesystem::exists(sceneRegistryLocation))
208 {
209 KG_WARN("No .kgregistry file exists in project path!");
210 return;
211 }
212 YAML::Node data;
213 try
214 {
215 data = YAML::LoadFile(sceneRegistryLocation.string());
216 }
217 catch (YAML::ParserException e)
218 {
219 KG_ERROR("Failed to load .kgscene file '{0}'\n {1}", sceneRegistryLocation.string(), e.what());
220 return;
221 }
222
223 // Opening registry node
224 if (!data["Registry"]) { return; }
225
226 std::string registryName = data["Registry"].as<std::string>();
227 KG_INFO("Deserializing Scene Registry");
228
229 // Opening all assets
230 auto assets = data["Assets"];
231 if (assets)
232 {
233 for (auto asset : assets)
234 {
235 Assets::Asset newAsset{};
236 newAsset.Handle = asset["AssetHandle"].as<uint64_t>();
237
238 // Retrieving metadata for asset
239 auto metadata = asset["MetaData"];
240 newAsset.Data.CheckSum = metadata["CheckSum"].as<std::string>();
241 newAsset.Data.IntermediateLocation = metadata["IntermediateLocation"].as<std::string>();
242 newAsset.Data.Type = Utility::StringToAssetType(metadata["AssetType"].as<std::string>());
243
244 // Retrieving shader specific metadata
245 if (newAsset.Data.Type == Assets::Scene)
246 {
247 Ref<Assets::SceneMetaData> sceneMetaData = CreateRef<Assets::SceneMetaData>();
248 newAsset.Data.SpecificFileData = sceneMetaData;
249 }
250
251 // Add asset to in memory registry
252 s_SceneRegistry.insert({ newAsset.Handle, newAsset });
253 }
254 }

258 {

260 const auto& sceneRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Scenes/SceneRegistry.kgreg";

290 void AssetManager::SerializeScene(Ref<Scenes::Scene> scene, const std::filesystem::path& filepath)
291 {
292 bool submitScene = true;
293 YAML::Emitter out;
294 out << YAML::BeginMap; // Start of File Map
295 { // Physics
296 out << YAML::Key << "Physics" << YAML::BeginMap; // Physics Map
297 out << YAML::Key << "Gravity" << YAML::Value << scene->m_PhysicsSpecification.Gravity;
298 out << YAML::EndMap; // Physics Maps
299 }
300
301 out << YAML::Key << "Entities" << YAML::Value << YAML::BeginSeq;
302 scene->m_Registry.each([&](auto entityID)
303 {
304 Scenes::Entity entity = { entityID, scene.get() };
305 if (!entity) { return; }
306
307 bool success = Utility::SerializeEntity(out, entity);
308 if (!success)
309 {
310 submitScene = false;
311 }
312 });
313 out << YAML::EndSeq;
314 out << YAML::EndMap; // Start of File Map
315 if (submitScene)
316 {
317 std::ofstream fout(filepath);
318 fout << out.c_str();
319 KG_INFO("Successfully Serialized Scene at {}", filepath);
320 }
321 else
322 {
323 KG_WARN("Failed to Serialize Scene");
324 }

351 bool AssetManager::DeserializeScene(Ref<Scenes::Scene> scene, const std::filesystem::path& filepath)
352 {
353 YAML::Node data;

363
364 KG_INFO("Deserializing scene");
365
366 auto physics = data["Physics"];
367 scene->GetPhysicsSpecification().Gravity = physics["Gravity"].as<Math::vec2>();
368
369 auto entities = data["Entities"];
370 if (entities)
371 {
372 for (auto entity : entities)
373 {
374 uint64_t uuid = entity["Entity"].as<uint64_t>();
375
376 std::string name;
377 auto tagComponent = entity["TagComponent"];
378 if (tagComponent) { name = tagComponent["Tag"].as<std::string>(); }
379 //KG_CORE_TRACE("Deserialize entity with ID = {0}, name = {1}", uuid, name);
380
381 Scenes::Entity deserializedEntity = scene->CreateEntityWithUUID(uuid, name);
382
383 auto transformComponent = entity["TransformComponent"];
384 if (transformComponent)
385 {
386 auto& tc = deserializedEntity.GetComponent<Scenes::TransformComponent>();
387 tc.Translation = transformComponent["Translation"].as<Math::vec3>();
388 tc.Rotation = transformComponent["Rotation"].as<Math::vec3>();
389 tc.Scale = transformComponent["Scale"].as<Math::vec3>();
390 }
391
392 auto classInstanceComponent = entity["ClassInstanceComponent"];
393 if (classInstanceComponent)
394 {
395 auto& cInstComp = deserializedEntity.AddComponent<Scenes::ClassInstanceComponent>();
396 cInstComp.ClassHandle = classInstanceComponent["ClassHandle"].as<uint64_t>();
397 if (cInstComp.ClassHandle != Assets::EmptyHandle)
398 {
399 if (!s_EntityClassRegistry.contains(cInstComp.ClassHandle))
400 {
401 KG_ERROR("Could not find entity class for class instance component");
402 return false;
403 }
404 cInstComp.ClassReference = GetEntityClass(cInstComp.ClassHandle);
405
406 auto instanceFields = classInstanceComponent["InstanceFields"];
407 if (instanceFields)
408 {
409 for (auto field : instanceFields)
410 {
411 WrappedVarType type = Utility::StringToWrappedVarType(field["Type"].as<std::string>());
412 Ref<WrappedVariable> newField = Utility::DeserializeWrappedVariableData(type, field);
413 cInstComp.Fields.push_back(newField);
414 }
415 }
416 }
417 }
418
419 auto cameraComponent = entity["CameraComponent"];
420 if (cameraComponent)
421 {
422 auto& cc = deserializedEntity.AddComponent<Scenes::CameraComponent>();
423
424 const auto& cameraProps = cameraComponent["Camera"];
425
426 cc.Camera.SetProjectionType((Scenes::SceneCamera::ProjectionType)cameraProps["ProjectionType"].as<int>());
427 cc.Camera.SetPerspectiveVerticalFOV(cameraProps["PerspectiveFOV"].as<float>());
428 cc.Camera.SetPerspectiveNearClip(cameraProps["PerspectiveNear"].as<float>());
429 cc.Camera.SetPerspectiveFarClip(cameraProps["PerspectiveFar"].as<float>());
430
431 cc.Camera.SetOrthographicSize(cameraProps["OrthographicSize"].as<float>());
432 cc.Camera.SetOrthographicNearClip(cameraProps["OrthographicNear"].as<float>());
433 cc.Camera.SetOrthographicFarClip(cameraProps["OrthographicFar"].as<float>());
434
435 cc.Primary = cameraComponent["Primary"].as<bool>();
436 }
437
438 auto shapeComponent = entity["ShapeComponent"];
439 if (shapeComponent)
440 {
441 auto& sc = deserializedEntity.AddComponent<Scenes::ShapeComponent>();
442 sc.CurrentShape = Utility::StringToShapeType(shapeComponent["CurrentShape"].as<std::string>());
443
444 if (shapeComponent["VertexColors"])
445 {
446 auto vertexColors = shapeComponent["VertexColors"];
447 sc.VertexColors = CreateRef<std::vector<Math::vec4>>();
448 for (auto color : vertexColors)
449 {
450 sc.VertexColors->push_back(color["Color"].as<Math::vec4>());
451 }
452 }
453
454 if (shapeComponent["TextureHandle"])
455 {
456 AssetHandle textureHandle = shapeComponent["TextureHandle"].as<uint64_t>();
457 sc.Texture = AssetManager::GetTexture(textureHandle);
458 sc.TextureHandle = textureHandle;
459 }
460
461 if (shapeComponent["ShaderHandle"])
462 {
463 AssetHandle shaderHandle = shapeComponent["ShaderHandle"].as<uint64_t>();
464 sc.Shader = AssetManager::GetShader(shaderHandle);
465 if (!sc.Shader)
466 {
467 auto shaderSpecificationNode = shapeComponent["ShaderSpecification"];
468 Rendering::ShaderSpecification shaderSpec{};
469 // ShaderSpecification Section
470 shaderSpec.ColorInput = Utility::StringToColorInputType(shaderSpecificationNode["ColorInputType"].as<std::string>());
471 shaderSpec.AddProjectionMatrix = shaderSpecificationNode["AddProjectionMatrix"].as<bool>();
472 shaderSpec.AddEntityID = shaderSpecificationNode["AddEntityID"].as<bool>();
473 shaderSpec.AddCircleShape = shaderSpecificationNode["AddCircleShape"].as<bool>();
474 shaderSpec.TextureInput = Utility::StringToTextureInputType(shaderSpecificationNode["TextureInput"].as<std::string>());
475 shaderSpec.DrawOutline = shaderSpecificationNode["DrawOutline"].as<bool>();
476 shaderSpec.RenderType = Utility::StringToRenderingType(shaderSpecificationNode["RenderType"].as<std::string>());
477 auto [newHandle, newShader] = AssetManager::GetShader(shaderSpec);
478 shaderHandle = newHandle;
479 sc.Shader = newShader;
480 }
481 sc.ShaderHandle = shaderHandle;
482 sc.ShaderSpecification = sc.Shader->GetSpecification();
483 YAML::Binary binary = shapeComponent["Buffer"].as<YAML::Binary>();
484 Buffer buffer{ binary.size() };

487 if (sc.CurrentShape != Rendering::ShapeTypes::None)
488 {
489 if (sc.ShaderSpecification.RenderType == Rendering::RenderingType::DrawIndex)

494 }
495
496 if (sc.ShaderSpecification.RenderType == Rendering::RenderingType::DrawTriangle)
497 {
498 sc.Vertices = CreateRef<std::vector<Math::vec3>>(Utility::ShapeTypeToShape(sc.CurrentShape).GetTriangleVertices());
499 sc.TextureCoordinates = CreateRef<std::vector<Math::vec2>>(Utility::ShapeTypeToShape(sc.CurrentShape).GetTriangleTextureCoordinates());
500 }
501 }
502 }
503 }
504
505 auto rigidbody2DComponent = entity["Rigidbody2DComponent"];
506 if (rigidbody2DComponent)
507 {
508 auto& rb2d = deserializedEntity.AddComponent<Scenes::Rigidbody2DComponent>();
509 rb2d.Type = Utility::StringToRigidBody2DBodyType(rigidbody2DComponent["BodyType"].as<std::string>());
510 rb2d.FixedRotation = rigidbody2DComponent["FixedRotation"].as<bool>();
511 }
512
513 auto boxCollider2DComponent = entity["BoxCollider2DComponent"];
514 if (boxCollider2DComponent)
515 {
516 auto& bc2d = deserializedEntity.AddComponent<Scenes::BoxCollider2DComponent>();
517 bc2d.Offset = boxCollider2DComponent["Offset"].as<Math::vec2>();
518 bc2d.Size = boxCollider2DComponent["Size"].as<Math::vec2>();
519 bc2d.Density = boxCollider2DComponent["Density"].as<float>();
520 bc2d.Friction = boxCollider2DComponent["Friction"].as<float>();
521 bc2d.Restitution = boxCollider2DComponent["Restitution"].as<float>();
522 bc2d.RestitutionThreshold = boxCollider2DComponent["RestitutionThreshold"].as<float>();
523 }
524
525 auto circleCollider2DComponent = entity["CircleCollider2DComponent"];
526 if (circleCollider2DComponent)
527 {
528 auto& cc2d = deserializedEntity.AddComponent<Scenes::CircleCollider2DComponent>();
529 cc2d.Offset = circleCollider2DComponent["Offset"].as<Math::vec2>();
530 cc2d.Radius = circleCollider2DComponent["Radius"].as<float>();
531 cc2d.Density = circleCollider2DComponent["Density"].as<float>();
532 cc2d.Friction = circleCollider2DComponent["Friction"].as<float>();

535 }

537 }

542 AssetHandle AssetManager::CreateNewScene(const std::string& sceneName)
543 {
544 // Create Checksum
545 const std::string currentCheckSum = Utility::FileSystem::ChecksumFromString(sceneName);
546
547 if (currentCheckSum.empty())
548 {
549 KG_ERROR("Failed to generate checksum from file!");
550 return {};
551 }
552
553 // Compare currentChecksum to registered assets
554 for (const auto& [handle, asset] : s_SceneRegistry)
555 {
556 if (asset.Data.CheckSum == currentCheckSum)
557 {
558 KG_INFO("Attempt to instantiate duplicate font asset");
559 return handle;
560 }
561 }
562
563 // Create New Asset/Handle
564 AssetHandle newHandle{};
565 Assets::Asset newAsset{};
566 newAsset.Handle = newHandle;
567
568 // Create File
569 CreateSceneFile(sceneName, newAsset);
570 newAsset.Data.CheckSum = currentCheckSum;
571
572 // Register New Asset and Create Scene
573 s_SceneRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
574 SerializeSceneRegistry(); // Update Registry File on Disk
575
576 return newHandle;
577 }
