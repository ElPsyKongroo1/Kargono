79  struct ScriptFieldInstance
80  {
81  // Represent Field Type inside instance
82  ScriptField Field;
83  //============================
84  // Constructor
85  //============================
86  // This constructor sets the current buffer to 0
87  ScriptFieldInstance()
88  {
89  memset(m_Buffer, 0, sizeof(m_Buffer));
90  }
91  //============================
92  // Getter/Setters
93  //============================
94  // Provides access to buffer
95  // Template used to allow access using different
96  // types of fields (Ex: A float would access the first
97  // for bytes of m_Buffer while a long would access all
98  // 8 bytes.
99  template<typename T>
100  T GetValue()
101  {
102  KG_ASSERT(sizeof(T) <= 16, "Type too large!");
103  return *(T*)m_Buffer;
104  }
105  // Provides ability to set data inside m_Buffer based
106  // on type presented
107  template<typename T>
108  void SetValue(T value)
109  {
110  KG_ASSERT(sizeof(T) <= 16, "Type too large!");
111  memcpy(m_Buffer, &value, sizeof(T));
112  }
113  private:
114  // Byte buffer that holds field value at runtime
115  uint8_t m_Buffer[8];
