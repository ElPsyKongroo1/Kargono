103 bool AIStateEditorPanel::OnAssetEvent(Events::Event* event)
104 {
105
106 Events::ManageAsset* manageAsset = (Events::ManageAsset*)event;
107 // Manage script deletion event
108 if (manageAsset->GetAssetType() == Assets::AssetType::Script &&
109 manageAsset->GetAction() == Events::ManageAssetAction::PreDelete)
110 {
111 if (m_SelectOnUpdateScript.m_CurrentOption.m_Handle == manageAsset->GetAssetID())
112 {
113 m_SelectOnUpdateScript.m_CurrentOption = { "None", Assets::EmptyHandle };
114 }
115
116 if (m_SelectOnEnterStateScript.m_CurrentOption.m_Handle == manageAsset->GetAssetID())
117 {
118 m_SelectOnEnterStateScript.m_CurrentOption = { "None", Assets::EmptyHandle };
119 }
120
121 if (m_SelectOnExitStateScript.m_CurrentOption.m_Handle == manageAsset->GetAssetID())
122 {
123 m_SelectOnExitStateScript.m_CurrentOption = { "None", Assets::EmptyHandle };
124 }
125
126 if (m_SelectOnAIMessageScript.m_CurrentOption.m_Handle == manageAsset->GetAssetID())
127 {
128 m_SelectOnAIMessageScript.m_CurrentOption = { "None", Assets::EmptyHandle };
129 }
130
131 if (m_EditorAIState)
132 {
133 Assets::AssetService::RemoveScriptFromAIState(m_EditorAIState, manageAsset->GetAssetID());
134 }
135 }
136
137 if (manageAsset->GetAssetType() == Assets::AssetType::AIState &&
138 manageAsset->GetAction() == Events::ManageAssetAction::PreDelete)
139 {
140 if (manageAsset->GetAssetID() != m_EditorAIStateHandle)
141 {
142 return false;
143 }
144
145 // Handle deletion of asset
146 ResetPanelResources();
147 return true;
148 }
149
150 if (manageAsset->GetAssetType() == Assets::AssetType::AIState &&
151 manageAsset->GetAction() == Events::ManageAssetAction::UpdateAssetInfo)
152 {
153 if (manageAsset->GetAssetID() != m_EditorAIStateHandle)
154 {
155 return false;
156 }
157
158 // Update header
159 m_MainHeader.m_Label = Assets::AssetService::GetAIStateFileLocation(manageAsset->GetAssetID()).filename().string();
160 return true;
161 }
162
163 return false;
164 }

193 void AIStateEditorPanel::OpenAssetInEditor(std::filesystem::path& assetLocation)
194 {
195 // Ensure provided path is within the active asset directory
196 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
197 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
198 {
199 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
200 return;
201 }
202
203 // Look for asset in registry using the file location
204 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
205 Assets::AssetHandle assetHandle = Assets::AssetService::GetAIStateHandleFromFileLocation(relativePath);
206
207 // Validate resulting handle
208 if (!assetHandle)
209 {
210 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
211 return;
212 }
213
214 // Open the editor panel to be visible
215 s_MainWindow->m_ShowAIStateEditor = true;
216 EditorUI::EditorUIService::BringWindowToFront(m_PanelName);
217 EditorUI::EditorUIService::SetFocusedWindow(m_PanelName);
218
219 // Early out if asset is already open
220 if (m_EditorAIStateHandle == assetHandle)
221 {
222 return;
223 }
224
225 // Check if panel is already occupied by an asset
226 if (!m_EditorAIState)
227 {
228 // Open dialog to create editor AI State
229 OnOpenAIState(assetHandle);
230 }
231 else
232 {
233 // Add warning to close active AI state before opening a new AIState
234 s_MainWindow->OpenWarningMessage("An AI State is already active inside the editor. Please close the current AI State before opening a new one.");
235 }
