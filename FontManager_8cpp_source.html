148  void AssetManager::SerializeFontRegistry()
149  {

151  const auto& fontRegistryLocation = Projects::Project::GetAssetDirectory() / "Fonts/Intermediates/FontRegistry.kgreg";
152  YAML::Emitter out;
153 
154  out << YAML::BeginMap;
155  out << YAML::Key << "Registry" << YAML::Value << "Font";

158  // Asset

163 
164  out << YAML::Key << "MetaData" << YAML::Value;
165  out << YAML::BeginMap; // MetaData Map

175 
176  out << YAML::Key << "AtlasWidth" << YAML::Value << metadata->AtlasWidth;
177  out << YAML::Key << "AtlasHeight" << YAML::Value << metadata->AtlasHeight;
178  out << YAML::Key << "LineHeight" << YAML::Value << metadata->LineHeight;
179

186  out << YAML::Key << "TexCoordinateMin" << YAML::Value << characterStruct.TexCoordinateMin;
187  out << YAML::Key << "TexCoordinateMax" << YAML::Value << characterStruct.TexCoordinateMax;

191  out << YAML::EndMap;
192  }
193  out << YAML::EndSeq;
194  }
