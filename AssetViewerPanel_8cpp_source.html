32 bool AssetViewerPanel::OnAssetEvent(Events::Event* event)
33 {
34 Events::ManageAsset& manageEvent = *(Events::ManageAsset*)event;
35
36 // Handle deletion event
37 if (manageEvent.GetAction() == Events::ManageAssetAction::PreDelete)
38 {
39 // Search table for deleted asset
40 std::size_t assetLocation = m_AllAssetsTable.SearchEntries([&](const EditorUI::ListEntry& currentEntry)
41 {
42 // Check if type inside tree is the same
43 if (Utility::StringToAssetType(currentEntry.m_Label) != manageEvent.GetAssetType())
44 {
45 return false;
46 }
47
48 // Check if handle matches
49 if (currentEntry.m_Handle != manageEvent.GetAssetID())
50 {
51 return false;
52 }
53
54 // Index has been found
55 return true;
56
57 });
58
59 // Validate that table search was successful
60 KG_ASSERT(assetLocation != EditorUI::k_ListSearchIndex, "Asset being deleted was not found in asset table");
61
62 // Delete entry and validate deletion
63 bool deletionSuccess = m_AllAssetsTable.RemoveEntry(assetLocation);
64 KG_ASSERT(deletionSuccess, "Unable to delete asset inside AssetViewer");
65 }
66
67 // Handle all othe event types
68 else
69 {
70 RefreshPanelData();
71 }
72
73 return true;
74 }
