13  // This class represents a wrapper around a GLFW window. The GLFW
14  // window is the actual windows container itself that uses
15  // the Windows API. The window contains the viewport of OpenGL
16  // which can have differing dimensions. The window also serves
17  // to receive input events such as Key Presses, Mouse Clicks,
18  // and Window Resize Events. These events are registered with
19  // callback functions in the Init() function. The events are
20  // handled later in the engine through the Application class.
21  class WindowsWindow : public Kargono::Window
22  {
23  private:
24  //==============================
25  // Internal Structs
26  //==============================
27  // This struct holds data that describes the GLFW window. The member variable m_Data
28  // represents that information. The title is the name presented on the window,
29  // width/height are the dimensions, VSync indicates if VSync is currently enabled,
30  // Versions simply state what version of OpenGL we are using, and EventCallback is
31  // the function that is called when an event needs to be handled. Currently the
32  // EventCallback is always connected to the Application function OnEvent()
33  struct WindowData
34  {
35  std::string Title;
36  uint32_t Width, Height;
37  uint32_t ViewportWidth, ViewportHeight;
38  bool VSync;
39  uint8_t VersionMajor = 4, VersionMinor = 5;
40  Kargono::Events::EventCallbackFn EventCallback;
41  };
42 
43  public:

46  //==============================
47  // This constructor simply calls the Init() lifetime function.
48  // The Init Description will be more useful.

56  public:
57  // The main focus of this function is to initialize the underlying GLFW window.
58  // The ancillary uses of this function include:
59  // 1. It initializes the m_Window* with a reference to the actual GLFW window
60  // 2. It fills m_Data with relevant window information
61  // 3. It initializes GLFW and GLAD (GLAD retrieves the OpenGL function pointers
62  // from the GPU drivers so they are available to use)
63  // 4. It registers function callbacks with GLFW so events such as keyboard presses
64  // or window resize events can be handles by the engine.
65  // 5. It initializes the logo used by the application.

70  // This function simply closes the GLFW window associated with its instance. This window
