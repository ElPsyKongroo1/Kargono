559 void UIEditorViewportPanel::DrawProportionalGrid()
560 {
561 // Draw grid lines
562 if (m_PropertionalGridSections == 1)
563 {
564 return;
565 }
566
567 // Calculate grid line values and set up vertex data
568 size_t lineCount = {((size_t)m_PropertionalGridSections - 1) * 2};
569 size_t vertexCount = lineCount * 2;
570 std::vector<Math::vec3> gridVertices;
571 gridVertices.reserve(vertexCount);
572
573 // Fill vertex data
574 for (size_t iteration{ 1 }; iteration <= lineCount; iteration++)
575 {
576 float x = ((float)m_ViewportData.m_Width / m_PropertionalGridSections) * iteration;
577 float y = ((float)m_ViewportData.m_Height / m_PropertionalGridSections) * iteration;
578
579 // Horizontal line
580 gridVertices.push_back({ 0.0f, y, -1.0f });
581 gridVertices.push_back({ (float)m_ViewportData.m_Width, y, -1.0f });
582
583 // Vertical line
584 gridVertices.push_back({ x, 0.0f, -1.0f });
585 gridVertices.push_back({ x, (float)m_ViewportData.m_Height, -1.0f });
586 }
587
588 Rendering::Shader::SetDataAtInputLocation<Math::vec4>(Utility::ImVec4ToMathVec4(EditorUI::EditorUIService::s_GridMinor), "a_Color", s_LineInputSpec.m_Buffer, s_LineInputSpec.m_Shader);
589
590 for (size_t i = 0; i < vertexCount; i += 2)
591 {
592 s_OutputVector->clear();
593 s_OutputVector->push_back(gridVertices[i]);
594 s_OutputVector->push_back(gridVertices[i + 1]);
595 s_LineInputSpec.m_ShapeComponent->Vertices = s_OutputVector;
596 Rendering::RenderingService::SubmitDataToRenderer(s_LineInputSpec);
597 }
