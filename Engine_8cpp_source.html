73 void EngineService::Run()
74 {
75 KG_INFO("Starting Run Function");
76
77 if (!s_ActiveEngine)
78 {
79 return;
80 }
81
82 using namespace std::chrono_literals;
83
84 std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
85 std::chrono::time_point<std::chrono::high_resolution_clock> lastCycleTime = currentTime;
86 std::chrono::nanoseconds timestep{ 0 };
87
88 while (s_ActiveEngine->m_Running)
89 {
90 currentTime = std::chrono::high_resolution_clock::now();
91 timestep = currentTime - lastCycleTime;
92 lastCycleTime = currentTime;
93 s_ActiveEngine->m_Accumulator += timestep;
94 if (s_ActiveEngine->m_Accumulator < k_ConstantFrameTime)
95 {
96 continue;
97 }
98 s_ActiveEngine->m_Accumulator -= k_ConstantFrameTime;
99
100 {
101 KG_PROFILE_FRAME("Main Thread");
102
103 s_ActiveEngine->m_UpdateCount++;
104 AppTickService::OnUpdate(k_ConstantFrameTimeStep);
105
106 ExecuteMainThreadQueue();
107 if (!s_ActiveEngine->m_Minimized)
108 {
109 s_ActiveEngine->m_CurrentApp->OnUpdate(k_ConstantFrameTimeStep);
110 }
111 ProcessEventQueue();
112 s_ActiveEngine->m_Window->OnUpdate();
113 }
114 }
115 KG_INFO("Ending Run Function");
116 }

167 bool EngineService::OnUpdateEntityLocation(Events::UpdateEntityLocation& e)
168 {
169 Ref<Scenes::Scene> scene = Scenes::SceneService::GetActiveScene();
170 if (!scene) { return false; }
171 Scenes::Entity entity = scene->GetEntityByUUID(e.GetEntityID());
172 if (!entity) { return false; }
173 Math::vec3 translation = e.GetTranslation();
174 entity.GetComponent<Scenes::TransformComponent>().Translation = translation;
175
176 if (entity.HasComponent<Scenes::Rigidbody2DComponent>())

178 auto& rb2d = entity.GetComponent<Scenes::Rigidbody2DComponent>();
179 b2Body* body = (b2Body*)rb2d.RuntimeBody;
