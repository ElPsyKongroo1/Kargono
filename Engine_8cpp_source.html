80 void EngineService::Run()
81 {
82 KG_INFO("Starting Run Function");
83
84 if (!s_ActiveEngine)
85 {
86 return;
87 }
88
89 using namespace std::chrono_literals;
90
91 std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
92 std::chrono::time_point<std::chrono::high_resolution_clock> lastCycleTime = currentTime;
93 std::chrono::nanoseconds timestep{ 0 };
94
95 while (s_ActiveEngine->m_Running)
96 {
97 currentTime = std::chrono::high_resolution_clock::now();
98 timestep = currentTime - lastCycleTime;
99 lastCycleTime = currentTime;
100 s_ActiveEngine->m_Accumulator += timestep;
101 if (s_ActiveEngine->m_Accumulator < k_ConstantFrameTime)
102 {
103 continue;
104 }
105 s_ActiveEngine->m_Accumulator -= k_ConstantFrameTime;
106
107 {
108 KG_PROFILE_FRAME("Main Thread");
109
110 s_ActiveEngine->m_UpdateCount++;
111 AppTickService::OnUpdate(k_ConstantFrameTimeStep);
112 Utility::PassiveTimer::OnUpdate(k_ConstantFrameTimeStep);
113
114 ProcessFunctionQueue();
115 if (!s_ActiveEngine->m_Minimized)
116 {
117 s_ActiveEngine->m_CurrentApp->OnUpdate(k_ConstantFrameTimeStep);
118 }
119 ProcessEventQueue();
120 s_ActiveEngine->m_Window->OnUpdate();
121 }
122 }
123 KG_INFO("Ending Run Function");
124 }

131 void EngineService::OnEvent(Events::Event* e)
132 {
133 bool handled = false;
134 // Process Input Events
135 if (e->IsInCategory(Events::Input))
136 {
137 if (s_ActiveEngine->m_CurrentApp)
138 {
139 s_ActiveEngine->m_CurrentApp->OnInputEvent(e);
140 }
141 return;
142 }
143
144 // Process Physics Events
145 if (e->IsInCategory(Events::Physics))
146 {
147 if (s_ActiveEngine->m_CurrentApp)
148 {
149 s_ActiveEngine->m_CurrentApp->OnPhysicsEvent(e);
150 }
151 return;
152 }
153
154 // Process Network Events
155 if (e->IsInCategory(Events::Network))
156 {
157 switch (e->GetEventType())
158 {
159 case Events::EventType::UpdateEntityLocation:
160 handled = OnUpdateEntityLocation(*(Events::UpdateEntityLocation*)e);
161 break;
162 case Events::EventType::UpdateEntityPhysics:
163 handled = OnUpdateEntityPhysics(*(Events::UpdateEntityPhysics*)e);
164 break;
165 }
166
167 if (handled)
168 {
169 return;
170 }
171 if (s_ActiveEngine->m_CurrentApp)
172 {
173 s_ActiveEngine->m_CurrentApp->OnNetworkEvent(e);
174 }
175 return;
176 }
177
178 if (e->IsInCategory(Events::Scene))
179 {
180 if (s_ActiveEngine->m_CurrentApp)
181 {
182 s_ActiveEngine->m_CurrentApp->OnSceneEvent(e);
183 }
184 return;
185 }
186
187 if (e->IsInCategory(Events::Asset))
188 {
189 if (s_ActiveEngine->m_CurrentApp)
190 {
191 s_ActiveEngine->m_CurrentApp->OnAssetEvent(e);
192 }
193 return;
194 }
195
196 // Process Application Events
197 if (e->IsInCategory(Events::Application))
198 {
199 switch (e->GetEventType())
200 {
201 case Events::EventType::LogEvent:
202 handled = s_ActiveEngine->m_CurrentApp->OnLogEvent(e);
203 break;
204 case Events::EventType::WindowClose:
205 handled = OnWindowClose(*(Events::WindowCloseEvent*)e);
206 break;
207 case Events::EventType::WindowResize:
208 handled = OnWindowResize(*(Events::WindowResizeEvent*)e);
209 break;
210 case Events::EventType::CleanUpTimers:
211 handled = OnCleanUpTimers(*(Events::CleanUpTimersEvent*)e);
212 break;
213 case Events::EventType::AddTickGeneratorUsage:
214 handled = OnAddTickGeneratorUsage(*(Events::AddTickGeneratorUsage*)e);
215 break;
216 case Events::EventType::RemoveTickGeneratorUsage:
217 handled = OnRemoveTickGeneratorUsage(*(Events::RemoveTickGeneratorUsage*)e);
218 break;
219 case Events::EventType::AppTick:
220 handled = OnAppTickEvent(*(Events::AppTickEvent*)e);
221 break;
222 }
223
224 if (handled)
225 {
226 return;
227 }
228
229 if (s_ActiveEngine->m_CurrentApp)
230 {
231 s_ActiveEngine->m_CurrentApp->OnApplicationEvent(e);
232 }
233
234 return;
235 }
236
237 if (e->IsInCategory(Events::Editor))
238 {
239 if (s_ActiveEngine->m_CurrentApp)
240 {
241 s_ActiveEngine->m_CurrentApp->OnEditorEvent(e);
242 }
243 return;
244 }
245
246 }

268 bool EngineService::OnUpdateEntityLocation(Events::UpdateEntityLocation& e)
269 {
270 Ref<Scenes::Scene> scene = Scenes::SceneService::GetActiveScene();
271 if (!scene) { return false; }
272 ECS::Entity entity = scene->GetEntityByUUID(e.GetEntityID());
273 if (!entity) { return false; }
274 Math::vec3 translation = e.GetTranslation();
275 entity.GetComponent<ECS::TransformComponent>().Translation = translation;
276
277 if (entity.HasComponent<ECS::Rigidbody2DComponent>())
278 {
279 auto& rb2d = entity.GetComponent<ECS::Rigidbody2DComponent>();
280 b2Body* body = (b2Body*)rb2d.RuntimeBody;
281 body->SetTransform({ translation.x, translation.y }, body->GetAngle());
282 }
283 return false;

287 {
288 Ref<Scenes::Scene> scene = Scenes::SceneService::GetActiveScene();
289 if (!scene) { return false; }
290 ECS::Entity entity = scene->GetEntityByUUID(e.GetEntityID());
291 if (!entity) { return false; }

293 Math::vec2 linearVelocity = e.GetLinearVelocity();
294 entity.GetComponent<ECS::TransformComponent>().Translation = translation;

299 b2Body* body = (b2Body*)rb2d.RuntimeBody;
300 body->SetTransform({ translation.x, translation.y }, body->GetAngle());
301 body->SetLinearVelocity(b2Vec2(linearVelocity.x, linearVelocity.y));

393 {
394 KG_PROFILE_FUNCTION();
