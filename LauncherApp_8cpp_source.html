84 void LauncherApp::OnUpdate(Timestep ts)
85 {
86 EditorUI::EditorUIService::StartRendering();
87
88 // We are using the ImGuiWindowFlags_NoDocking flag to make the parent window not dockable into,
89 // because it would be confusing to have two docking targets within each others.
90 ImGuiWindowFlags window_flags = ImGuiWindowFlags_NoDocking;
91
92 const ImGuiViewport* viewport = ImGui::GetMainViewport();
93 ImGui::SetNextWindowPos(viewport->WorkPos);
94 ImGui::SetNextWindowSize(viewport->WorkSize);
95 ImGui::SetNextWindowViewport(viewport->ID);
96 ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0.0f);
97 ImGui::PushStyleVar(ImGuiStyleVar_WindowBorderSize, 0.0f);
98 window_flags |= ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoCollapse | ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoMove;
99 window_flags |= ImGuiWindowFlags_NoBringToFrontOnFocus | ImGuiWindowFlags_NoNavFocus;
100
101 // Important: note that we proceed even if Begin() returns false (aka window is collapsed).
102 // This is because we want to keep our DockSpace() active. If a DockSpace() is inactive,
103 // all active windows docked into it will lose their parent and become undocked.
104 // We cannot preserve the docking relationship between an active window and an inactive docking, otherwise
105 // any change of dockspace/settings would lead to windows being stuck in limbo and never being visible.
106
107 ImGui::PushStyleVar(ImGuiStyleVar_WindowPadding, ImVec2(0.0f, 0.0f));
108 EditorUI::EditorUIService::StartWindow("Launcher Screen", nullptr, window_flags);
109 ImGui::PopStyleVar(3);
110
111 EditorUI::EditorUIService::NewItemScreen(
112 "Open Existing Project", [&]()
113 {
114 SelectProject();
115 if (!m_SelectedProject.empty())
116 {
117 EngineService::EndRun();
118 }
119 },
120 "Create New Project", [&]()
121 {
122 s_CreateProjectSpec.OpenPopup = true;
123 });
124
125 EditorUI::EditorUIService::GenericPopup(s_CreateProjectSpec);
126
127
128
129 ImGui::PushStyleColor(ImGuiCol_Button, EditorUI::EditorUIService::s_PureEmpty);
130 ImDrawList* draw_list = ImGui::GetWindowDrawList();
131 ImVec2 initialScreenCursorPos = ImGui::GetWindowPos() + ImGui::GetCursorStartPos();
132 ImVec2 initialCursorPos = ImGui::GetCursorStartPos();
133
134
135 // Draw Background
136 ImVec2 windowSize = ImGui::GetWindowSize();
137 Ref<Rendering::Texture2D> icon {nullptr};
138 draw_list->AddRectFilled(ImVec2(initialScreenCursorPos.x + windowSize.x - 30.0f, initialScreenCursorPos.y),
139 ImVec2(initialScreenCursorPos.x + (windowSize.x), initialScreenCursorPos.y + 30.0f),
140 ImColor(EditorUI::EditorUIService::s_DarkBackgroundColor), 12.0f, ImDrawFlags_RoundCornersBottomLeft);
141
142 // Generate Download Image
143 icon = EditorUI::EditorUIService::s_IconDown;
144 ImGui::SetCursorPos(ImVec2(initialCursorPos.x + windowSize.x - 25, initialCursorPos.y + 4));
145 if (ImGui::ImageButton("Download Samples Button",
146 (ImTextureID)(uint64_t)icon->GetRendererID(),
147 ImVec2(14, 14), ImVec2{ 0, 1 }, ImVec2{ 1, 0 },
148 EditorUI::EditorUIService::s_PureEmpty,
149 EditorUI::EditorUIService::s_HighlightColor1))
150 {
151 ImGui::OpenPopup("Download Samples Popup");
152 }
153 if (ImGui::IsItemHovered())
154 {
155 ImGui::BeginTooltip();

157 ImGui::EndTooltip();
158 }
159
160
161 if (ImGui::BeginPopup("Download Samples Popup"))
162 {
163 if (ImGui::MenuItem("Get Pong"))
