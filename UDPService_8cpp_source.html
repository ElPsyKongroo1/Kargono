54  void UDPService::ReadMessage()
55  {
56  m_Socket.async_receive_from(asio::buffer(s_SocketReceiveBuffer.data(), k_MaxBufferSize), m_CurrentEndpoint,
57  [this](std::error_code ec, std::size_t length)
58  {
59  if (ec)
60  {
61  KG_ERROR("Failure to read UDP Message! {}", ec.message());
62  //Disconnect(m_CurrentEndpoint);
63  ReadMessage();
64  return;
65  }
66 
67  // Make sure CRC and Header can be read. Packet might have been broken apart.
68  if (length < (sizeof(uint32_t) + sizeof(MessageHeader)))
69  {
70  ReadMessage();
71  return;
72  }
73 
74  // Load in Message header from buffer
75  memcpy_s(&m_MsgTemporaryIn.Header,
76  sizeof(MessageHeader),
77  s_SocketReceiveBuffer.data() + sizeof(uint32_t),
78  sizeof(MessageHeader));
79 
80  // Get Payload Size
81  uint64_t payloadSize = m_MsgTemporaryIn.Header.PayloadSize;
82 
83  // Calculate Cyclic Redundency Check and compare with received value
84  uint32_t receivedCRC{};
85 
86  memcpy_s(&receivedCRC, sizeof(uint32_t), s_SocketReceiveBuffer.data(), sizeof(uint32_t));
87 
88  uint32_t currentCRC = Utility::FileSystem::ChecksumCRCFromBuffer(s_SocketReceiveBuffer.data() + sizeof(uint32_t),
89  sizeof(MessageHeader) + payloadSize);
90 
91  // If cyclic redundency check fails, move on! It happens...
92  if (currentCRC != receivedCRC)
93  {
94  ReadMessage();
95  return;
96  }
97 
98  // If CRC and Header are valid and payload is too big, remove connection.
99  if (payloadSize > k_MaxBufferSize)
100  {
101  KG_ERROR("Payload size of received UDP message is larger than buffer!");
102  Disconnect(m_CurrentEndpoint);
103  return;
104  }
105 
106  // Load in Payload into outgoing message!
107  if (payloadSize > 0)
108  {
109  m_MsgTemporaryIn.Body.resize(payloadSize);
110  memcpy_s(m_MsgTemporaryIn.Body.data(),
111  payloadSize,
112  s_SocketReceiveBuffer.data() + sizeof(MessageHeader) + sizeof(uint32_t),
113  payloadSize);
114  }
115 
116  AddToIncomingMessageQueue();
117 
118  });
119  }

121  void UDPService::WriteMessage()
122  {
123  uint64_t payloadSize = m_qMessagesOut.front().msg.Header.PayloadSize;
124 
125  if (m_qMessagesOut.front().msg.Size() > k_MaxBufferSize)
126  {
127  KG_ERROR("Attempt to write UDP message that is large that max buffer size!");
128  Disconnect(m_CurrentEndpoint);
129  return;
130  }
131 
132  // Load Header after CRC location
133  memcpy_s(s_SocketSendBuffer.data() + sizeof(uint32_t),
134  sizeof(MessageHeader),
135  &m_qMessagesOut.front().msg.Header,
136  sizeof(MessageHeader));
137 
138  // Load Buffer after Header
139  if (payloadSize > 0)
140  {
141  memcpy_s(s_SocketSendBuffer.data() + sizeof(uint32_t) + sizeof(MessageHeader),
142  payloadSize,
143  m_qMessagesOut.front().msg.Body.data(),
144  payloadSize);
145  }
146 
147  uint32_t hash = Utility::FileSystem::ChecksumCRCFromBuffer(s_SocketSendBuffer.data() + sizeof(uint32_t),
148  sizeof(MessageHeader) + payloadSize);
149 
150  // Fill CRC location
151  memcpy_s(s_SocketSendBuffer.data(), sizeof(uint32_t), &hash, sizeof(uint32_t));
152 
153  /*KG_TRACE("The local endpoint address/port are {} {} and remote is {} {}", m_Socket.local_endpoint().address().to_string(), m_Socket.local_endpoint().port(),
154  m_qMessagesOut.front().endpoint.address().to_string(), m_qMessagesOut.front().endpoint.port());*/
155 
156  m_Socket.async_send_to(asio::buffer(s_SocketSendBuffer.data(), sizeof(uint32_t) + sizeof(MessageHeader) + payloadSize), m_qMessagesOut.front().endpoint, [this](std::error_code ec, std::size_t length)
157  {
158  if (!ec)
159  {
160  m_qMessagesOut.pop_front();
161 
162  if (!m_qMessagesOut.empty())
163  {
164  WriteMessage();
165  }
166  }
167  else
168  {
169  KG_ERROR("Failure to write Header!");
170  Disconnect(m_CurrentEndpoint);
171  }
172  });
173  }
