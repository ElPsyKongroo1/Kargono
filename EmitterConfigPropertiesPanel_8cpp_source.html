229 EditorUI::EditorUIService::EditFloat(m_EmitterLifetimeSpec);

233 EditorUI::EditorUIService::EditFloat(m_ParticleLifetimeSpec);

237 EditorUI::EditorUIService::EditInteger(m_BufferSizeSpec);
