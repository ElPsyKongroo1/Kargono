52  void AudioEngine::PlayStereoSound(Ref<AudioBuffer> audioBuffer)
53  {
54  static ALfloat forwardAndUpVectors[] =
55  {
56  0, 0, 1, // Forward Vectors
57  0, 1, 0 // Up Vectors
58  };
59 
60  auto audioSource = s_AudioContext->m_StereoMusicSource.get();
61  uint32_t sourceID = audioSource->GetSourceID();
62 
63  alec(alSourceStop(sourceID));
64  if (!audioBuffer) { return; }
65  alec(alSourcei(sourceID, AL_BUFFER, audioBuffer->m_BufferID));
66  alec(alListener3f(AL_POSITION, 0, 0, 0));
67  alec(alListener3f(AL_VELOCITY, 0, 0, 0));
68  alec(alListenerfv(AL_ORIENTATION, forwardAndUpVectors));
69  alec(alSourcePlay(sourceID));
70  }

79  void AudioEngine::PlaySound(const AudioSourceSpecification& sourceSpec, const AudioListenerSpecification& listenerSpec)
80  {
81  auto audioSource = s_AudioContext->m_AudioSourceQueue.front();
82  uint32_t sourceID = audioSource->GetSourceID();
83 
84  alec(alSourceStop(sourceID));
85  s_AudioContext->m_AudioSourceQueue.pop();
86  alec(alSource3f(sourceID, AL_POSITION, sourceSpec.Position.x, sourceSpec.Position.y, sourceSpec.Position.z));
87  alec(alSource3f(sourceID, AL_VELOCITY, sourceSpec.Velocity.x, sourceSpec.Velocity.y, sourceSpec.Velocity.z));
88  alec(alSourcef(sourceID, AL_PITCH, sourceSpec.Pitch));
89  alec(alSourcef(sourceID, AL_GAIN, sourceSpec.Gain));
90  alec(alSourcei(sourceID, AL_LOOPING, static_cast<ALint>(sourceSpec.IsLooping)));
91  if (!sourceSpec.CurrentBuffer) { return; }
92  alec(alSourcei(sourceID, AL_BUFFER, sourceSpec.CurrentBuffer->m_BufferID));
93 
94  alec(alListener3f(AL_POSITION, listenerSpec.Position.x, listenerSpec.Position.y, listenerSpec.Position.z));
95  alec(alListener3f(AL_VELOCITY, listenerSpec.Velocity.x, listenerSpec.Velocity.y, listenerSpec.Velocity.z));
96  ALfloat forwardAndUpVectors[] =
97  {
98  listenerSpec.Forward.x, listenerSpec.Forward.y, listenerSpec.Forward.z, // Forward Vectors
99  listenerSpec.Up.x, listenerSpec.Up.y, listenerSpec.Up.z // Up Vectors
100  };
101  alec(alListenerfv(AL_ORIENTATION, forwardAndUpVectors));
102 
103  alec(alSourcePlay(sourceID));
104 
105  s_AudioContext->m_AudioSourceQueue.push(audioSource);
106 
107  }

135  void AudioEngine::Init()
136  {
137  // Find default audio device
138  s_AudioContext->m_CurrentDeviceName = alcGetString(nullptr, ALC_DEFAULT_DEVICE_SPECIFIER);
139  s_AudioContext->m_CurrentDeviceID = alcOpenDevice(s_AudioContext->m_CurrentDeviceName.c_str());
140  KG_ASSERT(s_AudioContext->m_CurrentDeviceID, "Failed to get the default device for OpenAL");
141  //KG_INFO("OpenAL Device: {}", alcGetString(s_AudioContext->m_CurrentDeviceID, ALC_DEVICE_SPECIFIER));
142 
143  // Create an OpenAL audio context from the device
144  s_AudioContext->m_ContextID = alcCreateContext(s_AudioContext->m_CurrentDeviceID, nullptr);
145  //OpenAL_ErrorCheck(context);
146 
147  // Activate this context so that OpenAL state modifications are applied to the context
148  bool makeCurrentValid = alcMakeContextCurrent(s_AudioContext->m_ContextID);
149  KG_ASSERT(makeCurrentValid, "Failed to make the OpenAL context the current context");
150 
151  // Create a listener in 3D space
152  s_AudioContext->m_DefaultListener = CreateScope<AudioListener>();
153 
154  // Initialize all Sound Sources
155  s_AudioContext->m_StereoMusicSource = CreateScope<AudioSource>();
156  alec(alSource3f(s_AudioContext->m_StereoMusicSource->GetSourceID(), AL_POSITION, 0, 0, 0));
157  alec(alSource3f(s_AudioContext->m_StereoMusicSource->GetSourceID(), AL_VELOCITY, 0, 0, 0));
158  alec(alSourcef(s_AudioContext->m_StereoMusicSource->GetSourceID(), AL_PITCH, 1.0f));
159  alec(alSourcef(s_AudioContext->m_StereoMusicSource->GetSourceID(), AL_GAIN, 1.0f));
160  alec(alSourcei(s_AudioContext->m_StereoMusicSource->GetSourceID(), AL_LOOPING, true));
161 
162  for (uint32_t iterator{0}; iterator < 15; iterator++)
163  {
164  s_AudioContext->m_AudioSourceQueue.push(CreateRef<AudioSource>());
165  }
166  KG_VERIFY(s_AudioContext, "Audio Engine Init");
167  }

169  void AudioEngine::Terminate()
170  {
171  s_DefaultSourceSpec.CurrentBuffer.reset();
172  s_AudioContext->m_StereoMusicSource.reset();
173  while (!s_AudioContext->m_AudioSourceQueue.empty())
174  {
175  KG_ASSERT(s_AudioContext->m_AudioSourceQueue.front().use_count() == 1, "Not all Audio Resources have been cleared!");
176  s_AudioContext->m_AudioSourceQueue.front().reset();
177  s_AudioContext->m_AudioSourceQueue.pop();
178  }
179  Assets::AssetManager::ClearAudioRegistry();
180 
181  // Close OpenAL Context
182  alcDestroyContext(s_AudioContext->m_ContextID);
183  alcMakeContextCurrent(nullptr);
184  alcCloseDevice(s_AudioContext->m_CurrentDeviceID);
185  s_AudioContext->m_CurrentDeviceName = "";
186  s_AudioContext->m_CurrentDeviceID = nullptr;
187  s_AudioContext->m_ContextID = nullptr;
188  s_AudioContext = nullptr;
189  KG_VERIFY(!s_AudioContext, "Close Audio Engine");
190  }
