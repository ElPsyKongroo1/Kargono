158 private:
159 void CalculateOffsetsAndStride ()
160 {
161 m_Stride = 0;
162 m_ElementLocations.clear();
163 if (m_Elements.empty()) { return; }
164
165 size_t offset = 0;
166 uint32_t iteration = 0;
167 for (auto& element : m_Elements)
168 {
169 m_ElementLocations.insert({ element.Name, static_cast<std::uint32_t>(iteration) });
170 element.Offset = offset;
171 offset += element.Size;
172 m_Stride += element.Size;
173 iteration++;
174 }
175 }
176 private:
177 std::vector<InputBufferElement> m_Elements {};
178 std::unordered_map<std::string, uint32_t> m_ElementLocations {};
