86 void AudioService::PlaySound(const AudioSourceSpecification& sourceSpec, const AudioListenerSpecification& listenerSpec)
87 {
88 if (s_AudioContext->Mute)
89 {
90 return;
91 }
92 auto audioSource = s_AudioContext->AudioSourceQueue.front();
93 uint32_t sourceID = audioSource->GetSourceID();
94
95 CallAndCheckALError(alSourceStop(sourceID));
96 s_AudioContext->AudioSourceQueue.pop();
97 CallAndCheckALError(alSource3f(sourceID, AL_POSITION, sourceSpec.Position.x, sourceSpec.Position.y, sourceSpec.Position.z));
98 CallAndCheckALError(alSource3f(sourceID, AL_VELOCITY, sourceSpec.Velocity.x, sourceSpec.Velocity.y, sourceSpec.Velocity.z));
99 CallAndCheckALError(alSourcef(sourceID, AL_PITCH, sourceSpec.Pitch));
100 CallAndCheckALError(alSourcef(sourceID, AL_GAIN, sourceSpec.Gain));
101 CallAndCheckALError(alSourcei(sourceID, AL_LOOPING, static_cast<ALint>(sourceSpec.IsLooping)));
102 if (!sourceSpec.CurrentBuffer) { return; }
103 CallAndCheckALError(alSourcei(sourceID, AL_BUFFER, sourceSpec.CurrentBuffer->m_BufferID));
104
105 CallAndCheckALError(alListener3f(AL_POSITION, listenerSpec.Position.x, listenerSpec.Position.y, listenerSpec.Position.z));
106 CallAndCheckALError(alListener3f(AL_VELOCITY, listenerSpec.Velocity.x, listenerSpec.Velocity.y, listenerSpec.Velocity.z));
107 ALfloat forwardAndUpVectors[] =
108 {
109 listenerSpec.Forward.x, listenerSpec.Forward.y, listenerSpec.Forward.z, // Forward Vectors
110 listenerSpec.Up.x, listenerSpec.Up.y, listenerSpec.Up.z // Up Vectors
111 };
112 CallAndCheckALError(alListenerfv(AL_ORIENTATION, forwardAndUpVectors));
113
114 CallAndCheckALError(alSourcePlay(sourceID));
115
116 s_AudioContext->AudioSourceQueue.push(audioSource);
