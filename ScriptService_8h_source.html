84 inline std::string GenerateFunctionSignature(WrappedFuncType funcType, const std::string& name, const Scripting::ExplicitFuncType& explicitType)
85 {
86 // Initialize return type and parameter list depending on provided types
87 WrappedVarType returnType;
88 std::vector<WrappedVarType> parameterTypes;
89 std::vector<FixedString32> parameterNames;
90 if (funcType == WrappedFuncType::ArbitraryFunction)
91 {
92 returnType = explicitType.m_ReturnType;
93 parameterTypes = explicitType.m_ParameterTypes;
94 parameterNames = explicitType.m_ParameterNames;
95 }
96 else
97 {
98 returnType = Utility::WrappedFuncTypeToReturnType(funcType);
99 parameterTypes = Utility::WrappedFuncTypeToParameterTypes(funcType);
100
101 // Check if a list of parameter names is available and is the correct size
102 if (explicitType.m_ParameterNames.size() == parameterTypes.size())
103 {
104 // Simply use the existing names if available
105 parameterNames = explicitType.m_ParameterNames;
106 }
107 else
108 {
109 // Fill parameterNames with default names
110 char letterIteration{ 'a' };
111 for (WrappedVarType type : parameterTypes)
112 {
113 parameterNames.emplace_back(letterIteration);
114 letterIteration++;
115 }
116 }
117 }
118
119 // Write out return value and function name
120 std::stringstream outputStream {};
121 outputStream << Utility::WrappedVarTypeToKGScript(returnType) << " " << name<< "(";
122
123 // Write out parameters into function signature
124 for (std::size_t iteration{ 0 }; static_cast<size_t>(iteration) < parameterTypes.size(); iteration++)
125 {
126 outputStream << Utility::WrappedVarTypeToKGScript(parameterTypes.at(iteration)) << " " << parameterNames.at(iteration);
127 if (iteration != parameterTypes.size() - 1)
128 {
129 outputStream << ',';
130 }
131 }
132 outputStream << ")";
133 return outputStream.str();
134 }

137 inline std::string GenerateFunctionStub(WrappedFuncType funcType, const std::string& name, const Scripting::ExplicitFuncType& explicitType)
138 {
139 std::stringstream outputStream {};
140 outputStream << GenerateFunctionSignature(funcType, name, explicitType);
141 outputStream << '\n';
142 outputStream << "{" << "\n";
143
144 // Get current return type
145 WrappedVarType returnType;
146 if (funcType == WrappedFuncType::ArbitraryFunction)
147 {
148 returnType = explicitType.m_ReturnType;
149 }
150 else
151 {
152 returnType = Utility::WrappedFuncTypeToReturnType(funcType);
153 }
154
155 // Write return statement if necessary
156 if (returnType != WrappedVarType::Void)
157 {
158 outputStream << "\treturn ";
159 switch (returnType)
160 {
161 case WrappedVarType::Bool:
162 {
163 outputStream << "false;\n";
164 break;
165 }
166 case WrappedVarType::String:
167 {
168 outputStream << "\"\";\n";
169 break;
170 }
171 case WrappedVarType::Vector3:
172 {
173 outputStream << "Math::vec3(0.0f, 0.0f, 0.0f);\n";
174 break;
175 }
176 case WrappedVarType::Float:
177 {
178 outputStream << "0.0f;\n";
179 break;
180 }
181 case WrappedVarType::UInteger16:
182 case WrappedVarType::UInteger32:
183 case WrappedVarType::Integer32:
184 case WrappedVarType::UInteger64:
185 {
186 outputStream << "0;\n";
187 break;
188 }
189
190 case WrappedVarType::Entity:
191 {
192 outputStream << "0;\n";
193 break;
194 }
195 case WrappedVarType::Void:
196 case WrappedVarType::None:
197 default:
198 {
199 KG_ERROR("Unsupported return type provided {}", Utility::WrappedVarTypeToString(returnType));
200 break;
201 }
202 }
203 }
