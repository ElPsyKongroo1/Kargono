127 // atlasPacker.setDimensionsConstraint();
128 atlasPacker.setPixelRange(2.0);
129 atlasPacker.setMiterLimit(1.0);
130 atlasPacker.setPadding(0);
131 atlasPacker.setScale(emSize);
132 int32_t remaining = atlasPacker.pack(glyphs.data(), (int32_t)glyphs.size());
133 KG_ASSERT(remaining == 0);
134
135 int32_t width, height;
136 atlasPacker.getDimensions(width, height);
137 emSize = atlasPacker.getScale();
138 uint32_t numAvailableThread = std::thread::hardware_concurrency() / 2;
139#define DEFAULT_ANGLE_THRESHOLD 3.0
140#define LCG_MULTIPLIER 6364136223846793005ull
141#define LCG_INCREMENT 1442695040888963407ull
142
143 // if MSDF || MTSDF
144 uint64_t coloringSeed = 0;
145 bool expensiveColoring = false;
146 if (expensiveColoring)
147 {
148 msdf_atlas::Workload([&glyphs = glyphs, &coloringSeed](int i, int threadNo) -> bool {
149 unsigned long long glyphSeed = (LCG_MULTIPLIER * (coloringSeed ^ i) + LCG_INCREMENT) * !!coloringSeed;
150 glyphs[i].edgeColoring(msdfgen::edgeColoringInkTrap, DEFAULT_ANGLE_THRESHOLD, glyphSeed);
151 return true;
152 }, static_cast<int32_t>(glyphs.size())).finish(numAvailableThread);
153 }
154 else {
155 unsigned long long glyphSeed = coloringSeed;
156 for (msdf_atlas::GlyphGeometry& glyph : glyphs)
157 {
158 glyphSeed *= LCG_MULTIPLIER;
159 glyph.edgeColoring(msdfgen::edgeColoringInkTrap, DEFAULT_ANGLE_THRESHOLD, glyphSeed);
160 }
161 }
162
163
164 newFont->m_AtlasTexture = Utility::CreateAndCacheAtlas<uint8_t, float, 3, msdf_atlas::msdfGenerator>("Test", (float)emSize, glyphs, fontGeometry, width, height);
165
166 msdfgen::destroyFont(font);
167 msdfgen::deinitializeFreetype(ft);
168
169 const auto& metrics = fontGeometry.getMetrics();
170 newFont->m_LineHeight = static_cast<float>(metrics.lineHeight);
171
172 const auto& glyphMetrics = fontGeometry.getGlyphs();
173 newFont->m_Characters.clear();
174 for (auto& glyphGeometry : glyphMetrics)
175 {
176 unsigned char character = static_cast<uint8_t>(glyphGeometry.getCodepoint());
177 Character characterStruct{};
178
179 // Fill the texture location inside Atlas
180 double al, ab, ar, at;
181 glyphGeometry.getQuadAtlasBounds(al, ab, ar, at);
182 characterStruct.TexCoordinateMin = { (float)al, (float)ab };
183 characterStruct.TexCoordinateMax = { (float)ar, (float)at };
184 // Fill the Bounding Box Size when Rendering
185 double pl, pb, pr, pt;
186 glyphGeometry.getQuadPlaneBounds(pl, pb, pr, pt);
187 characterStruct.QuadMin = { (float)pl, (float)pb };
188 characterStruct.QuadMax = { (float)pr, (float)pt };
189 // Fill the Advance
190 characterStruct.Advance = (float)glyphGeometry.getAdvance();
191 // Fill Glyph Size
192 int32_t glyphWidth, glyphHeight;
193 glyphGeometry.getBoxSize(glyphWidth, glyphHeight);
194 characterStruct.Size = { glyphWidth, glyphHeight };
195 newFont->m_Characters.insert_or_assign(character, characterStruct);
196 }
197
198 return newFont;
