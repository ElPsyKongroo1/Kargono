16 void AssetManager::DeserializeTextureRegistry()
17 {
18 // Clear current registry and open registry in current project
19 s_TextureRegistry.clear();
20 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project to serialize from!");
21 const auto& textureRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Textures/Intermediates/TextureRegistry.kgreg";
22
23 if (!std::filesystem::exists(textureRegistryLocation))
24 {
25 KG_WARN("No .kgregistry file exists in project path!");
26 return;
27 }
28 YAML::Node data;
29 try
30 {
31 data = YAML::LoadFile(textureRegistryLocation.string());
32 }
33 catch (YAML::ParserException e)
34 {
35 KG_ERROR("Failed to load .kgscene file '{0}'\n {1}", textureRegistryLocation.string(), e.what());
36 return;
37 }
38
39 // Opening registry node
40 if (!data["Registry"]) { return; }
41
42 std::string registryName = data["Registry"].as<std::string>();
43 KG_INFO("Deserializing Texture Registry");
44
45 // Opening all assets
46 auto assets = data["Assets"];
47 if (assets)
48 {
49 for (auto asset : assets)
50 {
51 Assets::Asset newAsset{};
52 newAsset.Handle = asset["AssetHandle"].as<uint64_t>();
53
54 // Retrieving metadata for asset
55 auto metadata = asset["MetaData"];
56 newAsset.Data.CheckSum = metadata["CheckSum"].as<std::string>();
57 newAsset.Data.IntermediateLocation = metadata["IntermediateLocation"].as<std::string>();
58 newAsset.Data.Type = Utility::StringToAssetType(metadata["AssetType"].as<std::string>());
59
60 // Retrieving texture specific metadata
61 if (newAsset.Data.Type == Assets::Texture)
62 {
63 Ref<Assets::TextureMetaData> texMetaData = CreateRef<Assets::TextureMetaData>();
64
65 texMetaData->Height = metadata["TextureHeight"].as<int32_t>();
66 texMetaData->Width = metadata["TextureWidth"].as<int32_t>();
67 texMetaData->Channels = metadata["TextureChannels"].as<int32_t>();
68 texMetaData->InitialFileLocation = metadata["InitialFileLocation"].as<std::string>();
69
70 newAsset.Data.SpecificFileData = texMetaData;
71 }
72
73 // Add asset to in memory registry
74 s_TextureRegistry.insert({ newAsset.Handle, newAsset });
75
76 }
77 }

80 void AssetManager::SerializeTextureRegistry()
81 {
82 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project to serialize to!");
83 const auto& textureRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Textures/Intermediates/TextureRegistry.kgreg";
84 YAML::Emitter out;
85
86 out << YAML::BeginMap;
87 out << YAML::Key << "Registry" << YAML::Value << "Untitled";
88 out << YAML::Key << "Assets" << YAML::Value << YAML::BeginSeq;
89
90 // Asset
91 for (auto& [handle, asset] : s_TextureRegistry)
92 {
93 out << YAML::BeginMap; // Asset Map
94 out << YAML::Key << "AssetHandle" << YAML::Value << static_cast<uint64_t>(handle);
95
96 out << YAML::Key << "MetaData" << YAML::Value;
97 out << YAML::BeginMap; // MetaData Map
98 out << YAML::Key << "CheckSum" << YAML::Value << asset.Data.CheckSum;
99 out << YAML::Key << "IntermediateLocation" << YAML::Value << asset.Data.IntermediateLocation.string();
100 out << YAML::Key << "AssetType" << YAML::Value << Utility::AssetTypeToString(asset.Data.Type);
101
102 if (asset.Data.Type == Assets::AssetType::Texture)
103 {
104 Assets::TextureMetaData* metadata = static_cast<Assets::TextureMetaData*>(asset.Data.SpecificFileData.get());
105 out << YAML::Key << "TextureHeight" << YAML::Value << metadata->Height;
106 out << YAML::Key << "TextureWidth" << YAML::Value << metadata->Width;
107 out << YAML::Key << "TextureChannels" << YAML::Value << metadata->Channels;
108 out << YAML::Key << "InitialFileLocation" << YAML::Value << metadata->InitialFileLocation.string();
109 }
110
111 out << YAML::EndMap; // MetaData Map
112
113 out << YAML::EndMap; // Asset Map
114 }
115
116 out << YAML::EndSeq;
117 out << YAML::EndMap;
118
119 Utility::FileSystem::CreateNewDirectory(textureRegistryLocation.parent_path());
120
121 std::ofstream fout(textureRegistryLocation);
122 fout << out.c_str();

125 AssetHandle AssetManager::ImportNewTextureFromFile(const std::filesystem::path& filePath)
126 {
127 // Create Checksum
128 std::string currentCheckSum = Utility::FileSystem::ChecksumFromFile(filePath);
129
130 if (currentCheckSum.empty())
131 {
132 KG_ERROR("Failed to generate checksum from file!");
133 return {};
134 }
135
136 // Compare currentChecksum to registered assets
137 bool isAssetDuplicate = false;
138 AssetHandle currentHandle{};
139 for (const auto& [handle, asset] : s_TextureRegistry)
140 {
141 if (asset.Data.CheckSum == currentCheckSum)
142 {
143 isAssetDuplicate = true;
144 currentHandle = handle;
145 break;
146 }
147 }
148
149 if (isAssetDuplicate)
150 {
151 KG_ERROR("THERE IS A DUPLICATE!");
152 return currentHandle;
153 }
154
155 // Create New Asset/Handle
156 AssetHandle newHandle{};
157
158 Assets::Asset newAsset{};
159 newAsset.Handle = newHandle;
160
161 // Create Intermediate
162 CreateTextureIntermediateFromFile(filePath, newAsset);
163 newAsset.Data.CheckSum = currentCheckSum;
164
165 // Register New Asset and Create Texture
166 s_TextureRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
167 SerializeTextureRegistry(); // Update Registry File on Disk
168
169 s_Textures.insert({ newHandle, InstantiateTextureIntoMemory(newAsset) });
170
171 return newHandle;
172
173 }

225 void AssetManager::CreateTextureIntermediateFromFile(const std::filesystem::path& filePath, Assets::Asset& newAsset)
226 {
227 // Create Texture Binary Intermediate
228 int32_t width, height, channels;
229 stbi_set_flip_vertically_on_load(1);
230 Buffer buffer{};
231 stbi_uc* data = nullptr;
232 {
233 data = stbi_load(filePath.string().c_str(), &width, &height, &channels, 0);
234 }
235
236 buffer.Allocate(static_cast<unsigned long long>(width) * height * channels * sizeof(uint8_t));
237 buffer.Data = data;
238
239 // Save Binary Intermediate into File
240 std::string intermediatePath = "Textures/Intermediates/" + (std::string)newAsset.Handle + ".kgtexture";
241 std::filesystem::path intermediateFullPath = Projects::ProjectService::GetActiveAssetDirectory() / intermediatePath;
242 Utility::FileSystem::WriteFileBinary(intermediateFullPath, buffer);
243
244 // Check that save was successful
245 if (!data)
246 {
247 KG_ERROR("Failed to load data from file in texture importer!");
248 buffer.Release();
249 return;
250 }
251
252 // Load data into In-Memory Metadata object
253 newAsset.Data.Type = Assets::AssetType::Texture;
254 newAsset.Data.IntermediateLocation = intermediatePath;
255 Ref<Assets::TextureMetaData> metadata = CreateRef<Assets::TextureMetaData>();
256 metadata->Width = width;
257 metadata->Height = height;
258 metadata->Channels = channels;
259 metadata->InitialFileLocation = Utility::FileSystem::GetRelativePath(Projects::ProjectService::GetActiveAssetDirectory(), filePath);
260 newAsset.Data.SpecificFileData = metadata;
261
262 buffer.Release();
