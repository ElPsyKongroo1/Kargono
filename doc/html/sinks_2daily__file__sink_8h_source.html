26/*
27 * Generator of daily log file names in format basename.YYYY-MM-DD.ext
28 */
29struct daily_filename_calculator
30{
31 // Create filename for the form basename.YYYY-MM-DD
32 static filename_t calc_filename(const filename_t &filename, const tm &now_tm)
33 {
34 filename_t basename, ext;
35 std::tie(basename, ext) = details::file_helper::split_by_extension(filename);
36 return fmt_lib::format(SPDLOG_FMT_STRING(SPDLOG_FILENAME_T("{}_{:04d}-{:02d}-{:02d}{}")), basename, now_tm.tm_year + 1900,
37 now_tm.tm_mon + 1, now_tm.tm_mday, ext);
38 }
39};

41/*
42 * Generator of daily log file names with strftime format.
43 * Usages:
44 * auto sink = std::make_shared<spdlog::sinks::daily_file_format_sink_mt>("myapp-%Y-%m-%d:%H:%M:%S.log", hour, minute);"
45 * auto logger = spdlog::daily_logger_format_mt("loggername, "myapp-%Y-%m-%d:%X.log", hour, minute)"
46 *
47 */
48struct daily_filename_format_calculator
49{
50 static filename_t calc_filename(const filename_t &file_path, const tm &now_tm)
51 {
52#if defined(_WIN32) && defined(SPDLOG_WCHAR_FILENAMES)
53 std::wstringstream stream;
54#else
55 std::stringstream stream;
56#endif
57 stream << std::put_time(&now_tm, file_path.c_str());
58 return stream.str();
59 }
60};

62/*
63 * Rotating file sink based on date.
64 * If truncate != false , the created file will be truncated.
65 * If max_files > 0, retain only the last max_files and delete previous.
66 */
67template<typename Mutex, typename FileNameCalc = daily_filename_calculator>
68class daily_file_sink final : public base_sink<Mutex>
69{
70public:
71 // create daily file sink which rotates on given time
72 daily_file_sink(filename_t base_filename, int rotation_hour, int rotation_minute, bool truncate = false, uint16_t max_files = 0,
73 const file_event_handlers &event_handlers = {})
74 : base_filename_(std::move(base_filename))
75 , rotation_h_(rotation_hour)
76 , rotation_m_(rotation_minute)
77 , file_helper_{event_handlers}
78 , truncate_(truncate)
79 , max_files_(max_files)
80 , filenames_q_()
81 {
82 if (rotation_hour < 0 || rotation_hour > 23 || rotation_minute < 0 || rotation_minute > 59)
83 {
84 throw_spdlog_ex("daily_file_sink: Invalid rotation time in ctor");
85 }
86
87 auto now = log_clock::now();
88 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
89 file_helper_.open(filename, truncate_);
90 rotation_tp_ = next_rotation_tp_();
91
92 if (max_files_ > 0)
93 {
94 init_filenames_q_();
95 }
96 }
97
98 filename_t filename()
99 {
100 std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
101 return file_helper_.filename();
102 }
103
104protected:
105 void sink_it_(const details::log_msg &msg) override
106 {
107 auto time = msg.time;
108 bool should_rotate = time >= rotation_tp_;
109 if (should_rotate)
110 {
111 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(time));
112 file_helper_.open(filename, truncate_);
113 rotation_tp_ = next_rotation_tp_();
114 }
115 memory_buf_t formatted;
116 base_sink<Mutex>::formatter_->format(msg, formatted);
117 file_helper_.write(formatted);
118
119 // Do the cleaning only at the end because it might throw on failure.
120 if (should_rotate && max_files_ > 0)
121 {
122 delete_old_();
123 }
124 }
125
126 void flush_() override
127 {
128 file_helper_.flush();
129 }
130
131private:
132 void init_filenames_q_()
133 {
134 using details::os::path_exists;
135
136 filenames_q_ = details::circular_q<filename_t>(static_cast<size_t>(max_files_));
137 std::vector<filename_t> filenames;
138 auto now = log_clock::now();
139 while (filenames.size() < max_files_)
140 {
141 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
142 if (!path_exists(filename))
143 {
144 break;
145 }
146 filenames.emplace_back(filename);
147 now -= std::chrono::hours(24);
148 }
149 for (auto iter = filenames.rbegin(); iter != filenames.rend(); ++iter)
150 {
151 filenames_q_.push_back(std::move(*iter));
152 }
153 }
154
155 tm now_tm(log_clock::time_point tp)
156 {
157 time_t tnow = log_clock::to_time_t(tp);
158 return spdlog::details::os::localtime(tnow);
159 }
160
161 log_clock::time_point next_rotation_tp_()
162 {
163 auto now = log_clock::now();
164 tm date = now_tm(now);
165 date.tm_hour = rotation_h_;
166 date.tm_min = rotation_m_;
167 date.tm_sec = 0;
168 auto rotation_time = log_clock::from_time_t(std::mktime(&date));
169 if (rotation_time > now)
170 {
171 return rotation_time;
172 }
173 return {rotation_time + std::chrono::hours(24)};
174 }
175
176 // Delete the file N rotations ago.
177 // Throw spdlog_ex on failure to delete the old file.
178 void delete_old_()
179 {
180 using details::os::filename_to_str;
181 using details::os::remove_if_exists;
182
183 filename_t current_file = file_helper_.filename();
184 if (filenames_q_.full())
185 {
186 auto old_filename = std::move(filenames_q_.front());
187 filenames_q_.pop_front();
188 bool ok = remove_if_exists(old_filename) == 0;
189 if (!ok)
190 {
191 filenames_q_.push_back(std::move(current_file));
192 throw_spdlog_ex("Failed removing daily file " + filename_to_str(old_filename), errno);
193 }
194 }
195 filenames_q_.push_back(std::move(current_file));
196 }
197
198 filename_t base_filename_;
199 int rotation_h_;
200 int rotation_m_;
201 log_clock::time_point rotation_tp_;
202 details::file_helper file_helper_;
203 bool truncate_;
204 uint16_t max_files_;
205 details::circular_q<filename_t> filenames_q_;
206};
