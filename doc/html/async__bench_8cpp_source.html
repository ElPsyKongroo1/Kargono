41int count_lines(const char *filename)
42{
43 int counter = 0;
44 auto *infile = fopen(filename, "r");
45 int ch;
46 while (EOF != (ch = getc(infile)))
47 {
48 if ('\n' == ch)
49 counter++;
50 }
51 fclose(infile);
52
53 return counter;
54}

162void bench_mt(int howmany, std::shared_ptr<spdlog::logger> logger, int thread_count)
163{
164 using std::chrono::high_resolution_clock;
165 vector<std::thread> threads;
166 auto start = high_resolution_clock::now();
167
168 int msgs_per_thread = howmany / thread_count;
169 int msgs_per_thread_mod = howmany % thread_count;
170 for (int t = 0; t < thread_count; ++t)
171 {
172 if (t == 0 && msgs_per_thread_mod)
173 threads.push_back(std::thread(thread_fun, logger, msgs_per_thread + msgs_per_thread_mod));
174 else
175 threads.push_back(std::thread(thread_fun, logger, msgs_per_thread));
176 }
177
178 for (auto &t : threads)
179 {
180 t.join();
181 };
182
183 auto delta = high_resolution_clock::now() - start;
184 auto delta_d = duration_cast<duration<double>>(delta).count();
185 spdlog::info("Elapsed: {} secs\t {:L}/sec", delta_d, int(howmany / delta_d));
186}
