88// Terminate the library
89//
90static void terminate(void)
91{
92 int i;
93
94 memset(&_glfw.callbacks, 0, sizeof(_glfw.callbacks));
95
96 while (_glfw.windowListHead)
97 glfwDestroyWindow((GLFWwindow*) _glfw.windowListHead);
98
99 while (_glfw.cursorListHead)
100 glfwDestroyCursor((GLFWcursor*) _glfw.cursorListHead);
101
102 for (i = 0; i < _glfw.monitorCount; i++)
103 {
104 _GLFWmonitor* monitor = _glfw.monitors[i];
105 if (monitor->originalRamp.size)
106 _glfw.platform.setGammaRamp(monitor, &monitor->originalRamp);
107 _glfwFreeMonitor(monitor);
108 }
109
110 _glfw_free(_glfw.monitors);
111 _glfw.monitors = NULL;
112 _glfw.monitorCount = 0;
113
114 _glfw_free(_glfw.mappings);
115 _glfw.mappings = NULL;
116 _glfw.mappingCount = 0;
117
118 _glfwTerminateVulkan();
119 _glfw.platform.terminateJoysticks();
120 _glfw.platform.terminate();
121
122 _glfw.initialized = GLFW_FALSE;
123
124 while (_glfw.errorListHead)
125 {
126 _GLFWerror* error = _glfw.errorListHead;
127 _glfw.errorListHead = error->next;
128 _glfw_free(error);
129 }
130
131 _glfwPlatformDestroyTls(&_glfw.contextSlot);
132 _glfwPlatformDestroyTls(&_glfw.errorSlot);
133 _glfwPlatformDestroyMutex(&_glfw.errorLock);
134
135 memset(&_glfw, 0, sizeof(_glfw));
136}

143// Encode a Unicode code point to a UTF-8 stream
144// Based on cutef8 by Jeff Bezanson (Public Domain)
145//
146size_t _glfwEncodeUTF8(char* s, uint32_t codepoint)
147{
148 size_t count = 0;
149
150 if (codepoint < 0x80)
151 s[count++] = (char) codepoint;
152 else if (codepoint < 0x800)
153 {
154 s[count++] = (codepoint >> 6) | 0xc0;
155 s[count++] = (codepoint & 0x3f) | 0x80;
156 }
157 else if (codepoint < 0x10000)
158 {
159 s[count++] = (codepoint >> 12) | 0xe0;
160 s[count++] = ((codepoint >> 6) & 0x3f) | 0x80;
161 s[count++] = (codepoint & 0x3f) | 0x80;
162 }
163 else if (codepoint < 0x110000)
164 {
165 s[count++] = (codepoint >> 18) | 0xf0;
166 s[count++] = ((codepoint >> 12) & 0x3f) | 0x80;
167 s[count++] = ((codepoint >> 6) & 0x3f) | 0x80;
168 s[count++] = (codepoint & 0x3f) | 0x80;
169 }
170
171 return count;
172}

206void* _glfw_calloc(size_t count, size_t size)
207{
208 if (count && size)
209 {
210 void* block;
211
212 if (count > SIZE_MAX / size)
213 {
214 _glfwInputError(GLFW_INVALID_VALUE, "Allocation size overflow");
215 return NULL;
216 }
217
218 block = _glfw.allocator.allocate(count * size, _glfw.allocator.user);
219 if (block)
220 return memset(block, 0, count * size);
221 else
222 {
223 _glfwInputError(GLFW_OUT_OF_MEMORY, NULL);
224 return NULL;
225 }
226 }
227 else
228 return NULL;
229}

231void* _glfw_realloc(void* block, size_t size)
232{
233 if (block && size)
234 {
235 void* resized = _glfw.allocator.reallocate(block, size, _glfw.allocator.user);
236 if (resized)
237 return resized;
238 else
239 {
240 _glfwInputError(GLFW_OUT_OF_MEMORY, NULL);
241 return NULL;
242 }
243 }
244 else if (block)
245 {
246 _glfw_free(block);
247 return NULL;
248 }
249 else
250 return _glfw_calloc(1, size);
251}

264// Notifies shared code of an error
265//
266void _glfwInputError(int code, const char* format, ...)
267{
268 _GLFWerror* error;
269 char description[_GLFW_MESSAGE_SIZE];
270
271 if (format)
272 {
273 va_list vl;
274
275 va_start(vl, format);
276 vsnprintf(description, sizeof(description), format, vl);
277 va_end(vl);
278
279 description[sizeof(description) - 1] = '\0';
280 }
281 else
282 {
283 if (code == GLFW_NOT_INITIALIZED)
284 strcpy(description, "The GLFW library is not initialized");
285 else if (code == GLFW_NO_CURRENT_CONTEXT)
286 strcpy(description, "There is no current context");
287 else if (code == GLFW_INVALID_ENUM)
288 strcpy(description, "Invalid argument for enum parameter");
289 else if (code == GLFW_INVALID_VALUE)
290 strcpy(description, "Invalid value for parameter");
291 else if (code == GLFW_OUT_OF_MEMORY)
292 strcpy(description, "Out of memory");
293 else if (code == GLFW_API_UNAVAILABLE)
294 strcpy(description, "The requested API is unavailable");
295 else if (code == GLFW_VERSION_UNAVAILABLE)
296 strcpy(description, "The requested API version is unavailable");
297 else if (code == GLFW_PLATFORM_ERROR)
298 strcpy(description, "A platform-specific error occurred");
299 else if (code == GLFW_FORMAT_UNAVAILABLE)
300 strcpy(description, "The requested format is unavailable");
301 else if (code == GLFW_NO_WINDOW_CONTEXT)
302 strcpy(description, "The specified window has no context");
303 else if (code == GLFW_CURSOR_UNAVAILABLE)
304 strcpy(description, "The specified cursor shape is unavailable");
305 else if (code == GLFW_FEATURE_UNAVAILABLE)
306 strcpy(description, "The requested feature cannot be implemented for this platform");
307 else if (code == GLFW_FEATURE_UNIMPLEMENTED)
308 strcpy(description, "The requested feature has not yet been implemented for this platform");
309 else if (code == GLFW_PLATFORM_UNAVAILABLE)
310 strcpy(description, "The requested platform is unavailable");
311 else
312 strcpy(description, "ERROR: UNKNOWN GLFW ERROR");
313 }
314
315 if (_glfw.initialized)
316 {
317 error = _glfwPlatformGetTls(&_glfw.errorSlot);
318 if (!error)
319 {
320 error = _glfw_calloc(1, sizeof(_GLFWerror));
321 _glfwPlatformSetTls(&_glfw.errorSlot, error);
322 _glfwPlatformLockMutex(&_glfw.errorLock);
323 error->next = _glfw.errorListHead;
324 _glfw.errorListHead = error;
325 _glfwPlatformUnlockMutex(&_glfw.errorLock);
326 }
327 }
328 else
329 error = &_glfwMainThreadError;
330
331 error->code = code;
332 strcpy(error->description, description);
333
334 if (_glfwErrorCallback)
335 _glfwErrorCallback(code, description);
336}

343GLFWAPI int glfwInit(void)
344{
345 if (_glfw.initialized)
346 return GLFW_TRUE;
347
348 memset(&_glfw, 0, sizeof(_glfw));
349 _glfw.hints.init = _glfwInitHints;
350
351 _glfw.allocator = _glfwInitAllocator;
352 if (!_glfw.allocator.allocate)
353 {
354 _glfw.allocator.allocate = defaultAllocate;
355 _glfw.allocator.reallocate = defaultReallocate;
356 _glfw.allocator.deallocate = defaultDeallocate;
357 }
358
359 if (!_glfwSelectPlatform(_glfw.hints.init.platformID, &_glfw.platform))
360 return GLFW_FALSE;
361
362 if (!_glfw.platform.init())
363 {
364 terminate();
365 return GLFW_FALSE;
366 }
367
368 if (!_glfwPlatformCreateMutex(&_glfw.errorLock) ||
369 !_glfwPlatformCreateTls(&_glfw.errorSlot) ||
370 !_glfwPlatformCreateTls(&_glfw.contextSlot))
371 {
372 terminate();
373 return GLFW_FALSE;
374 }
375
376 _glfwPlatformSetTls(&_glfw.errorSlot, &_glfwMainThreadError);
377
378 _glfwInitGamepadMappings();
379
380 _glfwPlatformInitTimer();
381 _glfw.timer.offset = _glfwPlatformGetTimerValue();
382
383 _glfw.initialized = GLFW_TRUE;
384
385 glfwDefaultWindowHints();
386 return GLFW_TRUE;
387}

394 terminate();

453GLFWAPI int glfwGetError(const char** description)
454{
455 _GLFWerror* error;
456 int code = GLFW_NO_ERROR;
457
458 if (description)
459 *description = NULL;
460
461 if (_glfw.initialized)
462 error = _glfwPlatformGetTls(&_glfw.errorSlot);
463 else
464 error = &_glfwMainThreadError;
465
466 if (error)
467 {
468 code = error->code;
469 error->code = GLFW_NO_ERROR;
470 if (description && code)
471 *description = error->description;
472 }
473
474 return code;
475}
