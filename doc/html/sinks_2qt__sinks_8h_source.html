30class qt_sink : public base_sink<Mutex>
31{
32public:
33 qt_sink(QObject *qt_object, const std::string &meta_method)
34 {
35 qt_object_ = qt_object;
36 meta_method_ = meta_method;
37 }
38
39 ~qt_sink()
40 {
41 flush_();
42 }
43
44protected:
45 void sink_it_(const details::log_msg &msg) override
46 {
47 memory_buf_t formatted;
48 base_sink<Mutex>::formatter_->format(msg, formatted);
49 string_view_t str = string_view_t(formatted.data(), formatted.size());
50 QMetaObject::invokeMethod(qt_object_, meta_method_.c_str(), Qt::AutoConnection,
51 Q_ARG(QString, QString::fromUtf8(str.data(), static_cast<int>(str.size())).trimmed()));
52 }
53
54 void flush_() override {}
55
56private:
57 QObject *qt_object_ = nullptr;
58 std::string meta_method_;
59};

61// QT color sink to QTextEdit.
62// Color location is determined by the sink log pattern like in the rest of spdlog sinks.
63// Colors can be modified if needed using sink->set_color(level, qtTextCharFormat).
64// max_lines is the maximum number of lines that the sink will hold before removing the oldest lines.
65// Note: Only ascii (latin1) is supported by this sink.
66 template<typename Mutex>
67 class qt_color_sink : public base_sink<Mutex>
68 {
69 public:
70 qt_color_sink(QTextEdit *qt_text_edit, int max_lines)
71 : qt_text_edit_(qt_text_edit), max_lines_(max_lines)
72 {
73 if (!qt_text_edit_)
74 {
75 throw_spdlog_ex("qt_color_text_sink: text_edit is null");
76 }
77
78 default_color_ = qt_text_edit_->currentCharFormat();
79 // set colors
80 QTextCharFormat format;
81 // trace
82 format.setForeground(Qt::gray);
83 colors_.at(level::trace) = format;
84 // debug
85 format.setForeground(Qt::cyan);
86 colors_.at(level::debug) = format;
87 // info
88 format.setForeground(Qt::green);
89 colors_.at(level::info) = format;
90 // warn
91 format.setForeground(Qt::yellow);
92 colors_.at(level::warn) = format;
93 // err
94 format.setForeground(Qt::red);
95 colors_.at(level::err) = format;
96 // critical
97 format.setForeground(Qt::white);
98 format.setBackground(Qt::red);
99 colors_.at(level::critical) = format;
100 }
101
102 ~qt_color_sink()
103 {
104 flush_();
105 }
106
107 void set_default_color(QTextCharFormat format)
108 {
109 // std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
110 default_color_ = format;
111 }
112
113 void set_level_color(level::level_enum color_level, QTextCharFormat format)
114 {
115 // std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
116 colors_.at(static_cast<size_t>(color_level)) = format;
117 }
118
119 QTextCharFormat &get_level_color(level::level_enum color_level)
120 {
121 std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
122 return colors_.at(static_cast<size_t>(color_level));
123 }
124
125 QTextCharFormat &get_default_color()
126 {
127 std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
128 return default_color_;
129 }

131 protected:
132 struct invoke_params
133 {
134 invoke_params(int max_lines, QTextEdit *q_text_edit, QString payload, QTextCharFormat default_color,
135 QTextCharFormat level_color, int color_range_start, int color_range_end)
136 : max_lines(max_lines),
137 q_text_edit(q_text_edit),
138 payload(std::move(payload)),
139 default_color(default_color),
140 level_color(level_color),
141 color_range_start(color_range_start),
142 color_range_end(color_range_end)
143 {
144 }
145 int max_lines;
146 QTextEdit *q_text_edit;
147 QString payload;
148 QTextCharFormat default_color;
149 QTextCharFormat level_color;
150 int color_range_start;
151 int color_range_end;
152 };
153 void sink_it_(const details::log_msg &msg) override
154 {
155
156 using this_type = qt_color_sink<Mutex>;
157 memory_buf_t formatted;
158 base_sink<Mutex>::formatter_->format(msg, formatted);
159
160 const string_view_t str = string_view_t(formatted.data(), formatted.size());
161 // apply the color to the color range in the formatted message.
162 auto payload = QString::fromLatin1(str.data(), static_cast<int>(str.size()));
163
164 invoke_params params {
165 max_lines_, // max lines
166 qt_text_edit_, // text edit to append to
167 std::move(payload), // text to append
168 default_color_, // default color
169 colors_.at(msg.level), // color to apply
170 static_cast<int>(msg.color_range_start), // color range start
171 static_cast<int>(msg.color_range_end)}; // color range end
172
173 QMetaObject::invokeMethod(
174 qt_text_edit_,
175 [params]() {invoke_method_(params);},
176 Qt::AutoConnection);
177
178 }
179
180 void flush_() override {}
181
182 // Add colored text to the text edit widget. This method is invoked in the GUI thread.
183 // It is a static method to ensure that it is handled correctly even if the sink is destroyed prematurely
184 // before it is invoked.
185
186 static void invoke_method_(invoke_params params)
187 {
188 auto *document = params.q_text_edit->document();
189 QTextCursor cursor(document);
190
191 // remove first blocks if number of blocks exceeds max_lines
192 while(document->blockCount() > params.max_lines)
193 {
194 cursor.select(QTextCursor::BlockUnderCursor);
195 cursor.removeSelectedText();
196 cursor.deleteChar(); // delete the newline after the block
197 }
198
199 cursor.movePosition(QTextCursor::End);
200 cursor.setCharFormat(params.default_color);
201
202 // if color range not specified or not not valid, just append the text with default color
203 if(params.color_range_end <= params.color_range_start)
204 {
205 cursor.insertText(params.payload);
206 return;
207 }
208
209 // insert the text before the color range
210 cursor.insertText(params.payload.left(params.color_range_start));
211
212 // insert the colorized text
213 cursor.setCharFormat(params.level_color);
214 cursor.insertText(params.payload.mid(params.color_range_start, params.color_range_end - params.color_range_start));
215
216 // insert the text after the color range with default format
217 cursor.setCharFormat(params.default_color);
218 cursor.insertText(params.payload.mid(params.color_range_end));
219 }
220
221 QTextEdit *qt_text_edit_;
222 int max_lines_;
223 QTextCharFormat default_color_;
224 std::array<QTextCharFormat, level::n_levels> colors_;
225 };
