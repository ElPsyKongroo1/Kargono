45// An output iterator that truncates the output and counts the number of objects
46// written to it.
47template <typename OutputIt,
48 typename Enable = typename std::is_void<
49 typename std::iterator_traits<OutputIt>::value_type>::type>
50class truncating_iterator;
51
52template <typename OutputIt>
53class truncating_iterator<OutputIt, std::false_type>
54 : public truncating_iterator_base<OutputIt> {
55 mutable typename truncating_iterator_base<OutputIt>::value_type blackhole_;
56
57 public:
58 using value_type = typename truncating_iterator_base<OutputIt>::value_type;
59
60 truncating_iterator() = default;
61
62 truncating_iterator(OutputIt out, size_t limit)
63 : truncating_iterator_base<OutputIt>(out, limit) {}
64
65 truncating_iterator& operator++() {
66 if (this->count_++ < this->limit_) ++this->out_;
67 return *this;
68 }
69
70 truncating_iterator operator++(int) {
71 auto it = *this;
72 ++*this;
73 return it;
74 }
75
76 value_type& operator*() const {
77 return this->count_ < this->limit_ ? *this->out_ : blackhole_;
78 }
79};

231template <typename Char> struct runtime_named_field {
232 using char_type = Char;
233 basic_string_view<Char> name;

235 template <typename OutputIt, typename T>
236 constexpr static bool try_format_argument(
237 OutputIt& out,
238 // [[maybe_unused]] due to unused-but-set-parameter warning in GCC 7,8,9
239 [[maybe_unused]] basic_string_view<Char> arg_name, const T& arg) {
240 if constexpr (is_named_arg<typename std::remove_cv<T>::type>::value) {
241 if (arg_name == arg.name) {

253 FMT_THROW(format_error("argument with specified name is not found"));

313template <typename Args, size_t POS, int ID, typename T, typename S>
314constexpr auto parse_tail(T head, S format_str) {
315 if constexpr (POS !=
316 basic_string_view<typename S::char_type>(format_str).size()) {
317 constexpr auto tail = compile_format_string<Args, POS, ID>(format_str);
318 if constexpr (std::is_same<remove_cvref_t<decltype(tail)>,
319 unknown_format>())
320 return tail;
321 else
322 return make_concat(head, tail);
323 } else {
324 return head;
325 }

328template <typename T, typename Char> struct parse_specs_result {
329 formatter<T, Char> fmt;
330 size_t end;
331 int next_arg_id;
332};

334enum { manual_indexing_id = -1 };

336template <typename T, typename Char>
337constexpr parse_specs_result<T, Char> parse_specs(basic_string_view<Char> str,
338 size_t pos, int next_arg_id) {
339 str.remove_prefix(pos);
340 auto ctx =
341 compile_parse_context<Char>(str, max_value<int>(), nullptr, next_arg_id);
342 auto f = formatter<T, Char>();
343 auto end = f.parse(ctx);
344 return {f, pos + fmt::detail::to_unsigned(end - str.data()),
345 next_arg_id == 0 ? manual_indexing_id : ctx.next_arg_id()};
346}

386template <typename T, typename Args, size_t END_POS, int ARG_INDEX, int NEXT_ID,
387 typename S>
388constexpr auto parse_replacement_field_then_tail(S format_str) {
389 using char_type = typename S::char_type;
390 constexpr auto str = basic_string_view<char_type>(format_str);
391 constexpr char_type c = END_POS != str.size() ? str[END_POS] : char_type();
392 if constexpr (c == '}') {
393 return parse_tail<Args, END_POS + 1, NEXT_ID>(
394 field<char_type, typename field_type<T>::type, ARG_INDEX>(),
395 format_str);
396 } else if constexpr (c != ':') {
397 FMT_THROW(format_error("expected ':'"));
398 } else {
399 constexpr auto result = parse_specs<typename field_type<T>::type>(
400 str, END_POS + 1, NEXT_ID == manual_indexing_id ? 0 : NEXT_ID);
401 if constexpr (result.end >= str.size() || str[result.end] != '}') {
402 FMT_THROW(format_error("expected '}'"));
403 return 0;
404 } else {
405 return parse_tail<Args, result.end + 1, result.next_arg_id>(
406 spec_field<char_type, typename field_type<T>::type, ARG_INDEX>{
407 result.fmt},
408 format_str);
409 }
410 }
411}

413// Compiles a non-empty format string and returns the compiled representation
414// or unknown_format() on unrecognized input.
415template <typename Args, size_t POS, int ID, typename S>
416constexpr auto compile_format_string(S format_str) {
417 using char_type = typename S::char_type;
418 constexpr auto str = basic_string_view<char_type>(format_str);
419 if constexpr (str[POS] == '{') {
420 if constexpr (POS + 1 == str.size())
421 FMT_THROW(format_error("unmatched '{' in format string"));
422 if constexpr (str[POS + 1] == '{') {
423 return parse_tail<Args, POS + 2, ID>(make_text(str, POS, 1), format_str);
424 } else if constexpr (str[POS + 1] == '}' || str[POS + 1] == ':') {
425 static_assert(ID != manual_indexing_id,
426 "cannot switch from manual to automatic argument indexing");
427 constexpr auto next_id =
428 ID != manual_indexing_id ? ID + 1 : manual_indexing_id;
429 return parse_replacement_field_then_tail<get_type<ID, Args>, Args,
430 POS + 1, ID, next_id>(
431 format_str);
432 } else {
433 constexpr auto arg_id_result =
434 parse_arg_id<ID>(str.data() + POS + 1, str.data() + str.size());
435 constexpr auto arg_id_end_pos = arg_id_result.arg_id_end - str.data();
436 constexpr char_type c =
437 arg_id_end_pos != str.size() ? str[arg_id_end_pos] : char_type();
438 static_assert(c == '}' || c == ':', "missing '}' in format string");
439 if constexpr (arg_id_result.arg_id.kind == arg_id_kind::index) {
440 static_assert(
441 ID == manual_indexing_id || ID == 0,
442 "cannot switch from automatic to manual argument indexing");
443 constexpr auto arg_index = arg_id_result.arg_id.val.index;
444 return parse_replacement_field_then_tail<get_type<arg_index, Args>,
445 Args, arg_id_end_pos,
446 arg_index, manual_indexing_id>(
447 format_str);
448 } else if constexpr (arg_id_result.arg_id.kind == arg_id_kind::name) {
449 constexpr auto arg_index =
450 get_arg_index_by_name(arg_id_result.arg_id.val.name, Args{});
451 if constexpr (arg_index != invalid_arg_index) {
452 constexpr auto next_id =
453 ID != manual_indexing_id ? ID + 1 : manual_indexing_id;
454 return parse_replacement_field_then_tail<
455 decltype(get_type<arg_index, Args>::value), Args, arg_id_end_pos,
456 arg_index, next_id>(format_str);
457 } else {
458 if constexpr (c == '}') {
459 return parse_tail<Args, arg_id_end_pos + 1, ID>(
460 runtime_named_field<char_type>{arg_id_result.arg_id.val.name},
461 format_str);
462 } else if constexpr (c == ':') {
463 return unknown_format(); // no type info for specs parsing
464 }
465 }
466 }
467 }
468 } else if constexpr (str[POS] == '}') {
469 if constexpr (POS + 1 == str.size())
470 FMT_THROW(format_error("unmatched '}' in format string"));
471 return parse_tail<Args, POS + 2, ID>(make_text(str, POS, 1), format_str);
472 } else {
473 constexpr auto end = parse_text(str, POS + 1);
474 if constexpr (end - POS > 1) {
475 return parse_tail<Args, end, ID>(make_text(str, POS, end - POS),
476 format_str);
477 } else {
478 return parse_tail<Args, end, ID>(code_unit<char_type>{str[POS]},
479 format_str);
480 }
481 }
482}

484template <typename... Args, typename S,
485 FMT_ENABLE_IF(detail::is_compiled_string<S>::value)>
486constexpr auto compile(S format_str) {
487 constexpr auto str = basic_string_view<typename S::char_type>(format_str);
488 if constexpr (str.size() == 0) {
489 return detail::make_text(str, 0, 0);
490 } else {
491 constexpr auto result =
492 detail::compile_format_string<detail::type_list<Args...>, 0, 0>(
493 format_str);
494 return result;
495 }

521template <typename S, typename... Args,
522 FMT_ENABLE_IF(detail::is_compiled_string<S>::value)>
523FMT_INLINE std::basic_string<typename S::char_type> format(const S&,
524 Args&&... args) {
525 if constexpr (std::is_same<typename S::char_type, char>::value) {
526 constexpr auto str = basic_string_view<typename S::char_type>(S());
527 if constexpr (str.size() == 2 && str[0] == '{' && str[1] == '}') {
528 const auto& first = detail::first(args...);
529 if constexpr (detail::is_named_arg<
530 remove_cvref_t<decltype(first)>>::value) {
531 return fmt::to_string(first.value);
532 } else {
533 return fmt::to_string(first);
534 }
535 }
536 }
537 constexpr auto compiled = detail::compile<Args...>(S());
538 if constexpr (std::is_same<remove_cvref_t<decltype(compiled)>,
539 detail::unknown_format>()) {
540 return fmt::format(
541 static_cast<basic_string_view<typename S::char_type>>(S()),
542 std::forward<Args>(args)...);
543 } else {
544 return fmt::format(compiled, std::forward<Args>(args)...);
545 }
