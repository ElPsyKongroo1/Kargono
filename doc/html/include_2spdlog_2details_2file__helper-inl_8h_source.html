32SPDLOG_INLINE void file_helper::open(const filename_t &fname, bool truncate)
33{
34 close();
35 filename_ = fname;
36
37 auto *mode = SPDLOG_FILENAME_T("ab");
38 auto *trunc_mode = SPDLOG_FILENAME_T("wb");
39
40 if (event_handlers_.before_open)
41 {
42 event_handlers_.before_open(filename_);
43 }
44 for (int tries = 0; tries < open_tries_; ++tries)
45 {
46 // create containing folder if not exists already.
47 os::create_dir(os::dir_name(fname));
48 if (truncate)
49 {
50 // Truncate by opening-and-closing a tmp file in "wb" mode, always
51 // opening the actual log-we-write-to in "ab" mode, since that
52 // interacts more politely with eternal processes that might
53 // rotate/truncate the file underneath us.
54 std::FILE *tmp;
55 if (os::fopen_s(&tmp, fname, trunc_mode))
56 {
57 continue;
58 }
59 std::fclose(tmp);
60 }
61 if (!os::fopen_s(&fd_, fname, mode))
62 {
63 if (event_handlers_.after_open)
64 {
65 event_handlers_.after_open(filename_, fd_);
66 }
67 return;
68 }
69
70 details::os::sleep_for_millis(open_interval_);
71 }
72
73 throw_spdlog_ex("Failed opening file " + os::filename_to_str(filename_) + " for writing", errno);
74}

144//
145// return file path and its extension:
146//
147// "mylog.txt" => ("mylog", ".txt")
148// "mylog" => ("mylog", "")
149// "mylog." => ("mylog.", "")
150// "/dir1/dir2/mylog.txt" => ("/dir1/dir2/mylog", ".txt")
151//
152// the starting dot in filenames is ignored (hidden files):
153//
154// ".mylog" => (".mylog". "")
155// "my_folder/.mylog" => ("my_folder/.mylog", "")
156// "my_folder/.mylog.txt" => ("my_folder/.mylog", ".txt")
157SPDLOG_INLINE std::tuple<filename_t, filename_t> file_helper::split_by_extension(const filename_t &fname)
158{
159 auto ext_index = fname.rfind('.');
160
161 // no valid extension found - return whole path and empty string as
162 // extension
163 if (ext_index == filename_t::npos || ext_index == 0 || ext_index == fname.size() - 1)
164 {
165 return std::make_tuple(fname, filename_t());
166 }
167
168 // treat cases like "/etc/rc.d/somelogfile or "/abc/.hiddenfile"
169 auto folder_index = fname.find_last_of(details::os::folder_seps_filename);
170 if (folder_index != filename_t::npos && folder_index >= ext_index - 1)
171 {
172 return std::make_tuple(fname, filename_t());
173 }
174
175 // finally - return a valid base and extension tuple
176 return std::make_tuple(fname.substr(0, ext_index), fname.substr(ext_index));
177}
