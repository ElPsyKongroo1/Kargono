65static _GLFWwindow* findWindowFromDecorationSurface(struct wl_surface* surface,
66 int* which)
67{
68 int focus;
69 _GLFWwindow* window = _glfw.windowListHead;
70 if (!which)
71 which = &focus;
72 while (window)
73 {
74 if (surface == window->wl.decorations.top.surface)
75 {
76 *which = topDecoration;
77 break;
78 }
79 if (surface == window->wl.decorations.left.surface)
80 {
81 *which = leftDecoration;
82 break;
83 }
84 if (surface == window->wl.decorations.right.surface)
85 {
86 *which = rightDecoration;
87 break;
88 }
89 if (surface == window->wl.decorations.bottom.surface)
90 {
91 *which = bottomDecoration;
92 break;
93 }
94 window = window->next;
95 }
96 return window;
97}

148static void setCursor(_GLFWwindow* window, const char* name)
149{
150 struct wl_buffer* buffer;
151 struct wl_cursor* cursor;
152 struct wl_cursor_image* image;
153 struct wl_surface* surface = _glfw.wl.cursorSurface;
154 struct wl_cursor_theme* theme = _glfw.wl.cursorTheme;
155 int scale = 1;
156
157 if (window->wl.scale > 1 && _glfw.wl.cursorThemeHiDPI)
158 {
159 // We only support up to scale=2 for now, since libwayland-cursor
160 // requires us to load a different theme for each size.
161 scale = 2;
162 theme = _glfw.wl.cursorThemeHiDPI;
163 }
164
165 cursor = wl_cursor_theme_get_cursor(theme, name);
166 if (!cursor)
167 {
168 _glfwInputError(GLFW_PLATFORM_ERROR,
169 "Wayland: Standard cursor not found");
170 return;
171 }
172 // TODO: handle animated cursors too.
173 image = cursor->images[0];
174
175 if (!image)
176 return;
177
178 buffer = wl_cursor_image_get_buffer(image);
179 if (!buffer)
180 return;
181 wl_pointer_set_cursor(_glfw.wl.pointer, _glfw.wl.pointerEnterSerial,
182 surface,
183 image->hotspot_x / scale,
184 image->hotspot_y / scale);
185 wl_surface_set_buffer_scale(surface, scale);
186 wl_surface_attach(surface, buffer, 0, 0);
187 wl_surface_damage(surface, 0, 0,
188 image->width, image->height);
189 wl_surface_commit(surface);
190 _glfw.wl.cursorPreviousName = name;
191}

193static void pointerHandleMotion(void* data,
194 struct wl_pointer* pointer,
195 uint32_t time,
196 wl_fixed_t sx,
197 wl_fixed_t sy)
198{
199 _GLFWwindow* window = _glfw.wl.pointerFocus;
200 const char* cursorName = NULL;
201 double x, y;
202
203 if (!window)
204 return;
205
206 if (window->cursorMode == GLFW_CURSOR_DISABLED)
207 return;
208 x = wl_fixed_to_double(sx);
209 y = wl_fixed_to_double(sy);
210 window->wl.cursorPosX = x;
211 window->wl.cursorPosY = y;
212
213 switch (window->wl.decorations.focus)
214 {
215 case mainWindow:
216 _glfwInputCursorPos(window, x, y);
217 _glfw.wl.cursorPreviousName = NULL;
218 return;
219 case topDecoration:
220 if (y < _GLFW_DECORATION_WIDTH)
221 cursorName = "n-resize";
222 else
223 cursorName = "left_ptr";
224 break;
225 case leftDecoration:
226 if (y < _GLFW_DECORATION_WIDTH)
227 cursorName = "nw-resize";
228 else
229 cursorName = "w-resize";
230 break;
231 case rightDecoration:
232 if (y < _GLFW_DECORATION_WIDTH)
233 cursorName = "ne-resize";
234 else
235 cursorName = "e-resize";
236 break;
237 case bottomDecoration:
238 if (x < _GLFW_DECORATION_WIDTH)
239 cursorName = "sw-resize";
240 else if (x > window->wl.width + _GLFW_DECORATION_WIDTH)
241 cursorName = "se-resize";
242 else
243 cursorName = "s-resize";
244 break;
245 default:
246 assert(0);
247 }
248 if (_glfw.wl.cursorPreviousName != cursorName)
249 setCursor(window, cursorName);
250}

252static void pointerHandleButton(void* data,
253 struct wl_pointer* pointer,
254 uint32_t serial,
255 uint32_t time,
256 uint32_t button,
257 uint32_t state)
258{
259 _GLFWwindow* window = _glfw.wl.pointerFocus;
260 int glfwButton;
261 uint32_t edges = XDG_TOPLEVEL_RESIZE_EDGE_NONE;
262
263 if (!window)
264 return;
265 if (button == BTN_LEFT)
266 {
267 switch (window->wl.decorations.focus)
268 {
269 case mainWindow:
270 break;
271 case topDecoration:
272 if (window->wl.cursorPosY < _GLFW_DECORATION_WIDTH)
273 edges = XDG_TOPLEVEL_RESIZE_EDGE_TOP;
274 else
275 xdg_toplevel_move(window->wl.xdg.toplevel, _glfw.wl.seat, serial);
276 break;
277 case leftDecoration:
278 if (window->wl.cursorPosY < _GLFW_DECORATION_WIDTH)
279 edges = XDG_TOPLEVEL_RESIZE_EDGE_TOP_LEFT;
280 else
281 edges = XDG_TOPLEVEL_RESIZE_EDGE_LEFT;
282 break;
283 case rightDecoration:
284 if (window->wl.cursorPosY < _GLFW_DECORATION_WIDTH)
285 edges = XDG_TOPLEVEL_RESIZE_EDGE_TOP_RIGHT;
286 else
287 edges = XDG_TOPLEVEL_RESIZE_EDGE_RIGHT;
288 break;
289 case bottomDecoration:
290 if (window->wl.cursorPosX < _GLFW_DECORATION_WIDTH)
291 edges = XDG_TOPLEVEL_RESIZE_EDGE_BOTTOM_LEFT;
292 else if (window->wl.cursorPosX > window->wl.width + _GLFW_DECORATION_WIDTH)
293 edges = XDG_TOPLEVEL_RESIZE_EDGE_BOTTOM_RIGHT;
294 else
295 edges = XDG_TOPLEVEL_RESIZE_EDGE_BOTTOM;
296 break;
297 default:
298 assert(0);
299 }
300 if (edges != XDG_TOPLEVEL_RESIZE_EDGE_NONE)
301 {
302 xdg_toplevel_resize(window->wl.xdg.toplevel, _glfw.wl.seat,
303 serial, edges);
304 return;
305 }
306 }
307 else if (button == BTN_RIGHT)
308 {
309 if (window->wl.decorations.focus != mainWindow && window->wl.xdg.toplevel)
310 {
311 xdg_toplevel_show_window_menu(window->wl.xdg.toplevel,
312 _glfw.wl.seat, serial,
313 window->wl.cursorPosX,
314 window->wl.cursorPosY);
315 return;
316 }
317 }
318
319 // Donâ€™t pass the button to the user if it was related to a decoration.
320 if (window->wl.decorations.focus != mainWindow)
321 return;
322
323 _glfw.wl.serial = serial;
324
325 /* Makes left, right and middle 0, 1 and 2. Overall order follows evdev
326 * codes. */
327 glfwButton = button - BTN_LEFT;
328
329 _glfwInputMouseClick(window,
330 glfwButton,
331 state == WL_POINTER_BUTTON_STATE_PRESSED
332 ? GLFW_PRESS
333 : GLFW_RELEASE,
334 _glfw.wl.xkb.modifiers);
335}

337static void pointerHandleAxis(void* data,
338 struct wl_pointer* pointer,
339 uint32_t time,
340 uint32_t axis,
341 wl_fixed_t value)
342{
343 _GLFWwindow* window = _glfw.wl.pointerFocus;
344 double x = 0.0, y = 0.0;
345 // Wayland scroll events are in pointer motion coordinate space (think two
346 // finger scroll). The factor 10 is commonly used to convert to "scroll
347 // step means 1.0.
348 const double scrollFactor = 1.0 / 10.0;
349
350 if (!window)
351 return;
352
353 assert(axis == WL_POINTER_AXIS_HORIZONTAL_SCROLL ||
354 axis == WL_POINTER_AXIS_VERTICAL_SCROLL);
355
356 if (axis == WL_POINTER_AXIS_HORIZONTAL_SCROLL)
357 x = -wl_fixed_to_double(value) * scrollFactor;
358 else if (axis == WL_POINTER_AXIS_VERTICAL_SCROLL)
359 y = -wl_fixed_to_double(value) * scrollFactor;
360
361 _glfwInputScroll(window, x, y);
362}

372static void keyboardHandleKeymap(void* data,
373 struct wl_keyboard* keyboard,
374 uint32_t format,
375 int fd,
376 uint32_t size)
377{
378 struct xkb_keymap* keymap;
379 struct xkb_state* state;
380 struct xkb_compose_table* composeTable;
381 struct xkb_compose_state* composeState;
382
383 char* mapStr;
384 const char* locale;
385
386 if (format != WL_KEYBOARD_KEYMAP_FORMAT_XKB_V1)
387 {
388 close(fd);
389 return;
390 }
391
392 mapStr = mmap(NULL, size, PROT_READ, MAP_SHARED, fd, 0);
393 if (mapStr == MAP_FAILED) {
394 close(fd);
395 return;
396 }
397
398 keymap = xkb_keymap_new_from_string(_glfw.wl.xkb.context,
399 mapStr,
400 XKB_KEYMAP_FORMAT_TEXT_V1,
401 0);
402 munmap(mapStr, size);
403 close(fd);
404
405 if (!keymap)
406 {
407 _glfwInputError(GLFW_PLATFORM_ERROR,
408 "Wayland: Failed to compile keymap");
409 return;
410 }
411
412 state = xkb_state_new(keymap);
413 if (!state)
414 {
415 _glfwInputError(GLFW_PLATFORM_ERROR,
416 "Wayland: Failed to create XKB state");
417 xkb_keymap_unref(keymap);
418 return;
419 }
420
421 // Look up the preferred locale, falling back to "C" as default.
422 locale = getenv("LC_ALL");
423 if (!locale)
424 locale = getenv("LC_CTYPE");
425 if (!locale)
426 locale = getenv("LANG");
427 if (!locale)
428 locale = "C";
429
430 composeTable =
431 xkb_compose_table_new_from_locale(_glfw.wl.xkb.context, locale,
432 XKB_COMPOSE_COMPILE_NO_FLAGS);
433 if (composeTable)
434 {
435 composeState =
436 xkb_compose_state_new(composeTable, XKB_COMPOSE_STATE_NO_FLAGS);
437 xkb_compose_table_unref(composeTable);
438 if (composeState)
439 _glfw.wl.xkb.composeState = composeState;
440 else
441 _glfwInputError(GLFW_PLATFORM_ERROR,
442 "Wayland: Failed to create XKB compose state");
443 }
444 else
445 {
446 _glfwInputError(GLFW_PLATFORM_ERROR,
447 "Wayland: Failed to create XKB compose table");
448 }
449
450 xkb_keymap_unref(_glfw.wl.xkb.keymap);
451 xkb_state_unref(_glfw.wl.xkb.state);
452 _glfw.wl.xkb.keymap = keymap;
453 _glfw.wl.xkb.state = state;
454
455 _glfw.wl.xkb.controlMask =
456 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Control");
457 _glfw.wl.xkb.altMask =
458 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Mod1");
459 _glfw.wl.xkb.shiftMask =
460 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Shift");
461 _glfw.wl.xkb.superMask =
462 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Mod4");
463 _glfw.wl.xkb.capsLockMask =
464 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Lock");
465 _glfw.wl.xkb.numLockMask =
466 1 << xkb_keymap_mod_get_index(_glfw.wl.xkb.keymap, "Mod2");
467}

518static xkb_keysym_t composeSymbol(xkb_keysym_t sym)
519{
520 if (sym == XKB_KEY_NoSymbol || !_glfw.wl.xkb.composeState)
521 return sym;
522 if (xkb_compose_state_feed(_glfw.wl.xkb.composeState, sym)
523 != XKB_COMPOSE_FEED_ACCEPTED)
524 return sym;
525 switch (xkb_compose_state_get_status(_glfw.wl.xkb.composeState))
526 {
527 case XKB_COMPOSE_COMPOSED:
528 return xkb_compose_state_get_one_sym(_glfw.wl.xkb.composeState);
529 case XKB_COMPOSE_COMPOSING:
530 case XKB_COMPOSE_CANCELLED:
531 return XKB_KEY_NoSymbol;
532 case XKB_COMPOSE_NOTHING:
533 default:
534 return sym;
535 }
536}

538GLFWbool _glfwInputTextWayland(_GLFWwindow* window, uint32_t scancode)
539{
540 const xkb_keysym_t* keysyms;
541 const xkb_keycode_t keycode = scancode + 8;
542
543 if (xkb_state_key_get_syms(_glfw.wl.xkb.state, keycode, &keysyms) == 1)
544 {
545 const xkb_keysym_t keysym = composeSymbol(keysyms[0]);
546 const uint32_t codepoint = _glfwKeySym2Unicode(keysym);
547 if (codepoint != GLFW_INVALID_CODEPOINT)
548 {
549 const int mods = _glfw.wl.xkb.modifiers;
550 const int plain = !(mods & (GLFW_MOD_CONTROL | GLFW_MOD_ALT));
551 _glfwInputChar(window, codepoint, mods, plain);
552 }
553 }
554
555 return xkb_keymap_key_repeats(_glfw.wl.xkb.keymap, keycode);
556}

558static void keyboardHandleKey(void* data,
559 struct wl_keyboard* keyboard,
560 uint32_t serial,
561 uint32_t time,
562 uint32_t scancode,
563 uint32_t state)
564{
565 _GLFWwindow* window = _glfw.wl.keyboardFocus;
566 if (!window)
567 return;
568
569 const int key = translateKey(scancode);
570 const int action =
571 state == WL_KEYBOARD_KEY_STATE_PRESSED ? GLFW_PRESS : GLFW_RELEASE;
572
573 _glfw.wl.serial = serial;
574 _glfwInputKey(window, key, scancode, action, _glfw.wl.xkb.modifiers);
575
576 struct itimerspec timer = {};
577
578 if (action == GLFW_PRESS)
579 {
580 const GLFWbool shouldRepeat = _glfwInputTextWayland(window, scancode);
581
582 if (shouldRepeat && _glfw.wl.keyboardRepeatRate > 0)
583 {
584 _glfw.wl.keyboardLastKey = key;
585 _glfw.wl.keyboardLastScancode = scancode;
586 if (_glfw.wl.keyboardRepeatRate > 1)
587 timer.it_interval.tv_nsec = 1000000000 / _glfw.wl.keyboardRepeatRate;
588 else
589 timer.it_interval.tv_sec = 1;
590
591 timer.it_value.tv_sec = _glfw.wl.keyboardRepeatDelay / 1000;
592 timer.it_value.tv_nsec = (_glfw.wl.keyboardRepeatDelay % 1000) * 1000000;
593 }
594 }
595
596 timerfd_settime(_glfw.wl.timerfd, 0, &timer, NULL);
597}

599static void keyboardHandleModifiers(void* data,
600 struct wl_keyboard* keyboard,
601 uint32_t serial,
602 uint32_t modsDepressed,
603 uint32_t modsLatched,
604 uint32_t modsLocked,
605 uint32_t group)
606{
607 _glfw.wl.serial = serial;
608
609 if (!_glfw.wl.xkb.keymap)
610 return;
611
612 xkb_state_update_mask(_glfw.wl.xkb.state,
613 modsDepressed,
614 modsLatched,
615 modsLocked,
616 0,
617 0,
618 group);
619
620 const xkb_mod_mask_t mask =
621 xkb_state_serialize_mods(_glfw.wl.xkb.state,
622 XKB_STATE_MODS_DEPRESSED |
623 XKB_STATE_LAYOUT_DEPRESSED |
624 XKB_STATE_MODS_LATCHED |
625 XKB_STATE_LAYOUT_LATCHED);
626
627 unsigned int mods = 0;
628
629 if (mask & _glfw.wl.xkb.controlMask)
630 mods |= GLFW_MOD_CONTROL;
631 if (mask & _glfw.wl.xkb.altMask)
632 mods |= GLFW_MOD_ALT;
633 if (mask & _glfw.wl.xkb.shiftMask)
634 mods |= GLFW_MOD_SHIFT;
635 if (mask & _glfw.wl.xkb.superMask)
636 mods |= GLFW_MOD_SUPER;
637 if (mask & _glfw.wl.xkb.capsLockMask)
638 mods |= GLFW_MOD_CAPS_LOCK;
639 if (mask & _glfw.wl.xkb.numLockMask)
640 mods |= GLFW_MOD_NUM_LOCK;
641
642 _glfw.wl.xkb.modifiers = mods;
643}
