34int main(int, char *[])
35{
36 // Log levels can be loaded from argv/env using "SPDLOG_LEVEL"
37 load_levels_example();
38
39 spdlog::info("Welcome to spdlog version {}.{}.{} !", SPDLOG_VER_MAJOR, SPDLOG_VER_MINOR, SPDLOG_VER_PATCH);
40
41 spdlog::warn("Easy padding in numbers like {:08d}", 12);
42 spdlog::critical("Support for int: {0:d}; hex: {0:x}; oct: {0:o}; bin: {0:b}", 42);
43 spdlog::info("Support for floats {:03.2f}", 1.23456);
44 spdlog::info("Positional args are {1} {0}..", "too", "supported");
45 spdlog::info("{:>8} aligned, {:<8} aligned", "right", "left");
46
47 // Runtime log levels
48 spdlog::set_level(spdlog::level::info); // Set global log level to info
49 spdlog::debug("This message should not be displayed!");
50 spdlog::set_level(spdlog::level::trace); // Set specific logger's log level
51 spdlog::debug("This message should be displayed..");
52
53 // Customize msg format for all loggers
54 spdlog::set_pattern("[%H:%M:%S %z] [%^%L%$] [thread %t] %v");
55 spdlog::info("This an info message with custom format");
56 spdlog::set_pattern("%+"); // back to default format
57 spdlog::set_level(spdlog::level::info);
58
59 // Backtrace support
60 // Loggers can store in a ring buffer all messages (including debug/trace) for later inspection.
61 // When needed, call dump_backtrace() to see what happened:
62 spdlog::enable_backtrace(10); // create ring buffer with capacity of 10 messages
63 for (int i = 0; i < 100; i++)
64 {
65 spdlog::debug("Backtrace message {}", i); // not logged..
66 }
67 // e.g. if some error happened:
68 spdlog::dump_backtrace(); // log them now!
69

74 rotating_example();

90 // Flush all *registered* loggers using a worker thread every 3 seconds.
91 // note: registered loggers *must* be thread safe for this to work correctly!
92 spdlog::flush_every(std::chrono::seconds(3));
93
94 // Apply some function on all registered loggers
95 spdlog::apply_all([&](std::shared_ptr<spdlog::logger> l) { l->info("End of example."); });
96
97 // Release all spdlog resources, and drop all loggers in the registry.
98 // This is optional (only mandatory if using windows + async log).
99 spdlog::shutdown();
100 }

106 return 1;
107 }

177// Log binary data as hex.
178// Many types of std::container<char> types can be used.
179// Iterator ranges are supported too.
180// Format flags:
181// {:X} - print in uppercase.
182// {:s} - don't separate each byte with space.
183// {:p} - don't print the position on each line start.
184// {:n} - don't split the output to lines.
185
186#include "spdlog/fmt/bin_to_hex.h"
187void binary_example()
188{
189 std::vector<char> buf(80);
190 for (int i = 0; i < 80; i++)
191 {
192 buf.push_back(static_cast<char>(i & 0xff));
193 }
194 spdlog::info("Binary example: {}", spdlog::to_hex(buf));
195 spdlog::info("Another binary example:{:n}", spdlog::to_hex(std::begin(buf), std::begin(buf) + 10));
196 // more examples:
197 // logger->info("uppercase: {:X}", spdlog::to_hex(buf));
198 // logger->info("uppercase, no delimiters: {:Xs}", spdlog::to_hex(buf));
199 // logger->info("uppercase, no delimiters, no position info: {:Xsp}", spdlog::to_hex(buf));
200 // logger->info("hexdump style: {:a}", spdlog::to_hex(buf));
201 // logger->info("hexdump style, 20 chars per line {:a}", spdlog::to_hex(buf, 20));
202}
