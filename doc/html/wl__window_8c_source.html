52static int createTmpfileCloexec(char* tmpname)
53{
54 int fd;
55
56 fd = mkostemp(tmpname, O_CLOEXEC);
57 if (fd >= 0)
58 unlink(tmpname);
59
60 return fd;
61}

63/*
64 * Create a new, unique, anonymous file of the given size, and
65 * return the file descriptor for it. The file descriptor is set
66 * CLOEXEC. The file is immediately suitable for mmap()'ing
67 * the given size at offset zero.
68 *
69 * The file should not have a permanent backing store like a disk,
70 * but may have if XDG_RUNTIME_DIR is not properly implemented in OS.
71 *
72 * The file name is deleted from the file system.
73 *
74 * The file is suitable for buffer sharing between processes by
75 * transmitting the file descriptor over Unix sockets using the
76 * SCM_RIGHTS methods.
77 *
78 * posix_fallocate() is used to guarantee that disk space is available
79 * for the file at the given size. If disk space is insufficient, errno
80 * is set to ENOSPC. If posix_fallocate() is not supported, program may
81 * receive SIGBUS on accessing mmap()'ed file contents instead.
82 */
83static int createAnonymousFile(off_t size)
84{
85 static const char template[] = "/glfw-shared-XXXXXX";
86 const char* path;
87 char* name;
88 int fd;
89 int ret;
90
91#ifdef HAVE_MEMFD_CREATE
92 fd = memfd_create("glfw-shared", MFD_CLOEXEC | MFD_ALLOW_SEALING);
93 if (fd >= 0)
94 {
95 // We can add this seal before calling posix_fallocate(), as the file
96 // is currently zero-sized anyway.
97 //
98 // There is also no need to check for the return value, we couldnâ€™t do
99 // anything with it anyway.
100 fcntl(fd, F_ADD_SEALS, F_SEAL_SHRINK | F_SEAL_SEAL);
101 }
102 else
103#elif defined(SHM_ANON)
104 fd = shm_open(SHM_ANON, O_RDWR | O_CLOEXEC, 0600);
105 if (fd < 0)
106#endif
107 {
108 path = getenv("XDG_RUNTIME_DIR");
109 if (!path)
110 {
111 errno = ENOENT;
112 return -1;
113 }
114
115 name = _glfw_calloc(strlen(path) + sizeof(template), 1);
116 strcpy(name, path);
117 strcat(name, template);
118
119 fd = createTmpfileCloexec(name);
120 _glfw_free(name);
121 if (fd < 0)
122 return -1;
123 }
124
125#if defined(SHM_ANON)
126 // posix_fallocate does not work on SHM descriptors
127 ret = ftruncate(fd, size);
128#else
129 ret = posix_fallocate(fd, 0, size);
130#endif
131 if (ret != 0)
132 {
133 close(fd);
134 errno = ret;
135 return -1;
136 }
137 return fd;
138}

140static struct wl_buffer* createShmBuffer(const GLFWimage* image)
141{
142 struct wl_shm_pool* pool;
143 struct wl_buffer* buffer;
144 int stride = image->width * 4;
145 int length = image->width * image->height * 4;
146 void* data;
147
148 const int fd = createAnonymousFile(length);
149 if (fd < 0)
150 {
151 _glfwInputError(GLFW_PLATFORM_ERROR,
152 "Wayland: Creating a buffer file for %d B failed: %s",
153 length, strerror(errno));
154 return NULL;
155 }
156
157 data = mmap(NULL, length, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
158 if (data == MAP_FAILED)
159 {
160 _glfwInputError(GLFW_PLATFORM_ERROR,
161 "Wayland: mmap failed: %s", strerror(errno));
162 close(fd);
163 return NULL;
164 }
165
166 pool = wl_shm_create_pool(_glfw.wl.shm, fd, length);
167
168 close(fd);
169 unsigned char* source = (unsigned char*) image->pixels;
170 unsigned char* target = data;
171 for (int i = 0; i < image->width * image->height; i++, source += 4)
172 {
173 unsigned int alpha = source[3];
174
175 *target++ = (unsigned char) ((source[2] * alpha) / 255);
176 *target++ = (unsigned char) ((source[1] * alpha) / 255);
177 *target++ = (unsigned char) ((source[0] * alpha) / 255);
178 *target++ = (unsigned char) alpha;
179 }
180
181 buffer =
182 wl_shm_pool_create_buffer(pool, 0,
183 image->width,
184 image->height,
185 stride, WL_SHM_FORMAT_ARGB8888);
186 munmap(data, length);
187 wl_shm_pool_destroy(pool);
188
189 return buffer;
190}

222static void createDecorations(_GLFWwindow* window)
223{
224 unsigned char data[] = { 224, 224, 224, 255 };
225 const GLFWimage image = { 1, 1, data };
226 GLFWbool opaque = (data[3] == 255);
227

229 return;

232 window->wl.decorations.buffer = createShmBuffer(&image);
233 if (!window->wl.decorations.buffer)

236 createDecoration(&window->wl.decorations.top, window->wl.surface,
237 window->wl.decorations.buffer, opaque,

239 window->wl.width, _GLFW_DECORATION_TOP);
240 createDecoration(&window->wl.decorations.left, window->wl.surface,
241 window->wl.decorations.buffer, opaque,
242 -_GLFW_DECORATION_WIDTH, -_GLFW_DECORATION_TOP,
243 _GLFW_DECORATION_WIDTH, window->wl.height + _GLFW_DECORATION_TOP);

245 window->wl.decorations.buffer, opaque,
246 window->wl.width, -_GLFW_DECORATION_TOP,
247 _GLFW_DECORATION_WIDTH, window->wl.height + _GLFW_DECORATION_TOP);

249 window->wl.decorations.buffer, opaque,
250 -_GLFW_DECORATION_WIDTH, window->wl.height,
251 window->wl.width + _GLFW_DECORATION_HORIZONTAL, _GLFW_DECORATION_WIDTH);

307static void resizeWindow(_GLFWwindow* window)
308{
309 int scale = window->wl.scale;
310 int scaledWidth = window->wl.width * scale;
311 int scaledHeight = window->wl.height * scale;
312 wl_egl_window_resize(window->wl.native, scaledWidth, scaledHeight, 0, 0);
313 if (!window->wl.transparent)
314 setOpaqueRegion(window);
315 _glfwInputFramebufferSize(window, scaledWidth, scaledHeight);
316 _glfwInputWindowContentScale(window, scale, scale);
317
318 if (!window->wl.decorations.top.surface)
319 return;
320
321 // Top decoration.
322 wp_viewport_set_destination(window->wl.decorations.top.viewport,
323 window->wl.width, _GLFW_DECORATION_TOP);
324 wl_surface_commit(window->wl.decorations.top.surface);
325
326 // Left decoration.
327 wp_viewport_set_destination(window->wl.decorations.left.viewport,
328 _GLFW_DECORATION_WIDTH, window->wl.height + _GLFW_DECORATION_TOP);
329 wl_surface_commit(window->wl.decorations.left.surface);
330
331 // Right decoration.
332 wl_subsurface_set_position(window->wl.decorations.right.subsurface,
333 window->wl.width, -_GLFW_DECORATION_TOP);
334 wp_viewport_set_destination(window->wl.decorations.right.viewport,
335 _GLFW_DECORATION_WIDTH, window->wl.height + _GLFW_DECORATION_TOP);
336 wl_surface_commit(window->wl.decorations.right.surface);
337
338 // Bottom decoration.
339 wl_subsurface_set_position(window->wl.decorations.bottom.subsurface,

342 window->wl.width + _GLFW_DECORATION_HORIZONTAL, _GLFW_DECORATION_WIDTH);
343 wl_surface_commit(window->wl.decorations.bottom.surface);
344}

346static void checkScaleChange(_GLFWwindow* window)
347{
348 int scale = 1;
349 int monitorScale;
350
351 // Check if we will be able to set the buffer scale or not.
352 if (_glfw.wl.compositorVersion < 3)
353 return;
354
355 // Get the scale factor from the highest scale monitor.
356 for (int i = 0; i < window->wl.monitorsCount; ++i)
357 {
358 monitorScale = window->wl.monitors[i]->wl.scale;
359 if (scale < monitorScale)
360 scale = monitorScale;
361 }
362
363 // Only change the framebuffer size if the scale changed.
364 if (scale != window->wl.scale)
365 {
366 window->wl.scale = scale;
367 wl_surface_set_buffer_scale(window->wl.surface, scale);
368 resizeWindow(window);
369 }
370}

372static void surfaceHandleEnter(void *data,
373 struct wl_surface *surface,
374 struct wl_output *output)
375{
376 _GLFWwindow* window = data;
377 _GLFWmonitor* monitor = wl_output_get_user_data(output);
378
379 if (window->wl.monitorsCount + 1 > window->wl.monitorsSize)
380 {
381 ++window->wl.monitorsSize;
382 window->wl.monitors =
383 _glfw_realloc(window->wl.monitors,
384 window->wl.monitorsSize * sizeof(_GLFWmonitor*));
385 }
386
387 window->wl.monitors[window->wl.monitorsCount++] = monitor;
388
389 checkScaleChange(window);
390}

392static void surfaceHandleLeave(void *data,
393 struct wl_surface *surface,
394 struct wl_output *output)
395{
396 _GLFWwindow* window = data;
397 _GLFWmonitor* monitor = wl_output_get_user_data(output);
398 GLFWbool found = GLFW_FALSE;
399
400 for (int i = 0; i < window->wl.monitorsCount - 1; ++i)
401 {
402 if (monitor == window->wl.monitors[i])
403 found = GLFW_TRUE;
404 if (found)
405 window->wl.monitors[i] = window->wl.monitors[i + 1];
406 }
407 window->wl.monitors[--window->wl.monitorsCount] = NULL;
408
409 checkScaleChange(window);
410}

449static void xdgToplevelHandleConfigure(void* data,
450 struct xdg_toplevel* toplevel,
451 int32_t width,
452 int32_t height,
453 struct wl_array* states)
454{
455 _GLFWwindow* window = data;
456 float aspectRatio;
457 float targetRatio;
458 uint32_t* state;
459 GLFWbool maximized = GLFW_FALSE;
460 GLFWbool fullscreen = GLFW_FALSE;
461 GLFWbool activated = GLFW_FALSE;
462
463 wl_array_for_each(state, states)
464 {
465 switch (*state)
466 {
467 case XDG_TOPLEVEL_STATE_MAXIMIZED:
468 maximized = GLFW_TRUE;
469 break;
470 case XDG_TOPLEVEL_STATE_FULLSCREEN:
471 fullscreen = GLFW_TRUE;
472 break;
473 case XDG_TOPLEVEL_STATE_RESIZING:
474 break;
475 case XDG_TOPLEVEL_STATE_ACTIVATED:
476 activated = GLFW_TRUE;
477 break;
478 }
479 }
480
481 if (width != 0 && height != 0)
482 {
483 if (!maximized && !fullscreen)
484 {
485 if (window->numer != GLFW_DONT_CARE && window->denom != GLFW_DONT_CARE)
486 {
487 aspectRatio = (float)width / (float)height;
488 targetRatio = (float)window->numer / (float)window->denom;
489 if (aspectRatio < targetRatio)
490 height = width / targetRatio;
491 else if (aspectRatio > targetRatio)
492 width = height * targetRatio;
493 }
494 }
495
496 _glfwInputWindowSize(window, width, height);
497 _glfwSetWindowSizeWayland(window, width, height);
498 _glfwInputWindowDamage(window);
499 }
500
501 if (window->wl.wasFullscreen && window->autoIconify)
502 {
503 if (!activated || !fullscreen)
504 {
505 _glfwIconifyWindowWayland(window);
506 window->wl.wasFullscreen = GLFW_FALSE;
507 }
508 }
509 if (fullscreen && activated)
510 window->wl.wasFullscreen = GLFW_TRUE;
511}

711 cursor = window->wl.currentCursor;
712 if (cursor && cursor->wl.cursor)
713 {
714 cursor->wl.currentImage += 1;
715 cursor->wl.currentImage %= cursor->wl.cursor->image_count;
716 setCursorImage(window, &cursor->wl);

720static void handleEvents(int timeout)
721{
722 struct pollfd fds[] =
723 {
724 { wl_display_get_fd(_glfw.wl.display), POLLIN },
725 { _glfw.wl.timerfd, POLLIN },
726 { _glfw.wl.cursorTimerfd, POLLIN },
727 };
728
729 while (wl_display_prepare_read(_glfw.wl.display) != 0)
730 wl_display_dispatch_pending(_glfw.wl.display);
731
732 // If an error other than EAGAIN happens, we have likely been disconnected
733 // from the Wayland session; try to handle that the best we can.
734 if (wl_display_flush(_glfw.wl.display) < 0 && errno != EAGAIN)
735 {
736 _GLFWwindow* window = _glfw.windowListHead;
737 while (window)
738 {
739 _glfwInputWindowCloseRequest(window);
740 window = window->next;
741 }
742
743 wl_display_cancel_read(_glfw.wl.display);
744 return;
745 }
746
747 if (poll(fds, 3, timeout) > 0)
748 {
749 if (fds[0].revents & POLLIN)
750 {
751 wl_display_read_events(_glfw.wl.display);
752 wl_display_dispatch_pending(_glfw.wl.display);
753 }
754 else
755 wl_display_cancel_read(_glfw.wl.display);
756
757 if (fds[1].revents & POLLIN)
758 {
759 uint64_t repeats;
760
761 if (read(_glfw.wl.timerfd, &repeats, sizeof(repeats)) == 8)
762 {
763 for (uint64_t i = 0; i < repeats; i++)
764 {
765 _glfwInputKey(_glfw.wl.keyboardFocus,
766 _glfw.wl.keyboardLastKey,
767 _glfw.wl.keyboardLastScancode,
768 GLFW_PRESS,
769 _glfw.wl.xkb.modifiers);
770 _glfwInputTextWayland(_glfw.wl.keyboardFocus,
771 _glfw.wl.keyboardLastScancode);
772 }
773 }
774 }
775
776 if (fds[2].revents & POLLIN)
777 {
778 uint64_t repeats;
779
780 if (read(_glfw.wl.cursorTimerfd, &repeats, sizeof(repeats)) == 8)
781 incrementCursorImage(_glfw.wl.pointerFocus);
782 }
783 }
784 else
785 wl_display_cancel_read(_glfw.wl.display);
786}

1204const char* _glfwGetScancodeNameWayland(int scancode)
1205{
1206 if (scancode < 0 || scancode > 255 ||
1207 _glfw.wl.keycodes[scancode] == GLFW_KEY_UNKNOWN)
1208 {
1209 _glfwInputError(GLFW_INVALID_VALUE,
1210 "Wayland: Invalid scancode %i",
1211 scancode);
1212 return NULL;
1213 }
1214
1215 const int key = _glfw.wl.keycodes[scancode];
1216 const xkb_keycode_t keycode = scancode + 8;
1217 const xkb_layout_index_t layout =
1218 xkb_state_key_get_layout(_glfw.wl.xkb.state, keycode);
1219 if (layout == XKB_LAYOUT_INVALID)
1220 {
1221 _glfwInputError(GLFW_PLATFORM_ERROR,
1222 "Wayland: Failed to retrieve layout for key name");
1223 return NULL;
1224 }
1225
1226 const xkb_keysym_t* keysyms = NULL;
1227 xkb_keymap_key_get_syms_by_level(_glfw.wl.xkb.keymap,
1228 keycode,
1229 layout,
1230 0,
1231 &keysyms);
1232 if (keysyms == NULL)
1233 {
1234 _glfwInputError(GLFW_PLATFORM_ERROR,
1235 "Wayland: Failed to retrieve keysym for key name");
1236 return NULL;
1237 }
1238
1239 const uint32_t codepoint = _glfwKeySym2Unicode(keysyms[0]);
1240 if (codepoint == GLFW_INVALID_CODEPOINT)
1241 {
1242 _glfwInputError(GLFW_PLATFORM_ERROR,
1243 "Wayland: Failed to retrieve codepoint for key name");
1244 return NULL;
1245 }
1246
1247 const size_t count = _glfwEncodeUTF8(_glfw.wl.keynames[key], codepoint);
1248 if (count == 0)
1249 {
1250 _glfwInputError(GLFW_PLATFORM_ERROR,
1251 "Wayland: Failed to encode codepoint for key name");
1252 return NULL;
1253 }
1254
1255 _glfw.wl.keynames[key][count] = '\0';
1256 return _glfw.wl.keynames[key];
1257}
