101// Apply chosen cursor mode to a focused window
102//
103static void updateCursorMode(_GLFWwindow* window)
104{
105 if (window->cursorMode == GLFW_CURSOR_DISABLED)
106 {
107 _glfw.ns.disabledCursorWindow = window;
108 _glfwGetCursorPosCocoa(window,
109 &_glfw.ns.restoreCursorPosX,
110 &_glfw.ns.restoreCursorPosY);
111 _glfwCenterCursorInContentArea(window);
112 CGAssociateMouseAndMouseCursorPosition(false);
113 }
114 else if (_glfw.ns.disabledCursorWindow == window)
115 {
116 _glfw.ns.disabledCursorWindow = NULL;
117 _glfwSetCursorPosCocoa(window,
118 _glfw.ns.restoreCursorPosX,
119 _glfw.ns.restoreCursorPosY);
120 // NOTE: The matching CGAssociateMouseAndMouseCursorPosition call is
121 // made in _glfwSetCursorPosCocoa as part of a workaround
122 }
123
124 if (cursorInContentArea(window))
125 updateCursorImage(window);
126}

155// Translates macOS key modifiers into GLFW ones
156//
157static int translateFlags(NSUInteger flags)
158{
159 int mods = 0;
160
161 if (flags & NSEventModifierFlagShift)
162 mods |= GLFW_MOD_SHIFT;
163 if (flags & NSEventModifierFlagControl)
164 mods |= GLFW_MOD_CONTROL;
165 if (flags & NSEventModifierFlagOption)
166 mods |= GLFW_MOD_ALT;
167 if (flags & NSEventModifierFlagCommand)
168 mods |= GLFW_MOD_SUPER;
169 if (flags & NSEventModifierFlagCapsLock)
170 mods |= GLFW_MOD_CAPS_LOCK;
171
172 return mods;
173}

175// Translates a macOS keycode to a GLFW keycode
176//
177static int translateKey(unsigned int key)
178{
179 if (key >= sizeof(_glfw.ns.keycodes) / sizeof(_glfw.ns.keycodes[0]))
180 return GLFW_KEY_UNKNOWN;
181
182 return _glfw.ns.keycodes[key];
183}

185// Translate a GLFW keycode to a Cocoa modifier flag
186//
187static NSUInteger translateKeyToModifierFlag(int key)
188{
189 switch (key)
190 {
191 case GLFW_KEY_LEFT_SHIFT:
192 case GLFW_KEY_RIGHT_SHIFT:
193 return NSEventModifierFlagShift;
194 case GLFW_KEY_LEFT_CONTROL:
195 case GLFW_KEY_RIGHT_CONTROL:
196 return NSEventModifierFlagControl;
197 case GLFW_KEY_LEFT_ALT:
198 case GLFW_KEY_RIGHT_ALT:
199 return NSEventModifierFlagOption;
200 case GLFW_KEY_LEFT_SUPER:
201 case GLFW_KEY_RIGHT_SUPER:
202 return NSEventModifierFlagCommand;
203 case GLFW_KEY_CAPS_LOCK:
204 return NSEventModifierFlagCapsLock;
205 }
206
207 return 0;
208}

634- (BOOL)performDragOperation:(id <NSDraggingInfo>)sender
635{
636 const NSRect contentRect = [window->ns.view frame];
637 // NOTE: The returned location uses base 0,1 not 0,0
638 const NSPoint pos = [sender draggingLocation];
639 _glfwInputCursorPos(window, pos.x, contentRect.size.height - pos.y);
640
641 NSPasteboard* pasteboard = [sender draggingPasteboard];
642 NSDictionary* options = @{NSPasteboardURLReadingFileURLsOnlyKey:@YES};
643 NSArray* urls = [pasteboard readObjectsForClasses:@[[NSURL class]]
644 options:options];
645 const NSUInteger count = [urls count];
646 if (count)
647 {
648 char** paths = _glfw_calloc(count, sizeof(char*));
649
650 for (NSUInteger i = 0; i < count; i++)
651 paths[i] = _glfw_strdup([urls[i] fileSystemRepresentation]);
652
653 _glfwInputDrop(window, (int) count, (const char**) paths);
654
655 for (NSUInteger i = 0; i < count; i++)
656 _glfw_free(paths[i]);
657 _glfw_free(paths);
658 }
659
660 return YES;
661}

720- (void)insertText:(id)string replacementRange:(NSRange)replacementRange
721{
722 NSString* characters;
723 NSEvent* event = [NSApp currentEvent];
724 const int mods = translateFlags([event modifierFlags]);
725 const int plain = !(mods & GLFW_MOD_SUPER);
726
727 if ([string isKindOfClass:[NSAttributedString class]])
728 characters = [string string];
729 else
730 characters = (NSString*) string;
731
732 NSRange range = NSMakeRange(0, [characters length]);
733 while (range.length)
734 {
735 uint32_t codepoint = 0;
736
737 if ([characters getBytes:&codepoint
738 maxLength:sizeof(codepoint)
739 usedLength:NULL
740 encoding:NSUTF32StringEncoding
741 options:0
742 range:range
743 remainingRange:&range])
744 {
745 if (codepoint >= 0xf700 && codepoint <= 0xf7ff)
746 continue;
747
748 _glfwInputChar(window, codepoint, mods, plain);
749 }
750 }

783// Create the Cocoa window
784//
785static GLFWbool createNativeWindow(_GLFWwindow* window,
786 const _GLFWwndconfig* wndconfig,
787 const _GLFWfbconfig* fbconfig)
788{
789 window->ns.delegate = [[GLFWWindowDelegate alloc] initWithGlfwWindow:window];
790 if (window->ns.delegate == nil)
791 {
792 _glfwInputError(GLFW_PLATFORM_ERROR,
793 "Cocoa: Failed to create window delegate");
794 return GLFW_FALSE;
795 }
796
797 NSRect contentRect;
798
799 if (window->monitor)
800 {
801 GLFWvidmode mode;
802 int xpos, ypos;
803
804 _glfwGetVideoModeCocoa(window->monitor, &mode);
805 _glfwGetMonitorPosCocoa(window->monitor, &xpos, &ypos);
806
807 contentRect = NSMakeRect(xpos, ypos, mode.width, mode.height);
808 }
809 else
810 contentRect = NSMakeRect(0, 0, wndconfig->width, wndconfig->height);
811
812 window->ns.object = [[GLFWWindow alloc]
813 initWithContentRect:contentRect
814 styleMask:getStyleMask(window)
815 backing:NSBackingStoreBuffered
816 defer:NO];
817
818 if (window->ns.object == nil)
819 {
820 _glfwInputError(GLFW_PLATFORM_ERROR, "Cocoa: Failed to create window");
821 return GLFW_FALSE;
822 }
823
824 if (window->monitor)
825 [window->ns.object setLevel:NSMainMenuWindowLevel + 1];
826 else
827 {
828 [(NSWindow*) window->ns.object center];
829 _glfw.ns.cascadePoint =
830 NSPointToCGPoint([window->ns.object cascadeTopLeftFromPoint:
831 NSPointFromCGPoint(_glfw.ns.cascadePoint)]);
832
833 if (wndconfig->resizable)
834 {
835 const NSWindowCollectionBehavior behavior =
836 NSWindowCollectionBehaviorFullScreenPrimary |
837 NSWindowCollectionBehaviorManaged;
838 [window->ns.object setCollectionBehavior:behavior];
839 }
840
841 if (wndconfig->floating)
842 [window->ns.object setLevel:NSFloatingWindowLevel];
843
844 if (wndconfig->maximized)

1543const char* _glfwGetScancodeNameCocoa(int scancode)
1544{
1545 @autoreleasepool {
1546
1547 if (scancode < 0 || scancode > 0xff ||
1548 _glfw.ns.keycodes[scancode] == GLFW_KEY_UNKNOWN)
1549 {
1550 _glfwInputError(GLFW_INVALID_VALUE, "Invalid scancode %i", scancode);
1551 return NULL;
1552 }
1553
1554 const int key = _glfw.ns.keycodes[scancode];
1555
1556 UInt32 deadKeyState = 0;
1557 UniChar characters[4];
1558 UniCharCount characterCount = 0;
1559
1560 if (UCKeyTranslate([(NSData*) _glfw.ns.unicodeData bytes],
1561 scancode,
1562 kUCKeyActionDisplay,
1563 0,
1564 LMGetKbdType(),
1565 kUCKeyTranslateNoDeadKeysBit,
1566 &deadKeyState,
1567 sizeof(characters) / sizeof(characters[0]),
1568 &characterCount,
1569 characters) != noErr)
1570 {
1571 return NULL;
1572 }
1573
1574 if (!characterCount)
1575 return NULL;
1576
1577 CFStringRef string = CFStringCreateWithCharactersNoCopy(kCFAllocatorDefault,
1578 characters,
1579 characterCount,
1580 kCFAllocatorNull);
1581 CFStringGetCString(string,
1582 _glfw.ns.keynames[key],
1583 sizeof(_glfw.ns.keynames[key]),
1584 kCFStringEncodingUTF8);
