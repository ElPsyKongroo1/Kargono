23 void EmitterConfigManager::SerializeAsset(Ref<Particles::EmitterConfig> assetReference, const std::filesystem::path& assetPath)
24 {
25 YAML::Emitter out;
26 out << YAML::BeginMap; // Start of File Map
27
28 // Emitter specific data
29 out << YAML::Key << "BufferSize" << YAML::Value << assetReference->m_BufferSize;
30 out << YAML::Key << "SpawnRatePerSec" << YAML::Value << assetReference->m_SpawnRatePerSec;
31 out << YAML::Key << "EmitterLifecycle" << YAML::Value <<
32 Utility::EmitterLifecycleToString(assetReference->m_EmitterLifecycle);
33 out << YAML::Key << "EmitterLifetime" << YAML::Value << assetReference->m_EmitterLifetime;
34
35 // Spawning bounds sequence
36 out << YAML::Key << "SpawningBounds" << YAML::Value << YAML::BeginSeq; // Start spawning bounds sequence
37 out << YAML::Value << assetReference->m_SpawningBounds[0];
38 out << YAML::Value << assetReference->m_SpawningBounds[1];
39 out << YAML::EndSeq; // End of spawning bounds sequence
40
41 // Particle specific data
42 out << YAML::Key << "UseGravity" << YAML::Value << assetReference->m_UseGravity;
43 out << YAML::Key << "GravityAcceleration" << YAML::Value << assetReference->m_GravityAcceleration;
44 out << YAML::Key << "ParticleLifetime" << YAML::Value << assetReference->m_ParticleLifetime;
45 out << YAML::Key << "ColorInterpolationType" << YAML::Value <<
46 Utility::InterpolationTypeToString(assetReference->m_ColorInterpolationType);
47 out << YAML::Key << "ColorBegin" << YAML::Value << assetReference->m_ColorBegin;
48 out << YAML::Key << "ColorEnd" << YAML::Value << assetReference->m_ColorEnd;
49
50 out << YAML::Key << "SizeInterpolationType" << YAML::Value <<
51 Utility::InterpolationTypeToString(assetReference->m_SizeInterpolationType);
52 out << YAML::Key << "SizeBegin" << YAML::Value << assetReference->m_SizeBegin;
53 out << YAML::Key << "SizeEnd" << YAML::Value << assetReference->m_SizeEnd;

55 out << YAML::EndMap; // End of File Map

62 {
63 Ref<Particles::EmitterConfig> newEmitterConfig = CreateRef<Particles::EmitterConfig>();
64 YAML::Node data;
65 try
66 {
67 data = YAML::LoadFile(assetPath.string());

70 {
71 KG_WARN("Failed to load .kgui file '{0}'\n {1}", assetPath, e.what());
72 return nullptr;
73 }
74
75 // Get emitter specific data from YAML
76 newEmitterConfig->m_BufferSize = data["BufferSize"].as<size_t>();
77 newEmitterConfig->m_SpawnRatePerSec = data["SpawnRatePerSec"].as<size_t>();
78 newEmitterConfig->m_EmitterLifecycle =
79 Utility::StringToEmitterLifecycle(data["EmitterLifecycle"].as<std::string>());
80 newEmitterConfig->m_EmitterLifetime = data["EmitterLifetime"].as<float>();
81 YAML::Node spawningBoundsNode = data["SpawningBounds"];
82 size_t iteration{ 0 };
83 for (YAML::iterator::value_type bound : spawningBoundsNode)
84 {
85 newEmitterConfig->m_SpawningBounds[iteration] = bound.as<Math::vec3>();
86 iteration++;
87 }
88
89 // Get particle specific data from YAML
90 newEmitterConfig->m_UseGravity = data["UseGravity"].as<bool>();
91 newEmitterConfig->m_GravityAcceleration = data["GravityAcceleration"].as<Math::vec3>();
92 newEmitterConfig->m_ParticleLifetime = data["ParticleLifetime"].as<float>();
93
94 // Get particle color data from YAML
95 newEmitterConfig->m_ColorInterpolationType =
96 Utility::StringToInterpolationType(data["ColorInterpolationType"].as<std::string>());
97 newEmitterConfig->m_ColorBegin = data["ColorBegin"].as<Math::vec4>();
98 newEmitterConfig->m_ColorEnd = data["ColorEnd"].as<Math::vec4>();
99

103 newEmitterConfig->m_SizeBegin = data["SizeBegin"].as<Math::vec3>();
