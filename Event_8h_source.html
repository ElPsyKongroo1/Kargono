8//============================================================
9// Events Namespace
10//============================================================
11// This namespace provides the main classes that allow the Event Pipeline
12// to work correctly. The intended use of classes in this namespace
13// is:
14// 1. Create an OnEvent(Event) function in the main application layer that
15// links to other OnEvent(Event) functions depending on the particular scenerio.
16// (These linkages constitute the event pipeline)
17// 2. Provide this main OnEvent(Event) function to an event generating section
18// of the engine such as the Windowing Library (GLFW currently) that can receive
19// input events such as KeyPressed events.
20// 3. Add Dispatchers throughout the OnEvent(Event) function pipeline that link
21// to local functions that handle events. The dispatchers check if the event type
22// of the currently running event matches the dispatcher's type.
23// 4. Implement logic to handle events in different scenerios using the m_Handled
24// member of the event class.
25namespace Kargono::Events

64 // This enum provides an easy method to determine if a particular event

89 // an abstract class that holds multiple getter functions that

92 // informs the event pipeline that the event has been used
93 // to run a function call.
94 class Event
95 {
96 public:
97 //==============================

99 //==============================
100 // This constructor is simply set to prevent object slicing
101 // during the destruction of the parent class.
102 virtual ~Event() = default;
103
104 //==============================
105 // Getters/Setters
106 //==============================
107 // These getter functions provide a method to identify event types
108 // and manage the events externally. The ToString function is
109 // for debugging.
110 virtual EventType GetEventType() const = 0;
111 virtual int GetCategoryFlags() const = 0;
112 // This function provides an easy method for determining if an event
113 // is part of a broader category since EventCategory's are bit
114 // fields.
115 bool IsInCategory(EventCategory category)

117 return GetCategoryFlags() & category;
118 }
