9 // Events Namespace
10 //============================================================
11 // This namespace provides the main classes that allow the Event Pipeline
12 // to work correctly. The intended use of classes in this namespace
13 // is:
14 // 1. Create an OnEvent(Event) function in the main application layer that
15 // links to other OnEvent(Event) functions depending on the particular scenerio.
16 // (These linkages constitute the event pipeline)
17 // 2. Provide this main OnEvent(Event) function to an event generating section
18 // of the engine such as the Windowing Library (GLFW currently) that can receive
19 // input events such as KeyPressed events.
20 // 3. Add Dispatchers throughout the OnEvent(Event) function pipeline that link
21 // to local functions that handle events. The dispatchers check if the event type
22 // of the currently running event matches the dispatcher's type.
23 // 4. Implement logic to handle events in different scenerios using the m_Handled
24 // member of the event class.
25 namespace Kargono::Events

52  // This enum provides an easy method to determine if a particular event

54  // The enum is structured as a bitfield to allow a particular event

63  Mouse = BIT(3),
64  MouseButton = BIT(4),

69  //============================================================
70  // Event Class
71  //============================================================
72  // This Event class provides a uniform interface for different event
73  // types to be routed through the same event pipeline. This is
74  // an abstract class that holds multiple getter functions that
75  // allow for easy identification of the event type among other
76  // properties. This class also has a Handled member which
77  // informs the event pipeline that the event has been used
78  // to run a function call.
79  class Event
80  {
81  public:
82  //==============================
83  // Constructors and Destructors
84  //==============================
85  // This constructor is simply set to prevent object slicing
86  // during the destruction of the parent class.
87  virtual ~Event() = default;
88 
89  //==============================
90  // Getters/Setters
91  //==============================
92  // These getter functions provide a method to identify event types
93  // and manage the events externally. The ToString function is
94  // for debugging.
95  virtual EventType GetEventType() const = 0;
96  virtual const char* GetName() const = 0;
97  virtual int GetCategoryFlags() const = 0;
98  virtual std::string ToString() const { return GetName(); }
99  // This function provides an easy method for determining if an event
100  // is part of a broader category since EventCategory's are bit
101  // fields.
102  bool IsInCategory(EventCategory category)
103  {
104  return GetCategoryFlags() & category;
105  }
106  // This variable determines if an event has been handled.
107  // A handled event can be managed by the event pipeline.
108  // Ex: An event that has ran its functionality can be
109  // set to handled. Later in the event pipeline, there
110  // can be an early-out if the event is labeled handled.
111  bool Handled = false;
112  };

115  //============================================================
116  // Event Dispatcher Class
117  //============================================================
118  // This class is a supporting functionality class for the event class. This
119  // class is meant to be instantiated inside of the event pipeline with
120  // the currently managed event. Once this class is instantiated,
121  // it can be used to dispatch an event (determine if the event type
122  // is correct for a particular function and run the function if correct).
123  class EventDispatcher
124  {
125  // Templated alias for a function pointer is used in the
126  // Dispatch function.
127  template<typename K>
128  using EventFn = std::function<bool(K&)>;
129  public:
130  //==============================
131  // Constructors and Destructors
132  //==============================
133  // This constructor simply initializes the dispatcher with an event
134  // to handle. This construction method allows for one easy
135  // connection between the event and the dispatcher to be made.
136  // Multiple dispatches can be run with different function pointers
137  // with the same dispatch object and event.
138  EventDispatcher(Event& event) : m_Event(event)
139  {
140  }
141  //==============================
142  // Execute and Handle Event
143  //==============================
144  // This function provides a method for ensuring the event initialized with
145  // m_Event has the same EventType as the template type T. Once the
146  // event types match, the provided function is called and the event is
147  // set to handled. This allows for an easy API to externally handle
148  // different scenerios involving event management inside the event
149  // pipeline.
150  template<typename T>
151  bool Dispatch(EventFn<T> func)
152  {
153  if (m_Event.GetEventType() == T::GetStaticType())
154  {
155  m_Event.Handled |= func(*(T*)&m_Event);
156  return true;
157  }
158  return false;
159  }
160  private:
161  // m_Event holds a reference to the currently managed event inside the
162  // event pipeline. This event is stored internally so that it does
163  // not need to be provided to the Dispatch call repeatadly.
164  Event& m_Event;
165  };
