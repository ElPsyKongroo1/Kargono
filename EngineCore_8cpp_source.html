128  void EngineCore::Run()
129  {
130  KG_INFO("Starting Run Function");
131  using namespace std::chrono_literals;
132 
133  std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
134  std::chrono::time_point<std::chrono::high_resolution_clock> lastCycleTime = currentTime;
135  std::chrono::nanoseconds timestep{ 0 };
136 
137  while (m_Running)
138  {
139  currentTime = std::chrono::high_resolution_clock::now();
140  timestep = currentTime - lastCycleTime;
141  lastCycleTime = currentTime;
142  m_Accumulator += timestep;
143  if (m_Accumulator < k_ConstantFrameTime)
144  {
145  continue;
146  }
147  m_Accumulator -= k_ConstantFrameTime;
148 
149  {
150  KG_PROFILE_FRAME("Main Thread");

157  {
158  m_CurrentApp->OnUpdate(k_ConstantFrameTimeStep);
159  }
160  ProcessEventQueue();
161  m_Window->OnUpdate();
162  }
