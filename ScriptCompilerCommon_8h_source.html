286 ScriptToken GetReturnType()
287 {
288 if (FunctionCallNode* functionCallNodePtr = std::get_if<FunctionCallNode>(&Value))
289 {
290 FunctionCallNode& functionCallNode = *functionCallNodePtr;
291 return functionCallNode.ReturnType;
292 }
293 else if (UnaryOperationNode* unaryOperationNodePtr = std::get_if<UnaryOperationNode>(&Value))
294 {
295 UnaryOperationNode& unaryOperationNode = *unaryOperationNodePtr;
296 return unaryOperationNode.ReturnType;
297 }
298 else if (BinaryOperationNode* binaryOperationNodePtr = std::get_if<BinaryOperationNode>(&Value))
299 {
300 BinaryOperationNode& binaryOperationNode = *binaryOperationNodePtr;
301 return binaryOperationNode.ReturnType;
302 }
303 else if (InitializationListNode* initializationListNodePtr = std::get_if<InitializationListNode>(&Value))
304 {
305 InitializationListNode& initializationListNode = *initializationListNodePtr;
306 return initializationListNode.ReturnType;
307 }
308 else if (TernaryOperationNode* ternaryOperationNodePtr = std::get_if<TernaryOperationNode>(&Value))
309 {
310 TernaryOperationNode& ternaryOperationNode = *ternaryOperationNodePtr;
311 return ternaryOperationNode.ReturnType;
312 }
313 else if (TokenExpressionNode* scriptTokenPtr = std::get_if<TokenExpressionNode>(&Value))
314 {
315 TokenExpressionNode& tokenExpressionNode = *scriptTokenPtr;
316 return tokenExpressionNode.ReturnType;
317 }
318 else if (MemberNode* memberNodePtr = std::get_if<MemberNode>(&Value))
319 {
320 MemberNode& memberNode = *memberNodePtr;
321 return memberNode.ReturnType;
322 }
323
324 KG_WARN("Invalid expression type attempted to be parsed");
325 return {};
