295
296 ScriptToken GetReturnType()
297 {
298 if (FunctionCallNode* functionCallNodePtr = std::get_if<FunctionCallNode>(&Value))
299 {
300 FunctionCallNode& functionCallNode = *functionCallNodePtr;
301 return functionCallNode.ReturnType;
302 }
303 else if (CustomLiteralNode* customLiteralNodePtr = std::get_if<CustomLiteralNode>(&Value))
304 {
305 CustomLiteralNode& customLiteralNode = *customLiteralNodePtr;
306 return customLiteralNode.ReturnType;
307 }
308 else if (UnaryOperationNode* unaryOperationNodePtr = std::get_if<UnaryOperationNode>(&Value))
309 {
310 UnaryOperationNode& unaryOperationNode = *unaryOperationNodePtr;
311 return unaryOperationNode.ReturnType;
312 }
313 else if (BinaryOperationNode* binaryOperationNodePtr = std::get_if<BinaryOperationNode>(&Value))
314 {
315 BinaryOperationNode& binaryOperationNode = *binaryOperationNodePtr;
316 return binaryOperationNode.ReturnType;
317 }
318 else if (InitializationListNode* initializationListNodePtr = std::get_if<InitializationListNode>(&Value))
319 {
320 InitializationListNode& initializationListNode = *initializationListNodePtr;
321 return initializationListNode.ReturnType;
322 }
323 else if (TernaryOperationNode* ternaryOperationNodePtr = std::get_if<TernaryOperationNode>(&Value))
324 {
325 TernaryOperationNode& ternaryOperationNode = *ternaryOperationNodePtr;
326 return ternaryOperationNode.ReturnType;
327 }
328 else if (TokenExpressionNode* scriptTokenPtr = std::get_if<TokenExpressionNode>(&Value))
329 {
330 TokenExpressionNode& tokenExpressionNode = *scriptTokenPtr;
331 return tokenExpressionNode.ReturnType;
332 }
333 else if (MemberNode* memberNodePtr = std::get_if<MemberNode>(&Value))
334 {
335 MemberNode& memberNode = *memberNodePtr;
336 return memberNode.ReturnType;
337 }
338
339 KG_WARN("Invalid expression type attempted to be parsed");
340 return {};
341 }
