165 void AssetManager::SerializeEntityClass(Ref<Scenes::EntityClass> EntityClass, const std::filesystem::path& filepath)
166 {
167 YAML::Emitter out;
168 out << YAML::BeginMap; // Start of File Map
169
170 out << YAML::Key << "Name" << YAML::Value << EntityClass->GetName(); // Output State Name
171 out << YAML::Key << "FieldTypes" << YAML::Value;
172 out << YAML::BeginSeq; // Start Fields
173
174 for (auto& [name, field] : EntityClass->m_FieldTypes)
175 {
176 out << YAML::BeginMap; // Start Field
177
178 out << YAML::Key << "Name" << YAML::Value << name; // Name/Map Key

180
181 out << YAML::EndMap; // End Field
182 }
183
184 out << YAML::EndSeq; // End Fields
185
186 out << YAML::Key << "OnPhysicsCollisionStart" << YAML::Value <<
187 static_cast<uint64_t>(EntityClass->m_Scripts.OnPhysicsCollisionStartHandle);
188 out << YAML::Key << "OnPhysicsCollisionEnd" << YAML::Value <<
189 static_cast<uint64_t>(EntityClass->m_Scripts.OnPhysicsCollisionEndHandle);

191 static_cast<uint64_t>(EntityClass->m_Scripts.OnCreateHandle);
192 out << YAML::Key << "OnUpdate" << YAML::Value <<
193 static_cast<uint64_t>(EntityClass->m_Scripts.OnUpdateHandle);
194
195 out << YAML::Key << "AllScripts" << YAML::Value;
196 out << YAML::BeginSeq; // Start AllScripts
197
198 for (auto& script : EntityClass->m_Scripts.AllClassScripts)
199 {
200 out << YAML::Value << static_cast<uint64_t>(script); // Script ID
201 }
202
203 out << YAML::EndSeq; // End AllScripts
204
205 out << YAML::EndMap; // End of File Map
206
207 std::ofstream fout(filepath);
208 fout << out.c_str();
209 KG_INFO("Successfully Serialized EntityClass at {}", filepath);

278 scripts.OnPhysicsCollisionStart =

280 }

305 {

307 if (allScripts)

309 std::set<Assets::AssetHandle>& classScripts = EntityClass->m_Scripts.AllClassScripts;

322 {
323 // Create Checksum

325

358 void AssetManager::SaveEntityClass(AssetHandle entityClassHandle, Ref<Scenes::EntityClass> entityClass, Ref<Scenes::Scene> editorScene)
359 {
360 if (!s_EntityClassRegistry.contains(entityClassHandle))
361 {
362 KG_ERROR("Attempt to save EntityClass that does not exist in registry");
363 return;
364 }
365 // Store Field Location for Processing ClassInstanceComponent Data
366 std::vector<Scenes::ClassField> oldFields = GetEntityClass(entityClassHandle)->GetFields();
367 std::vector<Scenes::ClassField> newFields = entityClass->GetFields();
368
369 Assets::Asset EntityClassAsset = s_EntityClassRegistry[entityClassHandle];
370 SerializeEntityClass(entityClass, (Projects::ProjectService::GetActiveAssetDirectory() / EntityClassAsset.Data.IntermediateLocation).string());
371
372 // Create map that associates old data locations with new data locations for ClassInstanceComponents
373 std::unordered_map<uint32_t, uint32_t> transferFieldDataMap {};
374 uint32_t newFieldsIteration{ 0 };
375 for (auto& [newName, newType] : newFields)
376 {
377 uint32_t oldFieldsIteration{ 0 };
378 for (auto& [oldName, oldType] : oldFields)
379 {
380 if (oldName == newName && oldType == newType)
381 {
382 transferFieldDataMap.insert_or_assign(oldFieldsIteration, newFieldsIteration);
383 break;
384 }
385 oldFieldsIteration++;
386 }
387 newFieldsIteration++;
388 }
389
390 // Update Active Scene
391 Ref<Scenes::EntityClass> newEntityClass = GetEntityClass(entityClassHandle);
392 Ref<Scenes::Scene> activeScene = Scenes::SceneService::GetActiveScene();
393
394 Utility::TransferClassInstanceFieldData(activeScene, newEntityClass, entityClassHandle, transferFieldDataMap);
395 // Update Editor Scene if applicable
396 if (editorScene)
397 {
398 Utility::TransferClassInstanceFieldData(editorScene, newEntityClass,entityClassHandle, transferFieldDataMap);
399 }
400
401 for (auto& [handle, asset] : s_SceneRegistry)
402 {
403 const Ref<Scenes::Scene> scene = GetScene(handle);
404 if (!scene)
405 {
406 KG_WARN("Unable to load scene in SaveEntityClass");
407 continue;
408 }
409
410 Utility::TransferClassInstanceFieldData(scene, newEntityClass, entityClassHandle, transferFieldDataMap);
411 SaveScene(handle, scene);
412 }
