58 static std::unordered_map<GLenum, std::string> PreProcess(const std::string& source)
59 {
60 std::unordered_map<GLenum, std::string> shaderSources;
61
62 const char* typeToken = "#type";
63 size_t typeTokenLength = strlen(typeToken);
64 size_t pos = source.find(typeToken, 0); //Start of shader type declaration line
65 while (pos != std::string::npos)
66 {
67 size_t eol = source.find_first_of("\r\n", pos); //End of shader type declaration line
68 KG_ASSERT(eol != std::string::npos, "Syntax error");
69 size_t begin = pos + typeTokenLength + 1; //Start of shader type name (after "#type " keyword)
70 std::string type = source.substr(begin, eol - begin);
71 KG_ASSERT(Utility::ShaderTypeFromString(type), "Invalid shader type specified");
72
73 size_t nextLinePos = source.find_first_not_of("\r\n", eol); //Start of shader code after shader type declaration line
74 KG_ASSERT(nextLinePos != std::string::npos, "Syntax error");
75 pos = source.find(typeToken, nextLinePos); //Start of next shader type declaration line
76
77 shaderSources[Utility::ShaderTypeFromString(type)] = (pos == std::string::npos) ? source.substr(nextLinePos) : source.substr(nextLinePos, pos - nextLinePos);
78 }
79
80 return shaderSources;
81 }

265 Ref<Rendering::Shader> Assets::ShaderManager::DeserializeAsset(Assets::AssetInfo& asset, const std::filesystem::path& assetPath)
266 {
267 Assets::ShaderMetaData metadata = *asset.Data.GetSpecificMetaData<ShaderMetaData>();
268 std::unordered_map<GLenum, std::vector<uint32_t>> openGLSPIRV;
269 std::filesystem::path intermediatePath = Projects::ProjectService::GetActiveIntermediateDirectory() / asset.Data.IntermediateLocation;
270 static std::array<std::string, 2> stageTypes = { "vertex", "fragment" };
271
272 for (const auto& stage : stageTypes)
273 {
274 std::filesystem::path fullPath = intermediatePath.string() + Utility::ShaderBinaryFileExtension(Utility::ShaderTypeFromString(stage));
275
276 std::ifstream in(fullPath, std::ios::in | std::ios::binary);
277
278 if (in.is_open())
279 {
280 in.seekg(0, std::ios::end);
281 auto size = in.tellg();
282 in.seekg(0, std::ios::beg);
283
284 auto& data = openGLSPIRV[Utility::ShaderTypeFromString(stage)];
285 data.resize(size / sizeof(uint32_t));
286 in.read((char*)data.data(), size);
287 }
288 }
289
290 Ref<Kargono::Rendering::Shader> newShader = Rendering::Shader::Create(static_cast<std::string>(asset.Handle), openGLSPIRV);
291 newShader->SetSpecification(metadata.ShaderSpec);
292 newShader->SetInputLayout(metadata.InputLayout);
293 newShader->SetUniformList(metadata.UniformList);
294 openGLSPIRV.clear();
295 return newShader;
