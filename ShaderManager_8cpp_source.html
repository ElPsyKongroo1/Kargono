70  static std::unordered_map<GLenum, std::string> PreProcess(const std::string& source)
71  {
72  std::unordered_map<GLenum, std::string> shaderSources;
73 
74  const char* typeToken = "#type";
75  size_t typeTokenLength = strlen(typeToken);
76  size_t pos = source.find(typeToken, 0); //Start of shader type declaration line
77  while (pos != std::string::npos)
78  {
79  size_t eol = source.find_first_of("\r\n", pos); //End of shader type declaration line
80  KG_ASSERT(eol != std::string::npos, "Syntax error");
81  size_t begin = pos + typeTokenLength + 1; //Start of shader type name (after "#type " keyword)
82  std::string type = source.substr(begin, eol - begin);
83  KG_ASSERT(Utility::ShaderTypeFromString(type), "Invalid shader type specified");
84 
85  size_t nextLinePos = source.find_first_not_of("\r\n", eol); //Start of shader code after shader type declaration line
86  KG_ASSERT(nextLinePos != std::string::npos, "Syntax error");
87  pos = source.find(typeToken, nextLinePos); //Start of next shader type declaration line
88 
89  shaderSources[Utility::ShaderTypeFromString(type)] = (pos == std::string::npos) ? source.substr(nextLinePos) : source.substr(nextLinePos, pos - nextLinePos);
90  }
91 
92  return shaderSources;
