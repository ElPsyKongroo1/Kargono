59 static std::unordered_map<GLenum, std::string> PreProcess(const std::string& source)
60 {
61 std::unordered_map<GLenum, std::string> shaderSources;
62
63 const char* typeToken = "#type";
64 size_t typeTokenLength = strlen(typeToken);
65 size_t pos = source.find(typeToken, 0); //Start of shader type declaration line
66 while (pos != std::string::npos)
67 {
68 size_t eol = source.find_first_of("\r\n", pos); //End of shader type declaration line
69 KG_ASSERT(eol != std::string::npos, "Syntax error");
70 size_t begin = pos + typeTokenLength + 1; //Start of shader type name (after "#type " keyword)
71 std::string type = source.substr(begin, eol - begin);
72 KG_ASSERT(Utility::ShaderTypeFromString(type), "Invalid shader type specified");
73
74 size_t nextLinePos = source.find_first_not_of("\r\n", eol); //Start of shader code after shader type declaration line
75 KG_ASSERT(nextLinePos != std::string::npos, "Syntax error");
76 pos = source.find(typeToken, nextLinePos); //Start of next shader type declaration line
77
78 shaderSources[Utility::ShaderTypeFromString(type)] = (pos == std::string::npos) ? source.substr(nextLinePos) : source.substr(nextLinePos, pos - nextLinePos);
79 }
80
81 return shaderSources;
82 }

307 AssetHandle AssetManager::CreateNewShader(const Rendering::ShaderSpecification& shaderSpec)
308 {
309 // Create Checksum
310 auto [shaderSource, bufferLayout, uniformList] = Rendering::ShaderBuilder::BuildShader(shaderSpec);
311 std::string currentCheckSum = Utility::FileSystem::ChecksumFromString(shaderSource);
312
313 if (currentCheckSum.empty())
314 {
315 KG_ERROR("Failed to generate checksum from shaderSpec!");
316 return {};
317 }
318
319 // Compare currentChecksum to registered shaders
320 bool isAssetDuplicate = false;
321 AssetHandle currentHandle{};
322 for (const auto& [handle, asset] : s_ShaderRegistry)
323 {
324 if (asset.Data.CheckSum == currentCheckSum)
325 {
326 isAssetDuplicate = true;
327 currentHandle = handle;
328 break;
329 }
330 }
331
332 // If Asset is not a duplicate but has a similar spec to another asset, replace that asset! (Delete old asset and use it's UUID)
333 if (isAssetDuplicate)
334 {
335 return currentHandle;
336 }
337 // Create New Asset and Asset Handle
338 AssetHandle newHandle{};
339
340 Assets::Asset newAsset{};
341 newAsset.Handle = newHandle;
342
343 // Create intermediate and save intermediate to disk
344 CreateShaderIntermediate(shaderSource, newAsset, shaderSpec, bufferLayout, uniformList);
345 newAsset.Data.CheckSum = currentCheckSum;
346
347 // Register New Asset and Create Shader
348 s_ShaderRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
349 SerializeShaderRegistry(); // Update Registry File on Disk
350 Ref<Kargono::Rendering::Shader> newShader = InstantiateShaderIntoMemory(newAsset);
351 s_Shaders.insert({ newHandle, newShader });
352
353 return newHandle;
354 }

410 Ref<Kargono::Rendering::Shader> AssetManager::InstantiateShaderIntoMemory(Assets::Asset& asset)
411 {
412 Assets::ShaderMetaData metadata = *static_cast<Assets::ShaderMetaData*>(asset.Data.SpecificFileData.get());
413 std::unordered_map<GLenum, std::vector<uint32_t>> openGLSPIRV;
414 std::filesystem::path intermediatePath = Projects::ProjectService::GetActiveAssetDirectory() / asset.Data.IntermediateLocation;
415 std::vector<std::string> stageTypes = { "vertex", "fragment" };
416
417 for (const auto& stage : stageTypes)
418 {
419 std::filesystem::path fullPath = intermediatePath.string() + Utility::ShaderBinaryFileExtension(Utility::ShaderTypeFromString(stage));
420
421 std::ifstream in(fullPath, std::ios::in | std::ios::binary);
422
423 if (in.is_open())
424 {
425 in.seekg(0, std::ios::end);
426 auto size = in.tellg();
427 in.seekg(0, std::ios::beg);
428
429 auto& data = openGLSPIRV[Utility::ShaderTypeFromString(stage)];
430 data.resize(size / sizeof(uint32_t));
431 in.read((char*)data.data(), size);
432 }
433 }
434
435 Ref<Kargono::Rendering::Shader> newShader = Rendering::Shader::Create(static_cast<std::string>(asset.Handle), openGLSPIRV);
436 newShader->SetSpecification(metadata.ShaderSpec);
437 newShader->SetInputLayout(metadata.InputLayout);
438 newShader->SetUniformList(metadata.UniformList);
439 openGLSPIRV.clear();
440 return newShader;
441 }
