726  bool ScriptModuleBuilder::CompileModuleCode(bool createDebug)
727  {
728  Utility::FileSystem::CreateNewDirectory(Projects::ProjectService::GetActiveAssetDirectory() / "Scripting/Binary");
729  std::filesystem::path binaryPath { Projects::ProjectService::GetActiveAssetDirectory() / "Scripting/Binary/" };
730  std::filesystem::path binaryFile;
731  std::filesystem::path objectPath;
732  if (createDebug)
733  {
734  binaryFile = { binaryPath / "ExportBodyDebug.dll" };
735  objectPath = { binaryPath / "ExportBody.obj" };
736  }
737  else
738  {
739  binaryFile = { binaryPath / "ExportBody.dll" };
740  objectPath = { binaryPath / "ExportBody.obj" };
741  }
742 
743  UUID pdbID = UUID();
744  std::string pdbFileName = std::string(pdbID) + ".pdb";
745  std::filesystem::path debugSymbolsPath { binaryPath / pdbFileName };
746  std::filesystem::path sourcePath { Projects::ProjectService::GetActiveAssetDirectory() / "Scripting/Binary/ExportBody.cpp" };
747 
748  std::stringstream outputStream {};
749  outputStream << "("; // Parentheses to group all function calls together
750  // Access visual studio toolset console
751  outputStream << "\"C:\\Program Files\\Microsoft Visual Studio\\2022\\Community\\VC\\Auxiliary\\Build\\vcvars64.bat\"";
752 
753  // Start Next Command
754  //outputStream << " > Log\\ScriptCompilation.log 2>&1 "; // Sends error/info to log file
755  outputStream << " && "; // Combine commands
756 
757  // Cl command for compiling binary code
758  outputStream << "cl "; // Add Command
759  outputStream << "/c "; // Tell cl command to only compile code
760  if (createDebug)
761  {
762  outputStream << "/MDd "; // Specify Debug Multi-threaded DLL Runtime Library
763  }
764  else
765  {
766  outputStream << "/MD "; // Specify Multi-threaded DLL Runtime Library
767  }
768  outputStream << "/std:c++20 "; // Specify Language Version
769  outputStream << "/I../Dependencies/glm "; // Include GLM
770  outputStream << "/I../Engine/Source "; // Include Kargono as Include Directory
771  outputStream << "/EHsc "; // Specifies the handling of exceptions and call stack unwinding. Uses the commands /EH, /EHs, and /EHc together
772  outputStream << "/DKARGONO_EXPORTS "; // Define Macros for Exporting DLL Functions
773 
774  if (createDebug)
775  {
776  outputStream << "/Z7 "; // Add debug info to executable
777  }
778  outputStream << "/Fo" << "\"" << binaryPath.string() << "\"" << ' '; // Define Intermediate Location
779  outputStream << "\"" << sourcePath.string() << "\"" << " "; // Compile scripting source file (ExportBody.cpp)
780 
781  // Start Next Command
782  //outputStream << " > Log\\ScriptCompilation.log 2>&1 "; // Sends error/info to log file
783  outputStream << " && "; // Combine commands
784 
785  // Start Linking Stage
786  outputStream << "link "; // Start link command
787  outputStream << "/DLL "; // Specify output as a shared library
788  outputStream << "/ignore:4099 "; // Ignores warning about missing pbd files for .obj files (I generate the symbols inside of the .obj file with /Z7 flag)
789  if (createDebug)
790  {
791  outputStream << "/DEBUG "; // Specifies output as debug files
792  outputStream << "/PDB:" << "\"" << debugSymbolsPath.string() << "\"" << " "; // Specify .pdb file location/name
793  }
794  outputStream << "/OUT:" << "\"" << binaryFile.string() << "\"" <<" "; // Specify output directory
795  outputStream << "\"" <<objectPath.string() << "\""; // Object File to Link
796 
797  outputStream << ")"; // Parentheses to group all function calls together
798 
799  // Sends all three calls (open dev console, compiler, and linker) error/info to log file
800  if (createDebug)
801  {
802  outputStream << " >> Log\\BuildScriptLibraryDebug.log 2>&1 ";
803  }
804  else
805  {
806  outputStream << " >> Log\\BuildScriptLibrary.log 2>&1 ";
807  }
808 
809  // Call Command
810  return (system(outputStream.str().c_str()) != 1);
