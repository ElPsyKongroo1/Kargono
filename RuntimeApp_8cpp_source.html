541 void RuntimeApp::HandleUIMouseHovering()
542 {
543 // Get the active viewport bounds and mouse position
544 Kargono::ViewportData& activeViewport = EngineService::GetActiveWindow().GetActiveViewport();
545 Math::vec2 mousePos = Input::InputService::GetMousePosition();
546
547 // Make sure the mouse position is within bounds
548 if ((int)mousePos.x < 0 ||
549 (int)mousePos.y < 0 ||
550 (int)mousePos.x > (int)activeViewport.m_Width ||
551 (int)mousePos.y > (int)activeViewport.m_Height)
552 {
553 return;
554 }
555
556 // Make sure the y-position is oriented correctly
557 mousePos.y = (float)activeViewport.m_Height - mousePos.y;
558
559 // Extract mouse picking information from the active framebuffer
560 int pixelData = m_ViewportFramebuffer->ReadPixel(1, (int)mousePos.x, (int)mousePos.y);
561 // Extract lower 16 bits
562 m_HoveredWidgetID = (uint16_t)(pixelData & 0xFFFF);
563 // Extract upper 16 bits
564 m_HoveredWindowID = (uint16_t)((pixelData >> 16) & 0xFFFF);
565
566 // Exit early if no valid widget/window is available
567 if (m_HoveredWidgetID == RuntimeUI::k_InvalidWidgetID || m_HoveredWindowID == RuntimeUI::k_InvalidWindowID)
568 {
569 RuntimeUI::RuntimeUIService::ClearHoveredWidget();
570 return;
571 }
572
573 // Select the widget if applicable
574 RuntimeUI::RuntimeUIService::SetHoveredWidgetByIndex({ RuntimeUI::RuntimeUIService::GetActiveUIHandle(),
575 m_HoveredWindowID, m_HoveredWidgetID });
576
577 }
