216 {
217 Ref<Scenes::Scene> activeScene = Scenes::SceneService::GetActiveScene();
218 UUID entityOneID = event.GetEntityOne();

220 UUID entityTwoID = event.GetEntityTwo();
221 ECS::Entity entityTwo = activeScene->GetEntityByUUID(entityTwoID);
222
223 KG_ASSERT(entityOne);
224 KG_ASSERT(entityTwo);
225
226 bool collisionHandled = false;
227 if (entityOne.HasComponent<ECS::Rigidbody2DComponent>())

230 Assets::AssetHandle scriptHandle = component.OnCollisionStartScriptHandle;
231 Scripting::Script* script = component.OnCollisionStartScript.get();
232 if (scriptHandle != Assets::EmptyHandle)
233 {
234 collisionHandled = Utility::CallWrappedBoolEntityEntity(script->m_Function, entityOneID, entityTwoID);
235 }
236 }
237
238 if (!collisionHandled && entityTwo.HasComponent<ECS::Rigidbody2DComponent>())
239 {
240 ECS::Rigidbody2DComponent& component = entityTwo.GetComponent<ECS::Rigidbody2DComponent>();
241 Assets::AssetHandle scriptHandle = component.OnCollisionStartScriptHandle;

243 if (scriptHandle != Assets::EmptyHandle)
