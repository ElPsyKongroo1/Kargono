16 void AssetManager::DeserializeScriptRegistry()
17 {
18 // Clear current registry and open registry in current project
19 s_ScriptRegistry.clear();
20 s_Scripts.clear();
21 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project to serialize from!");
22 const auto& ScriptRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Scripting/ScriptRegistry.kgreg";
23
24 // Load in Engine Scripts
25 for (auto script : Scripting::ScriptService::GetAllEngineScripts())

27 Assets::Asset newAsset{};

126 void AssetManager::SerializeScriptRegistry()
127 {
128 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project to serialize to!");
129 const auto& ScriptRegistryLocation = Projects::ProjectService::GetActiveAssetDirectory() / "Scripting/ScriptRegistry.kgreg";
130 YAML::Emitter out;
131
132 out << YAML::BeginMap;
133 out << YAML::Key << "Registry" << YAML::Value << "Script";
134
135 // Section Labels
136 out << YAML::Key << "SectionLabels" << YAML::Value;
137 out << YAML::BeginSeq; // Start SectionLabels
138
139 for (auto& section : s_ScriptSectionLabels)
140 {
141 out << YAML::Value << section; // Section Name
142 }
143
144 out << YAML::EndSeq; // End SectionLabels
145
146 // Assets
147 out << YAML::Key << "Assets" << YAML::Value << YAML::BeginSeq;
148 for (auto& [handle, asset] : s_ScriptRegistry)
149 {
150 if (asset.Data.GetSpecificFileData<ScriptMetaData>()->ScriptType == Scripting::ScriptType::Engine)
151 {
152 continue;
153 }
154 out << YAML::BeginMap; // Asset Map
155 out << YAML::Key << "AssetHandle" << YAML::Value << static_cast<uint64_t>(handle);
156
157 out << YAML::Key << "MetaData" << YAML::Value;
158 out << YAML::BeginMap; // MetaData Map
159 out << YAML::Key << "CheckSum" << YAML::Value << asset.Data.CheckSum;
160 out << YAML::Key << "IntermediateLocation" << YAML::Value << asset.Data.IntermediateLocation.string();
161 out << YAML::Key << "AssetType" << YAML::Value << Utility::AssetTypeToString(asset.Data.Type);
162
163 if (asset.Data.Type == Assets::AssetType::Script)
164 {
165 Assets::ScriptMetaData* metadata = static_cast<Assets::ScriptMetaData*>(asset.Data.SpecificFileData.get());
166
167 out << YAML::Key << "Name" << YAML::Value << metadata->Name;
168 out << YAML::Key << "SectionLabel" << YAML::Value << metadata->SectionLabel;
169 out << YAML::Key << "ScriptType" << YAML::Value << Utility::ScriptTypeToString(metadata->ScriptType);
170 out << YAML::Key << "FunctionType" << YAML::Value << Utility::WrappedFuncTypeToString(metadata->FunctionType);
171 }
172
173 out << YAML::EndMap; // MetaData Map
174 out << YAML::EndMap; // Asset Map
175 }
176 out << YAML::EndSeq;
177 out << YAML::EndMap;
178
179 Utility::FileSystem::CreateNewDirectory(ScriptRegistryLocation.parent_path());
180
181 std::ofstream fout(ScriptRegistryLocation);
182 fout << out.c_str();
183 }

185 std::tuple<AssetHandle, bool> AssetManager::CreateNewScript(ScriptSpec& spec)
186 {
187 // Ensure all scripts have unique names
188 for (auto& [handle, asset] : s_ScriptRegistry)
189 {
190 Assets::ScriptMetaData metadata = *static_cast<Assets::ScriptMetaData*>(asset.Data.SpecificFileData.get());
191 if (metadata.Name == spec.Name)
192 {
193 KG_WARN("Unable to create new script. Script Name already exists in asset manager");
194 return std::make_tuple(0, false);
195 }
196 }
197
198 // Check if function type is valid
199 if (spec.FunctionType == WrappedFuncType::None)
200 {
201 KG_WARN("Unable to create new script. Invalid Function Type Provided!");
202 return std::make_tuple(0, false);
203 }
204
205 // If script is associated with a class, ensure class exists
206 Assets::AssetHandle classHandle {0};
207 if (spec.Type == Scripting::ScriptType::Class)
208 {
209 if (spec.SectionLabel == "None")
210 {
211 KG_WARN("Unable to create new script. Empty Section label in spec.");
212 return std::make_tuple(0, false);
213 }
214
215 bool isValid = false;
216 for (auto& [handle, asset] : s_EntityClassRegistry)
217 {
218 if (asset.Data.GetSpecificFileData<Assets::EntityClassMetaData>()->Name == spec.SectionLabel)
219 {
220 isValid = true;
221 classHandle = handle;
222 break;
223 }
224 }
225 if (!isValid)
226 {
227 KG_WARN("Unable to create new script. No Valid Entity Class Selected!");
228 return std::make_tuple(0, false);
229 }
230 }
231
232 // Create Checksum
233 const std::string currentCheckSum {};
234
235 // Create New Asset/Handle
236 AssetHandle newHandle{};
237 Assets::Asset newAsset{};
238 newAsset.Handle = newHandle;
239
240 // If script is associated with a class, add script to class's scripts
241 if (spec.Type == Scripting::ScriptType::Class)
242 {
243 Ref<Scenes::EntityClass> entityClass = GetEntityClass(classHandle);
244 if (!entityClass)
245 {
246 KG_WARN("Unable to create new script. Could not obtain valid entity class pointer!");
247 return std::make_tuple(0, false);
248 }
249
250 entityClass->m_Scripts.AllClassScripts.insert(newHandle);
251 SaveEntityClass(classHandle, entityClass);
252 }
253
254 // Create Intermediate
255 FillScriptMetadata(spec, newAsset);
256 newAsset.Data.CheckSum = currentCheckSum;
257
258 // Register New Asset and Create Script
259 s_ScriptRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
260 SerializeScriptRegistry(); // Update Registry File on Disk
261
262 s_Scripts.insert({ newHandle, InstantiateScriptIntoMemory(newAsset) });
263
264 return std::make_tuple(newHandle, true);
265 }

267 bool AssetManager::UpdateScript(AssetHandle scriptHandle, ScriptSpec& spec)
268 {
269 // Get original asset/metadata
270 Asset asset = s_ScriptRegistry.at(scriptHandle);
271 ScriptMetaData* metadata = asset.Data.GetSpecificFileData<ScriptMetaData>();
272
273 // Check if script exists in registry
274 if (!s_ScriptRegistry.contains(scriptHandle))
275 {
276 KG_WARN("Unable to update script. Does not exist in registry.");
277 return false;
278 }
279
280 // Check if an original entity class needs to be updated
281 AssetHandle classHandle{ 0 };
282 Ref<Scenes::EntityClass> entityClass {nullptr};
283 if (spec.Type == Scripting::ScriptType::Class)
284 {
285 for (auto& [handle, asset] : s_EntityClassRegistry)
286 {
287 if (asset.Data.GetSpecificFileData<Assets::EntityClassMetaData>()->Name == spec.SectionLabel)
288 {
289 classHandle = handle;
290 break;
291 }
292 }
293 // Cancel if entity class pointer cannot be reached
294 entityClass = GetEntityClass(classHandle);
295 if (!entityClass)
296 {
297 KG_WARN("Unable to create new script. Could not obtain valid entity class pointer!");
298 return false;
299 }
300 }
301
302 // Check if function type needs to be updated
303 std::string scriptFile {};
304 std::string matchingExpression {};
305 if (metadata->FunctionType != spec.FunctionType)
306 {
307 // Load file into scriptFile
308 scriptFile = Utility::FileSystem::ReadFileString(Projects::ProjectService::GetActiveAssetDirectory() / asset.Data.IntermediateLocation);
309 if (scriptFile.empty())
310 {
311 KG_WARN("Attempt to open script file failed");
312 return false;
313 }
314
315 // Build Regular Expression
316 std::string returnType {Utility::WrappedVarTypeToCPPString(Utility::WrappedFuncTypeToReturnType(metadata->FunctionType))};
317 std::string functionName { metadata->Name };
318 std::stringstream parameters {};
319 for (auto parameter : Utility::WrappedFuncTypeToParameterTypes(metadata->FunctionType))
320 {
321 std::string parameterString { Utility::WrappedVarTypeToCPPString(parameter) };
322 std::string parameterRegex { std::string("\\s*") + parameterString + std::string("\\s+") +
323 std::string("\\w+") + std::string("\\s*") + std::string(",?") };
324 parameters << parameterRegex;
325 }
326 matchingExpression = "^" + std::string("\\s*") + returnType +
327 std::string("\\s+") + functionName +
328 std::string("\\s*") + "\\(" + parameters.str() +
329 "\\)";
330
331 // Check if correct number of signatures were found
332 uint64_t count = Utility::Regex::GetMatchCount(scriptFile, matchingExpression);
333 if (count == 0)
334 {
335 KG_ERROR("Unable to locate function signature when replacing function type in script manager");
336 return false;
337 }
338 if (count > 1)
339 {
340 KG_ERROR("Too many matches for function signature when replacing function type in script manager");
341 return false;
342 }
343 }
344
345 // Update AllScripts inside original Entity Class if needed
346 if (metadata->ScriptType == Scripting::ScriptType::Class)
347 {
348 // Erase Script inside original entity class
349 for (auto& [classHandle, asset] : s_EntityClassRegistry)
350 {
351 Ref<Scenes::EntityClass> entityClass = GetEntityClass(classHandle);
352 if (!entityClass)
353 {
354 continue;
355 }
356 if (entityClass->GetScripts().AllClassScripts.contains(scriptHandle))
357 {
358 entityClass->GetScripts().AllClassScripts.erase(scriptHandle);
359 SaveEntityClass(classHandle, entityClass);
360 }
361 }
362 }
363
364 // Add script to new class if needed
365 if (spec.Type == Scripting::ScriptType::Class)
366 {
367 entityClass->m_Scripts.AllClassScripts.insert(scriptHandle);
368 SaveEntityClass(classHandle, entityClass);
369 }
370
371 // Update Function Signature if needed
372 if (metadata->FunctionType != spec.FunctionType)
373 {
374 // Replace with new signature
375 std::string output = Utility::Regex::ReplaceMatches(scriptFile, matchingExpression,
376 Utility::GenerateFunctionSignature(spec.FunctionType, spec.Name));
377
378 // Write back out to file
379 Utility::FileSystem::WriteFileString(Projects::ProjectService::GetActiveAssetDirectory() / asset.Data.IntermediateLocation, output);
380 }
381
382 // Update registry metadata
383 metadata->Name = spec.Name;
384 metadata->ScriptType = spec.Type;
385 metadata->SectionLabel = spec.SectionLabel;
386 metadata->FunctionType = spec.FunctionType;
387
388 SerializeScriptRegistry();
389
390 // Update In-Memory Script
391 if (s_Scripts.contains(scriptHandle))
392 {
393 Ref<Scripting::Script> script = s_Scripts.at(scriptHandle);
394 script->m_ScriptName = spec.Name;
395 script->m_FuncType = spec.FunctionType;
396 script->m_ScriptType = spec.Type;
397 script->m_SectionLabel = spec.SectionLabel;
398 script->m_Function = nullptr;
399 }
400
401 return true;
402 }
