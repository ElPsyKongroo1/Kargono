303 {

323 PrintToken(state.Name, indentation + 2);
324 }
325 else if constexpr (std::is_same_v<type, StatementAssignment>)
326 {
327 KG_INFO("{}Assignment Statement", GetIndentation(indentation));
328 KG_INFO("{}Assignemnt Declared Type", GetIndentation(indentation + 1));
329 PrintToken(state.Type, indentation + 2);
330 KG_INFO("{}Declared Name/Identifier", GetIndentation(indentation + 1));
331 PrintToken(state.Name, indentation + 2);

380 std::tuple<bool, Statement> TokenParser::ParseStatementNode()
381 {
382 Statement newStatementEmpty{};
383
384 ScriptToken tokenBuffer = GetCurrentToken();
385 if (tokenBuffer.Type == ScriptTokenType::Semicolon)
386 {
387 newStatementEmpty.emplace<StatementEmpty>();
388 Advance();
389 return { true, newStatementEmpty };
390 }
391
392 if ((tokenBuffer.Type == ScriptTokenType::StringLiteral || tokenBuffer.Type == ScriptTokenType::IntegerLiteral)
393 && GetCurrentToken(1).Type == ScriptTokenType::Semicolon)
394 {
395 StatementLiteral newStatementLiteral{ tokenBuffer };
396 newStatementEmpty.emplace<StatementLiteral>(newStatementLiteral);
397 Advance(2);
398 return { true, newStatementEmpty };
399 }
400
401 if (tokenBuffer.Type == ScriptTokenType::PrimitiveType && GetCurrentToken(1).Type == ScriptTokenType::Identifier
402 && GetCurrentToken(2).Type == ScriptTokenType::Semicolon)
403 {
404 StatementDeclaration newStatementDeclaration{ tokenBuffer, GetCurrentToken(1)};
405 newStatementEmpty.emplace<StatementDeclaration>(newStatementDeclaration);
406 Advance(3);
407 return { true, newStatementEmpty };
408 }
409
410 if (tokenBuffer.Type == ScriptTokenType::PrimitiveType && GetCurrentToken(1).Type == ScriptTokenType::Identifier
411 && GetCurrentToken(2).Type == ScriptTokenType::AssignmentOperator
412 && (GetCurrentToken(3).Type == ScriptTokenType::StringLiteral || GetCurrentToken(3).Type == ScriptTokenType::IntegerLiteral)
413 && GetCurrentToken(4).Type == ScriptTokenType::Semicolon)
414 {
415 std::vector<ScriptTokenType> acceptableTypes = Utility::PrimitiveTypeAcceptableInput(GetCurrentToken().Value);
416 bool success = false;
417 for (ScriptTokenType type : acceptableTypes)
418 {
419 if (type == GetCurrentToken(3).Type)
420 {
421 success = true;
422 }
423 }
424 if (!success)
425 {
426 StoreParseError(ParseErrorType::StateValue, "Invalid assignment statement. Value cannot be assigned to provided type");
427 return { false, newStatementEmpty };
428 }
429
430 StatementAssignment newStatementAssignment{ tokenBuffer, GetCurrentToken(1), GetCurrentToken(3) };
431 newStatementEmpty.emplace<StatementAssignment>(newStatementAssignment);
432 Advance(5);
433 return { true, newStatementEmpty };
434 }
435
436
437
438 return { false, newStatementEmpty };
439 }

441 std::tuple<bool, FunctionNode> TokenParser::ParseFunctionNode()
442 {
443 FunctionNode newFunctionNode{};
444 // Store return value
445 ScriptToken tokenBuffer = GetCurrentToken();
446 if (tokenBuffer.Type != ScriptTokenType::Keyword || tokenBuffer.Value != "void")
447 {
448 StoreParseError(ParseErrorType::FuncReturn, "Expecting a return type for function signature");
449 return { false, newFunctionNode };
450 }
451 newFunctionNode.ReturnType = tokenBuffer;
452
453 // Get Function Name
454 Advance();
455 tokenBuffer = GetCurrentToken();
456 if (tokenBuffer.Type != ScriptTokenType::Identifier)
457 {
458 StoreParseError(ParseErrorType::FuncName, "Expecting a function name in function signature");
459 return { false, newFunctionNode };
460 }
461 newFunctionNode.Name = { tokenBuffer };
462
463 // Parameter Open Parentheses
464 Advance();
465 tokenBuffer = GetCurrentToken();
466 if (tokenBuffer.Type != ScriptTokenType::OpenParentheses)
467 {
468 StoreParseError(ParseErrorType::FuncPunc, "Expecting an open parentheses in function signature");
469 return { false, newFunctionNode };
470 }
471
472 // Parameter List
473 Advance();
474 tokenBuffer = GetCurrentToken();
475 while (tokenBuffer.Type == ScriptTokenType::PrimitiveType)
476 {
477 FunctionParameter newParameter{};
478 // Store type of current parameter
479 newParameter.ParameterType = tokenBuffer;
480
481 // Check and store parameter name
482 Advance();
483 tokenBuffer = GetCurrentToken();
484 if (tokenBuffer.Type != ScriptTokenType::Identifier)
485 {
486 StoreParseError(ParseErrorType::FuncParam, "Expecting an identifier for parameter in function signature");
487 return { false, newFunctionNode };
488 }
489 newParameter.ParameterName = tokenBuffer;
490 newFunctionNode.Parameters.push_back(newParameter);
491
492
493 // Check for comma
494 Advance();
495 tokenBuffer = GetCurrentToken();
496 if (tokenBuffer.Type == ScriptTokenType::Comma)
497 {
498 if (GetCurrentToken(1).Type != ScriptTokenType::Identifier)
499 {
500 StoreParseError(ParseErrorType::FuncParam,
501 "Expecting an identifier after comma separator for parameter list in function signature");
502 return { false, newFunctionNode };
503 }
504
505 // Move to next token if comma is present
506 Advance();
507 }
508 }
509 // Check for close parentheses
510 tokenBuffer = GetCurrentToken();
511 if (tokenBuffer.Type != ScriptTokenType::CloseParentheses)
512 {
513 StoreParseError(ParseErrorType::FuncPunc, "Expecting an closing parentheses in function signature");
514 return { false, newFunctionNode };
515 }
516
517 // Check for open curly braces
518 Advance();
519 tokenBuffer = GetCurrentToken();
520 if (tokenBuffer.Type != ScriptTokenType::OpenCurlyBrace)
521 {
522 StoreParseError(ParseErrorType::FuncPunc, "Expecting an opening curly brace in function signature");
523 return { false, newFunctionNode };
524 }
525
526 Advance();
527 bool success = false;
528 Statement statement;
529 do
530 {
531 std::tie(success, statement) = ParseStatementNode();
532 if (success)
533 {
534 newFunctionNode.Statements.push_back(statement);
535 }
536 } while (success);
537
538 if (CheckForErrors())
539 {
540 StoreParseError(ParseErrorType::FuncBody, "Invalid function statement");
541 return { false, newFunctionNode };
542 }
543
544 // Check for close curly braces
545 tokenBuffer = GetCurrentToken();
546 if (tokenBuffer.Type != ScriptTokenType::CloseCurlyBrace)
547 {
548 StoreParseError(ParseErrorType::FuncPunc, "Expecting an closing curly brace in function signature");
549 return { false, newFunctionNode };
550 }
551
552 return { true, newFunctionNode };
553 }
