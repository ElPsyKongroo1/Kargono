122 bool UIEditorPanel::OnAssetEvent(Events::Event* event)
123 {
124 Events::ManageAsset* manageAsset = (Events::ManageAsset*)event;
125
126 // Manage script deletion event
127 if (manageAsset->GetAssetType() == Assets::AssetType::Script &&
128 manageAsset->GetAction() == Events::ManageAssetAction::Delete)
129 {
130 if (m_WidgetOnPress.CurrentOption.Handle == manageAsset->GetAssetID())
131 {
132 m_WidgetOnPress.CurrentOption = { "None", Assets::EmptyHandle };
133 }
134
135 if (m_EditorUI)
136 {
137 Assets::AssetService::RemoveScriptFromUserInterface(m_EditorUI, manageAsset->GetAssetID());
138 }
139 }
140
141 if (manageAsset->GetAssetType() == Assets::AssetType::UserInterface &&
142 manageAsset->GetAction() == Events::ManageAssetAction::Delete)
143 {
144 // Check if editor needs modification
145 if (manageAsset->GetAssetID() != m_EditorUIHandle)
146 {
147 return false;
148 }
149
150 // Handle deletion of asset
151 ResetPanelResources();
152 return true;
153 }
154
155 if (manageAsset->GetAssetType() == Assets::AssetType::UserInterface &&
156 manageAsset->GetAction() == Events::ManageAssetAction::UpdateAssetInfo)
157 {
158 // Check if editor needs modification
159 if (manageAsset->GetAssetID() != m_EditorUIHandle)
160 {
161 return false;
162 }
163
164 // Update header
165 m_MainHeader.Label = Assets::AssetService::GetUserInterfaceFileLocation(manageAsset->GetAssetID()).filename().string();
166 return true;
167 }
168
169 return false;
170 }

172 void UIEditorPanel::OpenAssetInEditor(std::filesystem::path& assetLocation)
173 {
174 // Ensure provided path is within the active asset directory
175 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
176 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
177 {
178 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
179 return;
180 }
181
182 // Look for asset in registry using the file location
183 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
184 Assets::AssetHandle assetHandle = Assets::AssetService::GetUserInterfaceHandleFromFileLocation(relativePath);
185
186 // Validate resulting handle
187 if (!assetHandle)
188 {
189 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
190 return;
191 }
192
193 // Open the editor panel to be visible
194 s_EditorApp->m_ShowUserInterfaceEditor = true;
195 EditorUI::EditorUIService::BringWindowToFront(m_PanelName);
196 EditorUI::EditorUIService::SetFocusedWindow(m_PanelName);
197
198 // Early out if asset is already open
199 if (m_EditorUIHandle == assetHandle)
200 {
201 return;
202 }
203
204 // Check if panel is already occupied by an asset
205 if (!m_EditorUI)
206 {
207 OnOpenUI(assetHandle);
208 }
209 else
210 {
211 // Add warning to close active AI state before opening a new AIState
212 s_EditorApp->OpenWarningMessage("An user interface is already active inside the editor. Please close the current user interface before opening a new one.");
213 }
