77 void ParticleService::OnUpdate(Timestep ts)
78 {
79 KG_ASSERT(s_ParticleContext);
80
81 // Get current time
82 float currentTime{ EngineService::GetActiveEngine().GetInApplicationTime() };
83
84 size_t iteration{ 0 };
85 for (auto& [uuid, emitter] : s_ParticleContext->m_AllEmitters)
86 {
87 // TODO: If a parent entity exists, maybe just set to inactive
88
89 // Set emitter location if a parent entity exists
90 if (emitter.m_ParentScene)
91 {
92 ECS::Entity entity = emitter.m_ParentScene->GetEntityByUUID(emitter.m_ParentEntityID);
93 ECS::TransformComponent entityTransform = entity.GetComponent<ECS::TransformComponent>();
94 emitter.m_Position = entityTransform.Translation;
95 }
96
97
98 // Check if emitter should be removed
99 if (emitter.m_Config->m_EmitterLifecycle == EmitterLifecycle::FixedTime && currentTime > emitter.m_EndTime)
100 {
101 // List of emitters to be removed
102 static std::vector<UUID> s_EmittersToRemove;
103
104 // Check if a list of emitters already exists
105 if (s_EmittersToRemove.size() > 0)
106 {
107 // Add current emitter to list of emitters to remove
108 s_EmittersToRemove.emplace_back(uuid);
109 }
110 else
111 {
112 // Submit job to main thread that removes indicated emitters from the s_ParticleContext
113 s_EmittersToRemove.emplace_back(uuid);
114 EngineService::SubmitToMainThread([]()
115 {
116 for (UUID id : s_EmittersToRemove)
117 {
118 s_ParticleContext->m_AllEmitters.erase(id);
119 }
120 s_EmittersToRemove.clear();
121 });
122 }
123
124 // Continue on to next emitter
125 continue;
126 }
127
128 // Spawn more particles
129 float spawnThreshold{ 1.0f / (float)emitter.m_Config->m_SpawnRatePerSec };
130 emitter.m_ParticleSpawnAccumulator += ts;
131 while (emitter.m_ParticleSpawnAccumulator > spawnThreshold)
132 {
133 // Decriment accumulator
134 emitter.m_ParticleSpawnAccumulator -= spawnThreshold;
135
136 // Spawn a particle
137 Particle& currentParticle = emitter.m_Particles[emitter.m_ParticleIndex];
138 currentParticle.m_Active = true;
139
140 // Set x,y,z position based on the bounds provide in the emitter's config
141 currentParticle.m_Position.x = emitter.m_Position.x + Utility::PseudoRandomService::GenerateFloatBounds
142 (
143 s_ParticleContext->m_RandomGenerator,
144 emitter.m_Config->m_SpawningBounds[0].x,
145 emitter.m_Config->m_SpawningBounds[1].x
146 );
147 currentParticle.m_Position.y = emitter.m_Position.y + Utility::PseudoRandomService::GenerateFloatBounds
148 (
149 s_ParticleContext->m_RandomGenerator,
150 emitter.m_Config->m_SpawningBounds[0].y,
151 emitter.m_Config->m_SpawningBounds[1].y
152 );
153 currentParticle.m_Position.z = emitter.m_Position.z + Utility::PseudoRandomService::GenerateFloatBounds
154 (
155 s_ParticleContext->m_RandomGenerator,
156 emitter.m_Config->m_SpawningBounds[0].z,
157 emitter.m_Config->m_SpawningBounds[1].z
158 );
159
160 currentParticle.m_StartTime = currentTime;
161 currentParticle.m_EndTime = currentTime + emitter.m_Config->m_ParticleLifetime;
162 currentParticle.m_Size = emitter.m_Config->m_SizeBegin;
163
164 //TODO: Generate random velocity TODO: CHANGE THIS
165 currentParticle.m_Velocity.x = Utility::PseudoRandomService::GenerateFloatBounds(s_ParticleContext->m_RandomGenerator, -1.0f, 1.0f);
166 currentParticle.m_Velocity.y = Utility::PseudoRandomService::GenerateFloatBounds(s_ParticleContext->m_RandomGenerator, -1.0f, 1.0f);
167 currentParticle.m_Velocity.z = 0.0f;
168
169 // Move iterator down
170 emitter.m_ParticleIndex = --emitter.m_ParticleIndex % emitter.m_Particles.size();
171 }
172
173 // Manage particles
174 for (Particles::Particle& particle : emitter.m_Particles)
175 {
176 // Check if particle is active
177 if (!particle.m_Active)
178 {
179 continue;
180 }
181
182 // Check particle lifetime
183 if (currentTime > particle.m_EndTime)
184 {
185 particle.m_Active = false;
186 continue;
187 }
188
189 // Adjust velocity based on gravity if being used
190 if (emitter.m_Config->m_UseGravity)
191 {
192 particle.m_Velocity += emitter.m_Config->m_GravityAcceleration;
193 }
194
195 // Move particle based on velocity
196 particle.m_Position += particle.m_Velocity * (float)ts;
197 }
198
199 iteration++;
200
201 }

205 void ParticleService::OnRender(const Math::mat4& viewProjection)
206 {
207 KG_ASSERT(s_ParticleContext);
208
209 // Start rendering context
210 Rendering::RenderingService::BeginScene(viewProjection);
211
212 // Get current time
213 float currentTime{ EngineService::GetActiveEngine().GetInApplicationTime() };
214
215 for (auto& [uuid, emitter] : s_ParticleContext->m_AllEmitters)
216 {
217 // Get color/size interpolation functions
218 Math::EaseVec3Function sizeEaseFunc
219 {
220 Math::Interpolation::GetEasingFunctionVec3(emitter.m_Config->m_SizeInterpolationType)
221 };
222
223 Math::EaseVec4Function colorEaseFunc
224 {
225 Math::Interpolation::GetEasingFunctionVec4(emitter.m_Config->m_ColorInterpolationType)
226 };
227
228 for (Particles::Particle& particle : emitter.m_Particles)
229 {
230 // Check if particle is active
231 if (!particle.m_Active)
232 {
233 continue;
234 }
235
236 // Calculate lerp progress value
237 float progress{ (currentTime - particle.m_StartTime) / (particle.m_EndTime - particle.m_StartTime) };
238
239 // Calculate size interpolation
240 Math::vec3 size = sizeEaseFunc
241 (
242 emitter.m_Config->m_SizeBegin,
243 emitter.m_Config->m_SizeEnd, progress
244 );
245
246 // Create background rendering data
247 s_ParticleContext->m_ParticleRenderSpec.m_TransformMatrix =
248 glm::translate(Math::mat4(1.0f), particle.m_Position) *
249 glm::scale(glm::mat4(1.0f), { size.x, size.y, size.z });
250
251 // Submit a color based on the current interpolation
252 Rendering::Shader::SetDataAtInputLocation<Math::vec4>
253 (
254 colorEaseFunc
255 (
256 emitter.m_Config->m_ColorBegin,
257 emitter.m_Config->m_ColorEnd,
258 progress
259 ),
260 "a_Color",
261 s_ParticleContext->m_ParticleRenderSpec.m_Buffer,
262 s_ParticleContext->m_ParticleRenderSpec.m_Shader
263 );
264
265 // Render the particle
266 Rendering::RenderingService::SubmitDataToRenderer(s_ParticleContext->m_ParticleRenderSpec);
267 }
268 }
269
270 // End rendering context and submit rendering data to GPU
271 Rendering::RenderingService::EndScene();

273 bool ParticleService::OnSceneEvent(Events::Event* event)
274 {
275 if (event->GetEventType() == Events::EventType::ManageEntity)
276 {
277 Events::ManageEntity* manageEntity = (Events::ManageEntity*)event;
278 if (manageEntity->GetAction() == Events::ManageEntityAction::Delete)
279 {
280 if (Scenes::SceneService::GetActiveScene().get() != manageEntity->GetSceneReference())
281 {
282 KG_WARN("Attempt to remove particle emitters from a scene that is not active");
283 return false;
284 }
285 if (manageEntity->GetEntityID() == k_EmptyUUID)
286 {
287 KG_WARN("Attempt to remove particle emitters using an empty entity handle");
288 return false;
289 }
290
291 // Search for entity to delete
292 UUID emitterToRemove{ k_EmptyUUID };
293 for (auto& [handle, emitter] : s_ParticleContext->m_AllEmitters)
294 {
295 if (emitter.m_ParentEntityID == manageEntity->GetEntityID())
296 {
297 emitterToRemove = emitter.m_ParentEntityID;
298 break;
299 }
300 }
301
302 // Remove indicated entity
303 if (emitterToRemove != k_EmptyUUID)
304 {
305 s_ParticleContext->m_AllEmitters.erase(emitterToRemove);
306 return true;
307 }
308
309 }
310 }
311
312 return false;
