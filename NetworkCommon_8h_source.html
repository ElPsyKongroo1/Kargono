118 // Operator Overload Definitions
119 //==============================
120 template<typename DataType>
121 Message& operator<<(Message& msg, const DataType& data)
122 {
123 // Check that the type of the data being pushed is trivially copyable
124 static_assert(std::is_standard_layout_v<DataType>, "Data is too complex");
125
126 // Cache current size of vector, as this will be the point we insert the data
127 size_t currentBufferSize = msg.Payload.size();
128
129 // Resize the vector to contain the new data being added
130 msg.Payload.resize(msg.Payload.size() + sizeof(DataType));
131
132 // Copy the data over
133 std::memcpy(msg.Payload.data() + currentBufferSize, &data, sizeof(DataType));
134
135 // Update message header size
136 msg.Header.PayloadSize = static_cast<uint32_t>(msg.Payload.size());
137
138 return msg;
139 }
140
141 template<typename DataType>
142 Message& operator>>(Message& msg, const DataType& data)
143 {
144 // Check that the type of the data being pushed is trivially copyable
145 static_assert(std::is_standard_layout<DataType>::value, "Data is too complex");
146
147 // Cache current size of vector, as this will be the point we insert the data
148 size_t bufferSizeAfterRemoval = msg.Payload.size() - sizeof(DataType);
149
150 // Copy the data over
151 std::memcpy((void*)&data, msg.Payload.data() + bufferSizeAfterRemoval, sizeof(DataType));
152
153 // Shrink the vector to remove the read bytes
154 msg.Payload.resize(bufferSizeAfterRemoval);
155
156 // Update message header size
157 msg.Header.PayloadSize = static_cast<uint32_t>(msg.GetEntireMessageSize());
158
159 return msg;
160 }
