34  static BrowserFileType DetermineFileType(const std::filesystem::directory_entry& entry)
35  {
36  if (entry.is_directory()) { return BrowserFileType::Directory; }
37  if (!entry.path().has_extension()) { return BrowserFileType::None; }
38  auto extension = entry.path().extension();
39  if (extension == ".jpg" || extension == ".png") { return BrowserFileType::Image; }
40  if (extension == ".wav" || extension == ".mp3") { return BrowserFileType::Audio; }
41  if (extension == ".kgreg") { return BrowserFileType::Registry; }
42  if (extension == ".kgui") { return BrowserFileType::UserInterface; }
43  if (extension == ".ttf") { return BrowserFileType::Font; }
44  if (extension == ".kgscene") { return BrowserFileType::Scene; }
45  if (extension == ".csproj") { return BrowserFileType::ScriptProject; }
46  if (extension == ".kginput") { return BrowserFileType::Input; }
47  if (extension == ".kgaudio" || extension == ".kgtexture" || extension == ".kgfont" ||
48  extension == ".kgshadervert" || extension == ".kgshaderfrag") {
49  return BrowserFileType::Binary;
50  }
51  return BrowserFileType::None;
52  }
