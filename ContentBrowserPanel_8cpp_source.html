32 static Panels::BrowserFileType DetermineFileType(const std::filesystem::path& entry)
33 {
34 if (std::filesystem::is_directory(entry)) { return Panels::BrowserFileType::Directory; }
35 if (!entry.has_extension()) { return Panels::BrowserFileType::None; }
36 auto extension = entry.extension();
37 if (extension == ".jpg" || extension == ".png") { return Panels::BrowserFileType::Image; }
38 if (extension == ".wav" || extension == ".mp3") { return Panels::BrowserFileType::Audio; }
39 if (extension == ".kgreg") { return Panels::BrowserFileType::Registry; }
40 if (extension == ".kgui") { return Panels::BrowserFileType::UserInterface; }
41 if (extension == ".ttf") { return Panels::BrowserFileType::Font; }
42 if (extension == ".kgscene") { return Panels::BrowserFileType::Scene; }
43 if (extension == ".csproj") { return Panels::BrowserFileType::ScriptProject; }
44 if (extension == ".kginput") { return Panels::BrowserFileType::Input; }
45 if (extension == ".kgaudio" || extension == ".kgtexture" || extension == ".kgfont" ||
46 extension == ".kgshadervert" || extension == ".kgshaderfrag") {
47 return Panels::BrowserFileType::Binary;
48 }
49 return Panels::BrowserFileType::None;
50 }
