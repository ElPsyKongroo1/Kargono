347 {
348 // Validate event type and asset type
349 if (event->GetEventType() != Events::EventType::ManageAsset)
350 {
351 return false;
352 }
353 Events::ManageAsset* manageAsset = (Events::ManageAsset*)event;
354 if (manageAsset->GetAssetType() != Assets::AssetType::GameState)
355 {
356 return false;
357 }
358
359 if (manageAsset->GetAssetID() != m_EditorGameStateHandle)
360 {
361 return false;
362 }
363
364 // Handle deletion of asset
365 if (manageAsset->GetAction() == Events::ManageAssetAction::Delete)
366 {
367 ResetPanelResources();
368 return true;
369 }
370
371 // Handle updating of asset
372 if (manageAsset->GetAction() == Events::ManageAssetAction::UpdateAssetInfo)
373 {
374 // Update game state header if necessary
375 m_MainHeader.Label = Assets::AssetService::GetGameStateFileLocation(manageAsset->GetAssetID()).filename().string();
376
377 return true;
378 }
379 return false;

407 void GameStatePanel::OpenAssetInEditor(std::filesystem::path& assetLocation)
408 {
409 // Ensure provided path is within the active asset directory
410 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
411 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
412 {
413 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
414 return;
415 }
416
417 // Look for asset in registry using the file location
418 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
419 Assets::AssetHandle assetHandle = Assets::AssetService::GetGameStateHandleFromFileLocation(relativePath);
420
421 // Validate resulting handle
422 if (!assetHandle)
423 {
424 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
425 return;
426 }
427
428 // Open the editor panel to be visible
429 s_EditorApp->m_ShowGameStateEditor = true;
430 EditorUI::EditorUIService::BringWindowToFront(m_PanelName);
431 EditorUI::EditorUIService::SetFocusedWindow(m_PanelName);
432
433 // Early out if asset is already open
434 if (m_EditorGameStateHandle == assetHandle)
435 {
436 return;
437 }
438
439 // Check if panel is already occupied by an asset
440 if (!m_EditorGameState)
441 {
442 OnOpenGameState(assetHandle);
443 }
444 else
445 {
446 // Add warning to close active AI state before opening a new AIState
447 s_EditorApp->OpenWarningMessage("An game state is already active inside the editor. Please close the current game state before opening a new one.");
448 }
