561  void ScriptEngine::LoadAssemblyClasses()
562  {
563  s_ScriptData->EntityClasses.clear();
564 
565  const MonoTableInfo* typeDefinitionsTable = mono_image_get_table_info(s_ScriptData->AppAssemblyImage, MONO_TABLE_TYPEDEF);
566  int32_t numTypes = mono_table_info_get_rows(typeDefinitionsTable);
567  MonoClass* entityClass = mono_class_from_name(s_ScriptData->CoreAssemblyImage, "Kargono", "Entity");
568 
569  for (int32_t i = 0; i < numTypes; i++)
570  {
571  uint32_t cols[MONO_TYPEDEF_SIZE];
572  mono_metadata_decode_row(typeDefinitionsTable, i, cols, MONO_TYPEDEF_SIZE);
573 
574  const char* nameSpace = mono_metadata_string_heap(s_ScriptData->AppAssemblyImage, cols[MONO_TYPEDEF_NAMESPACE]);
575  const char* className = mono_metadata_string_heap(s_ScriptData->AppAssemblyImage, cols[MONO_TYPEDEF_NAME]);
576  std::string fullName;
577  if (strlen(nameSpace) != 0) { fullName = fmt::format("{}.{}", nameSpace, className); }
578  else { fullName = className; }
579 
580  MonoClass* monoClass = mono_class_from_name(s_ScriptData->AppAssemblyImage, nameSpace, className);
581 
582  if (monoClass == entityClass) { continue; }
583 
584  if (std::string(className) == "CustomCalls")
585  {
586  s_ScriptData->CustomEngineCalls = CreateRef<ScriptClass>(nameSpace, className, false);
587  s_ScriptData->CustomEngineCallInstance = ScriptClassCustomCallInstance(s_ScriptData->CustomEngineCalls);
588  void* iter = nullptr;
589  MonoMethod* method;
590  KG_WARN("{} class methods:", className);
591  while (method = mono_class_get_methods(monoClass, &iter))
592  {
593  KG_WARN(" {}", mono_method_get_name(method));
594  if (std::string(mono_method_get_name(method)) == ".ctor") { continue; }
595  MonoMethodSignature* methodSignature = mono_method_get_signature(method, s_ScriptData->AppAssemblyImage, 0);
596  ScriptMethod newMethod = ScriptMethod(mono_method_get_name(method), mono_signature_get_param_count(methodSignature), method);
597  s_ScriptData->CustomEngineCalls->AddCustomMethod(newMethod);
598  }
599  }
600 
601  bool isEntity = mono_class_is_subclass_of(monoClass, entityClass, false);
602 
603  if (!isEntity) { continue; }
604 
605  Ref<ScriptClass> scriptClass = CreateRef<ScriptClass>(nameSpace, className);
606  scriptClass->SetConstructor(entityClass);
607  s_ScriptData->EntityClasses[fullName] = scriptClass;
608 
609  void* iter = nullptr;
610  MonoMethod* method;
611  KG_WARN("{} class methods:", className);
612  while (method = mono_class_get_methods(monoClass, &iter))
613  {
614  std::string methodName = mono_method_get_name(method);
615  KG_WARN(" {}", methodName);
616  if (methodName == ".ctor" || methodName == "OnUpdate" || methodName == "OnPhysicsCollision" || methodName == "OnCreate") { continue; }
617  MonoMethodSignature* methodSignature = mono_method_get_signature(method, s_ScriptData->AppAssemblyImage, 0);
618  ScriptMethod newMethod = ScriptMethod(mono_method_get_name(method), mono_signature_get_param_count(methodSignature), method);
619  scriptClass->AddCustomMethod(newMethod);
620  }
621 
622 
623  int fieldCount = mono_class_num_fields(monoClass);
624  KG_WARN("{} has {} fields:", className, fieldCount);
625  void* iterator = nullptr;
626  while (MonoClassField* field = mono_class_get_fields(monoClass, &iterator))
627  {
628  const char* fieldName = mono_field_get_name(field);
629  uint32_t flags = mono_field_get_flags(field);
630  if (flags & FIELD_ATTRIBUTE_PUBLIC)
631  {
632  MonoType* type = mono_field_get_type(field);
633  ScriptFieldType fieldType = Utility::MonoTypeToScriptFieldType(type);
634  KG_WARN(" {} ({})", fieldName, Utility::ScriptFieldTypeToString(fieldType));
635  scriptClass->AddField(fieldName, fieldType, field);
636  }
637  }
638  }
