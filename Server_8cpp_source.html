26  void Server::SessionClock()
27  {
28  using namespace std::chrono_literals;
29 
30  static const std::chrono::nanoseconds constantFrameTime { 1'000 * 1'000 * 1'000 / 60 };
31  std::chrono::nanoseconds accumulator{ 0 };
32  std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
33  std::chrono::time_point<std::chrono::high_resolution_clock> lastCycleTime = currentTime;
34  std::chrono::nanoseconds timeStep{ 0 };
35 
36  while (!m_StopThread)
37  {
38  currentTime = std::chrono::high_resolution_clock::now();
39  timeStep = currentTime - lastCycleTime;
40  lastCycleTime = currentTime;
41  accumulator += timeStep;
42  if (accumulator < constantFrameTime)
43  {
44  continue;
45  }
46  accumulator -= constantFrameTime;
47  m_UpdateCount++;
48  }

86  void Server::OnMessage(Ref<Kargono::Network::ConnectionToClient> client, Kargono::Network::Message& incomingMessage)
87  {
88  switch (static_cast<CustomMsgTypes>(incomingMessage.Header.ID))
89  {
90  case CustomMsgTypes::ServerPing:
91  {
92  KG_INFO("[{}]: Server Ping", client->GetID());
93  client->Send(incomingMessage);
94  break;
95  }
96 
97  case CustomMsgTypes::MessageAll:
98  {
99  KG_INFO("[{}]: Message All", client->GetID());
100  Kargono::Network::Message newMessage;
101  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::ServerMessage);
102  newMessage << client->GetID();
103  MessageAllClients(newMessage, client);
104  break;
105  }
106  case CustomMsgTypes::ClientChat:
107  {
108  KG_INFO("[{}]: Sent Chat", client->GetID());
109  incomingMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::ServerChat);
110  incomingMessage << client->GetID();
111  MessageAllClients(incomingMessage, client);
112  break;
113  }
114  case CustomMsgTypes::RequestJoinSession:
115  {
116  Kargono::Network::Message newMessage;
117 
118  // Deny Client Join if session slots are full
119  if (m_OnlySession.GetClientCount() >= Session::k_MaxClients)
120  {
121  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::DenyJoinSession);
122  client->Send(newMessage);
123  break;
124  }
125 
126  // Add Client to session
127  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::ApproveJoinSession);
128  uint16_t clientSlot = m_OnlySession.AddClient(client);
129  if (clientSlot == 0xFFFF)
130  {
131  return;
132  }
133  newMessage << clientSlot;
134 
135  // Send Approval Message to New Client
136  client->Send(newMessage);
137 
138  // Notify all other session clients that new client has been added
139  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UpdateSessionUserSlot);
140  for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
141  {
142  if (clientID == client->GetID()) { continue; }
143  connection->Send(newMessage);
144  }
145 
146  // Updated new client with all other client data
147  for (auto& [slot, clientID] : m_OnlySession.GetAllSlots())
148  {
149  if (clientID == client->GetID()) { continue; }
150 
151  Kargono::Network::Message otherClientMessage;
152  otherClientMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UpdateSessionUserSlot);
153  otherClientMessage << slot;
154  client->Send(otherClientMessage);
155  }
156 
157  // If enough clients are connected, start the session
158  if (m_OnlySession.GetClientCount() == Session::k_MaxClients)
159  {
160  m_OnlySession.InitSession();
161  }
162 
163  break;
164  }
165 
166  case CustomMsgTypes::RequestUserCount:
167  {
168  KG_INFO("[{}]: User Count Request", client->GetID());
169  Kargono::Network::Message newMessage;
170  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UpdateUserCount);
171  uint32_t numberOfUsers = static_cast<uint32_t>(m_Connections.size());
172  newMessage << numberOfUsers;
173  client->Send(newMessage);
174  break;
175  }
176 
177  case CustomMsgTypes::LeaveCurrentSession:
178  {
179  KG_INFO("[{}]: User Leaving Session", client->GetID());
180  Kargono::Network::Message newMessage;
181  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UserLeftSession);
182  newMessage << m_OnlySession.RemoveClient(client->GetID());
183 
184  // Notify all users in the same session that a client left
185  for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
186  {
187  connection->Send(newMessage);
188  }
189  client->Send(newMessage);
190 
191  break;
192  }
193 
194  case CustomMsgTypes::InitSyncPing:
195  {
196  m_OnlySession.ReceiveSyncPing(client->GetID());
197  break;
198  }
199 
200  case CustomMsgTypes::SessionReadyCheck:
201  {
202  m_OnlySession.ReadyCheck(client->GetID());
203  break;
204  }
205 
206  case CustomMsgTypes::EnableReadyCheck:
207  {
208  m_OnlySession.EnableReadyCheck();
209  break;
210  }
211 
212  case CustomMsgTypes::SendAllEntityLocation:
213  {
214  incomingMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UpdateEntityLocation);
215  // Forward entity location to all other clients
216  for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
217  {
218  if (clientID == client->GetID()) { continue; }
219  MessageClientUDP(connection, incomingMessage);
220  }
221  break;
222  }
223 
224  case CustomMsgTypes::SendAllEntityPhysics:
225  {
226  incomingMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::UpdateEntityPhysics);
227  // Forward entity Physics to all other clients
228  for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
229  {
230  if (clientID == client->GetID()) { continue; }
231  MessageClientUDP(connection, incomingMessage);
232  }
233  break;
234  }
235 
236  case CustomMsgTypes::SignalAll:
237  {
238  incomingMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::ReceiveSignal);
239  // Forward signal to all other session clients
240  for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
241  {
242  if (clientID == client->GetID()) { continue; }
243  connection->Send(incomingMessage);
244  }
245  break;
246  }
247 
248  case CustomMsgTypes::KeepAlive:
249  {
250  Kargono::Network::Message newMessage;
251  newMessage.Header.ID = static_cast<uint32_t>(CustomMsgTypes::KeepAlive);
252  MessageClientUDP(client, newMessage);

254  }

257  {
258  Kargono::Network::Message newMessage;

260  MessageClientUDP(client, newMessage);

262  }
263 
264  default:
