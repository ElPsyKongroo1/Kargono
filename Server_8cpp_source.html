168 Server::Server(uint16_t nPort, bool isLocal)
169 : m_TCPAcceptor(m_NetworkContext.AsioContext, asio::ip::tcp::endpoint(asio::ip::tcp::v4(), nPort))
170 {
171 // Create a resolver
172 asio::ip::tcp::resolver resolver(m_NetworkContext.AsioContext);
173
174 // Query for the local host entry
175 asio::ip::tcp::resolver::query query(asio::ip::host_name(), "");
176
177 // Perform the resolution and get the endpoint iterator
178 asio::ip::tcp::resolver::iterator iter = resolver.resolve(query);
179 while (iter != asio::ip::tcp::resolver::iterator())
180 {
181 if (iter->endpoint().address().is_v4()) { break; }
182 ++iter;
183 }
184 KG_ASSERT(iter != asio::ip::tcp::resolver::iterator(), "Could not get local host ip address!");
185
186 // Local udp socket
187 asio::ip::udp::socket serverUDPSocket {m_NetworkContext.AsioContext};
188 // Create Server UDP socket
189 asio::ip::tcp::endpoint localTCPEndpoint = m_TCPAcceptor.local_endpoint();
190 asio::ip::udp::endpoint localUDPEndpoint;
191 if (isLocal)
192 {
193 localUDPEndpoint = asio::ip::udp::endpoint(asio::ip::udp::v4(), localTCPEndpoint.port());
194 }
195 else
196 {
197 localUDPEndpoint = asio::ip::udp::endpoint(iter->endpoint().address(), localTCPEndpoint.port());
198 }
199
200 serverUDPSocket.open(localUDPEndpoint.protocol());
201 serverUDPSocket.bind(localUDPEndpoint);
202 m_UDPServer = CreateRef<ServerUDPConnection>(&m_NetworkContext, std::move(serverUDPSocket), m_IPAddressToConnection);

247 void Server::OpenMessageFromClient(ServerTCPConnection* client, Kargono::Network::Message& incomingMessage)
248 {
249 switch (static_cast<MessageType>(incomingMessage.Header.ID))
250 {
251 case MessageType::ServerPing:
252 {
253 KG_INFO("[{}]: Server Ping", client->GetID());
254 client->SendTCPMessage(incomingMessage);
255 break;
256 }
257
258 case MessageType::MessageAll:
259 {
260 KG_INFO("[{}]: Message All", client->GetID());
261 Kargono::Network::Message newMessage;
262 newMessage.Header.ID = static_cast<uint32_t>(MessageType::ServerMessage);
263 newMessage << client->GetID();
264 SendTCPMessageAll(newMessage, client);
265 break;
266 }
267 case MessageType::ClientChat:
268 {
269 KG_INFO("[{}]: Sent Chat", client->GetID());
270 incomingMessage.Header.ID = static_cast<uint32_t>(MessageType::ServerChat);
271 incomingMessage << client->GetID();
272 SendTCPMessageAll(incomingMessage, client);
273 break;
274 }
275 case MessageType::RequestJoinSession:
276 {
277 Kargono::Network::Message newMessage;
278
279 // Deny Client Join if session slots are full
280 if (m_OnlySession.GetClientCount() >= k_MaxSessionClients)
281 {
282 newMessage.Header.ID = static_cast<uint32_t>(MessageType::DenyJoinSession);
283 client->SendTCPMessage(newMessage);
284 break;
285 }
286
287 // Add Client to session
288 newMessage.Header.ID = static_cast<uint32_t>(MessageType::ApproveJoinSession);
289 uint16_t clientSlot = m_OnlySession.AddClient(client);
290 if (clientSlot == 0xFFFF)
291 {
292 return;
293 }
294 newMessage << clientSlot;
295
296 // Send Approval Message to New Client
297 client->SendTCPMessage(newMessage);
298
299 // Notify all other session clients that new client has been added
300 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UpdateSessionUserSlot);
301 for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
302 {
303 if (clientID == client->GetID()) { continue; }
304 connection->SendTCPMessage(newMessage);
305 }
306
307 // Updated new client with all other client data
308 for (auto& [slot, clientID] : m_OnlySession.GetAllSlots())
309 {
310 if (clientID == client->GetID()) { continue; }
311
312 Kargono::Network::Message otherClientMessage;
313 otherClientMessage.Header.ID = static_cast<uint32_t>(MessageType::UpdateSessionUserSlot);
314 otherClientMessage << slot;
315 client->SendTCPMessage(otherClientMessage);
316 }
317
318 // If enough clients are connected, start the session
319 if (m_OnlySession.GetClientCount() == k_MaxSessionClients)
320 {
321 m_OnlySession.InitSession();
322 }
323
324 break;
325 }
326
327 case MessageType::RequestUserCount:
328 {
329 KG_INFO("[{}]: User Count Request", client->GetID());
330 Kargono::Network::Message newMessage;
331 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UpdateUserCount);
332 uint32_t numberOfUsers = static_cast<uint32_t>(m_AllClientConnections.size());
333 newMessage << numberOfUsers;
334 client->SendTCPMessage(newMessage);
335 break;
336 }
337
338 case MessageType::LeaveCurrentSession:
339 {
340 KG_INFO("[{}]: User Leaving Session", client->GetID());
341 Kargono::Network::Message newMessage;
342 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UserLeftSession);
343 newMessage << m_OnlySession.RemoveClient(client->GetID());
344
345 // Notify all users in the same session that a client left
346 for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
347 {
348 connection->SendTCPMessage(newMessage);
349 }
350 client->SendTCPMessage(newMessage);
351
352 break;
353 }
354
355 case MessageType::InitSyncPing:
356 {
357 m_OnlySession.ReceiveSyncPing(client->GetID());
358 break;
359 }
360
361 case MessageType::SessionReadyCheck:
362 {
363 m_OnlySession.StoreClientReadyCheck(client->GetID());
364 break;
365 }
366
367 case MessageType::EnableReadyCheck:
368 {
369 m_OnlySession.EnableReadyCheck();
370 break;
371 }
372
373 case MessageType::SendAllEntityLocation:
374 {
375 incomingMessage.Header.ID = static_cast<uint32_t>(MessageType::UpdateEntityLocation);
376 // Forward entity location to all other clients
377 for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
378 {
379 if (clientID == client->GetID()) { continue; }
380 SendUDPMessage(connection, incomingMessage);
381 }
382 break;
383 }
384
385 case MessageType::SendAllEntityPhysics:
386 {
387 incomingMessage.Header.ID = static_cast<uint32_t>(MessageType::UpdateEntityPhysics);
388 // Forward entity Physics to all other clients
389 for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
390 {
391 if (clientID == client->GetID()) { continue; }
392 SendUDPMessage(connection, incomingMessage);
393 }
394 break;
395 }
396
397 case MessageType::SignalAll:
398 {
399 incomingMessage.Header.ID = static_cast<uint32_t>(MessageType::ReceiveSignal);
400 // Forward signal to all other session clients
401 for (auto& [clientID, connection] : m_OnlySession.GetAllClients())
402 {
403 if (clientID == client->GetID()) { continue; }
404 connection->SendTCPMessage(incomingMessage);
405 }
406 break;
407 }
408
409 case MessageType::KeepAlive:
410 {
411 Kargono::Network::Message newMessage;
412 newMessage.Header.ID = static_cast<uint32_t>(MessageType::KeepAlive);
413 SendUDPMessage(client, newMessage);
414 break;
415 }
416
417 case MessageType::UDPInit:
418 {
419 Kargono::Network::Message newMessage;
420 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UDPInit);
421 SendUDPMessage(client, newMessage);
422 break;
423 }
424
425 default:
426 {
427 KG_ERROR("Invalid message type sent to server in Server.cpp");
428 break;
429 }

530 void Server::SessionClock()
531 {
532 using namespace std::chrono_literals;
533
534 static const std::chrono::nanoseconds constantFrameTime { 1'000 * 1'000 * 1'000 / 60 };
535 std::chrono::nanoseconds accumulator{ 0 };
536 std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
537 std::chrono::time_point<std::chrono::high_resolution_clock> lastCycleTime = currentTime;
538 std::chrono::nanoseconds timeStep{ 0 };
539
540 while (!m_StopTimingThread)
541 {
542 currentTime = std::chrono::high_resolution_clock::now();
543 timeStep = currentTime - lastCycleTime;
544 lastCycleTime = currentTime;
545 accumulator += timeStep;
546 if (accumulator < constantFrameTime)
547 {
548 continue;
549 }
550 accumulator -= constantFrameTime;
551 m_UpdateCount++;
552 }
