320 bool ProjectComponentPanel::OnAssetEvent(Events::Event* event)
321 {
322 // Validate event type and asset type
323 if (event->GetEventType() != Events::EventType::ManageAsset)
324 {
325 return false;
326 }
327 Events::ManageAsset* manageAsset = (Events::ManageAsset*)event;
328 if (manageAsset->GetAssetType() != Assets::AssetType::ProjectComponent)
329 {
330 return false;
331 }
332
333 // Handle deletion of asset
334 if (manageAsset->GetAssetID() != m_EditorProjectComponentHandle)
335 {
336 return false;
337 }
338
339 // Handle deletion of asset
340 if (manageAsset->GetAction() == Events::ManageAssetAction::PreDelete)
341 {
342 ResetPanelResources();
343 return true;
344 }
345
346 // Handle updating of asset
347 if (manageAsset->GetAction() == Events::ManageAssetAction::UpdateAssetInfo)
348 {
349 // Update header
350 m_MainHeader.m_Label = Assets::AssetService::GetProjectComponentFileLocation(manageAsset->GetAssetID()).filename().string();
351 return true;
352 }
353 return false;

379 void ProjectComponentPanel::OpenAssetInEditor(std::filesystem::path& assetLocation)
380 {
381 // Ensure provided path is within the active asset directory
382 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
383 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
384 {
385 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
386 return;
387 }
388
389 // Look for asset in registry using the file location
390 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
391 Assets::AssetHandle assetHandle = Assets::AssetService::GetProjectComponentHandleFromFileLocation(relativePath);
392
393 // Validate resulting handle
394 if (!assetHandle)
395 {
396 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
397 return;
398 }
399
400 // Open the editor panel to be visible
401 s_MainWindow->m_ShowProjectComponent = true;
402 EditorUI::EditorUIService::BringWindowToFront(m_PanelName);
403 EditorUI::EditorUIService::SetFocusedWindow(m_PanelName);
404
405 // Early out if asset is already open
406 if (m_EditorProjectComponentHandle == assetHandle)
407 {
408 return;
409 }
410
411 // Check if panel is already occupied by an asset
412 if (!m_EditorProjectComponent)
413 {
414 OnOpenComponent(assetHandle);
415 }
416 else
417 {
418 // Add warning to close active AI state before opening a new AIState
419 s_MainWindow->OpenWarningMessage("An project component is already active inside the editor. Please close the current project component before opening a new one.");
420 }
