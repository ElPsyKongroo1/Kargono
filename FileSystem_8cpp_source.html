203 std::string FileSystem::ChecksumFromFile(const std::filesystem::path& filepath)
204 {
205 // Open the file to be read in binary mode
206 std::ifstream file(filepath, std::ios::binary);
207
208 // Error Checking input stream
209 if (!file)
210 {
211 KG_ERROR("Failed to generate checksum from file");
212 return {};
213 }
214
215 SHA256 sha256stream;
216
217 // Read the file in chunks
218 std::vector<char> buffer(4096);
219 std::streamsize bytesRead = 0;
220
221 while (file.read(buffer.data(), buffer.size()))
222 {
223 bytesRead = file.gcount();
224 // Update the SHA256 context with the buffer content
225 sha256stream.add(buffer.data(), bytesRead);
226 }
227
228 return sha256stream.getHash();
229 }

270 {
