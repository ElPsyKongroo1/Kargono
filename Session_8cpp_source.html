38 void Session::ReceiveSyncPing(uint32_t clientID)
39 {
40 // Record Current TimePoint
41 auto currentTime = std::chrono::high_resolution_clock::now();
42 float durationSeconds = std::chrono::duration_cast<std::chrono::nanoseconds>(currentTime - m_InitCache.RecentTimePoints.at(clientID)).count() * 0.001f * 0.001f * 0.001f;
43 float singleDirectionLatency = durationSeconds / 2.0f;
44
45 // Check if cache is already filled
46 if (m_InitCache.LatencyCacheFilled.at(clientID) == true)
47 {
48 KG_ERROR("Received sync ping when cache has already been filled!");
49 }
50
51 // Insert duration into latency cache
52 m_InitCache.LatencyCache.at(clientID).push_back(singleDirectionLatency);
53
54 // Check if latency cache for current client is full
55 if (m_InitCache.LatencyCache.at(clientID).size() >= k_MaxSyncPings)
56 {
57 m_InitCache.LatencyCacheFilled.insert_or_assign(clientID, true);
58
59 // Check if we are done with ping aggregation
60 for (auto& [clientID, isCacheFilled] : m_InitCache.LatencyCacheFilled)
61 {
62 if (isCacheFilled == false) { return; }
63 }
64
65 // If we have collected all of the pings, move on to next stage
66 CompleteSessionInit();
67 return;
68 }
69
70 // If Cache is not filled yet, send another sync ping
71 Kargono::Network::Message newMessage;
72 newMessage.Header.ID = static_cast<uint32_t>(MessageType::InitSyncPing);
73 m_InitCache.RecentTimePoints.insert_or_assign(clientID, std::chrono::high_resolution_clock::now());
74 m_ConnectedClients.at(clientID)->SendTCPMessage(newMessage);
75 }

77 void Session::CompleteSessionInit()
78 {
79 // Used for ending calculations
80 std::vector<float> allLatencies{};
81
82 for (auto& [clientID, latencies] : m_InitCache.LatencyCache)
83 {
84 // Calculate current mean
85 float meanLatency{ 0 };
86 float latencySum{ 0 };
87 for (auto latency : latencies)
88 {
89 latencySum += latency;
90 }
91
92 meanLatency = latencySum / static_cast<float>(latencies.size());
93
94 // Calculate Standard Deviation
95 float varianceSum{ 0 };
96
97 for (auto latency : latencies)
98 {
99 varianceSum += static_cast<float>(std::pow(latency - meanLatency, 2));
100 }
101
102 float standardDeviation = std::sqrtf(varianceSum / static_cast<float>(latencies.size()));
103
104
105
106 // Remove outliers
107 latencies.erase(std::remove_if(latencies.begin(), latencies.end(), [&](float latency)
108 {
109 // If the distance between the latency and meanLatency is greater than a standard
110 // deviation, remove the element.
111 if (std::fabs(latency - meanLatency) > standardDeviation)
112 {
113 return true;
114 }
115
116 // Keep the latency if it is within one standard deviation of the mean
117 return false;
118 }));
119
120 // Recalculate Mean Latency
121 meanLatency = 0;
122 latencySum = 0;
123 for (auto latency : latencies)
124 {
125 latencySum += latency;
126 }
127 meanLatency = latencySum / static_cast<float>(latencies.size());
128
129 // Store new Latency inside connection member
130 m_ConnectedClients.at(clientID)->SetTCPLatency(meanLatency);
131 allLatencies.push_back(meanLatency);
132 }
133
134 KG_INFO("[SERVER]: Completed Calculating Connection Latencies");
135
136 // Find highest latency
137 auto maxLatencyIterator = std::max_element(allLatencies.begin(), allLatencies.end());
138 KG_ASSERT(maxLatencyIterator != allLatencies.end(), "No largest element was found, probably an implementation error!");
139 float longestLatency = *maxLatencyIterator;
140
141 // Send start message to each client
142 float waitTime{ 0 };
143 Kargono::Network::Message newMessage;
144 newMessage.Header.ID = static_cast<uint32_t>(MessageType::StartSession);
145 for (auto& [clientID, connection] : m_ConnectedClients)
146 {
147 // Calculate wait time
148 waitTime = longestLatency - connection->GetTCPLatency();
149 if (waitTime < 0) { waitTime = 0; }
150
151 // Send Message
152 newMessage << waitTime;
153 m_ConnectedClients.at(clientID)->SendTCPMessage(newMessage);
154 }

159 // Start Thread
