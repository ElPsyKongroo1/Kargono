522  serializer << YAML::Key << "Assets" << YAML::Value << YAML::BeginSeq;
523  for (auto& [handle, asset] : m_AssetRegistry)

526  serializer << YAML::Key << "AssetHandle" << YAML::Value << static_cast<uint64_t>(handle);

529  serializer << YAML::BeginMap; // MetaData Map
530  serializer << YAML::Key << "CheckSum" << YAML::Value << asset.Data.CheckSum;
531  if (m_Flags.test(AssetManagerOptions::HasFileLocation))

533  serializer << YAML::Key << "FileLocation" << YAML::Value << asset.Data.FileLocation.string();
534  }
535  if (m_Flags.test(AssetManagerOptions::HasIntermediateLocation))
536  {
537  serializer << YAML::Key << "IntermediateLocation" << YAML::Value << asset.Data.IntermediateLocation.string();
538  }
539  serializer << YAML::Key << "AssetType" << YAML::Value << Utility::AssetTypeToString(asset.Data.Type);
540 
541  SerializeAssetSpecificMetadata(serializer, asset);
542 
543  serializer << YAML::EndMap; // Close metadata map
544  serializer << YAML::EndMap; // Close asset map
545  }
546  serializer << YAML::EndSeq; // Close asset sequence
547  serializer << YAML::EndMap; // Close registry map
548 
549  Utility::FileSystem::CreateNewDirectory(registryPath.parent_path());
