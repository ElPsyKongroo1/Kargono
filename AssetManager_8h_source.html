372 {
373 KG_ASSERT(m_Flags.test(AssetManagerOptions::HasFileImporting), "Attempt to import an asset for a file type that does not support importing");
374
375 // Ensure provided name is valid

377 {
378 KG_WARN("Empty/invalid name provided to import asset from file function");

382 // Ensure source path is valid

398 }
399
400 // Exit if extension is invalid
401 if (!foundValidExtension)

403 KG_WARN("Invalid file extension ({}) for {} asset type. Valid file extensions for {} include: ");
404 for (auto& extension : m_ValidImportFileExtensions)
405 {
406 KG_WARN(" {}", extension);
407 }
408 return Assets::EmptyHandle;

411 // Validate provided paths
412 bool exportingToBaseAssetDir{ true };

415 exportingToBaseAssetDir = true;

419 exportingToBaseAssetDir = false;

426
427 // Ensure provided path is not indicating a file
428 if (Utility::FileSystem::HasFileExtension(destinationPath))

435 Utility::FileSystem::CreateNewDirectory(destinationPath);

437
438
439 // Create Checksum
440 const std::string currentCheckSum = Utility::FileSystem::ChecksumFromFile(sourcePath);
441
442 // Ensure checksum is valid
443 if (currentCheckSum.empty())
444 {
445 KG_WARN("Generated empty checksum from file at {}", sourcePath.string());
446 return Assets::EmptyHandle;
447 }
448
449 // Ensure duplicate asset is not found in registry.
450 for (const auto& [handle, asset] : m_AssetRegistry)
451 {
452 // Ensure names do not match inside asset registry
453 if (asset.Data.FileLocation.stem().string() == newFileName)

469 newAsset.Data.FileLocation = Utility::FileSystem::ConvertToUnixStylePath(Utility::FileSystem::GetRelativePath(Projects::ProjectService::GetActiveAssetDirectory(), destinationPath / (newFileName + m_FileExtension)));

472

478 newAsset.Data.CheckSum = currentCheckSum;
479 }
480 else
481 {
482 KG_ERROR("Attempt to import a file that does not generate an intermediate. I have not decided what happens in this case.");
483 }
484
485 // TODO: Make sure to modify the code below if fixing the asset above
486 std::filesystem::path assetPath =
487 (m_Flags.test(AssetManagerOptions::HasIntermediateLocation) ?
488 Projects::ProjectService::GetActiveIntermediateDirectory() / newAsset.Data.IntermediateLocation :
489 Projects::ProjectService::GetActiveAssetDirectory() / newAsset.Data.FileLocation);
490
491 // Add new asset into asset registry
492 m_AssetRegistry.insert({ newHandle, newAsset });
493 SerializeAssetRegistry();
494
495 // Fill in-memory cache if appropriate
496 if (m_Flags.test(AssetManagerOptions::HasAssetCache))
497 {

499 }

510 void SerializeAssetRegistry()
511 {
512 // Get registry path
513 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project when attempting to serialize an asset");
514 const std::filesystem::path registryPath = Projects::ProjectService::GetActiveIntermediateDirectory() / m_RegistryLocation;
515
516 // Set up serializer
517 YAML::Emitter serializer;
518 serializer << YAML::BeginMap;
519 serializer << YAML::Key << "Registry" << YAML::Value << m_AssetName;
520
521 // Serialize other registry specific data
522 SerializeRegistrySpecificData(serializer);
523
524 // Asset
525 serializer << YAML::Key << "Assets" << YAML::Value << YAML::BeginSeq;
526 for (auto& [handle, asset] : m_AssetRegistry)
527 {
528 serializer << YAML::BeginMap; // Asset Map
529 serializer << YAML::Key << "AssetHandle" << YAML::Value << static_cast<uint64_t>(handle);
530
531 serializer << YAML::Key << "MetaData" << YAML::Value;
532 serializer << YAML::BeginMap; // MetaData Map
533 serializer << YAML::Key << "CheckSum" << YAML::Value << asset.Data.CheckSum;
534 if (m_Flags.test(AssetManagerOptions::HasFileLocation))
535 {
536 serializer << YAML::Key << "FileLocation" << YAML::Value << asset.Data.FileLocation.string();
537 }
538 if (m_Flags.test(AssetManagerOptions::HasIntermediateLocation))
539 {
540 serializer << YAML::Key << "IntermediateLocation" << YAML::Value << asset.Data.IntermediateLocation.string();
541 }
542 serializer << YAML::Key << "AssetType" << YAML::Value << Utility::AssetTypeToString(asset.Data.Type);
543
544 SerializeAssetSpecificMetadata(serializer, asset);
545
546 serializer << YAML::EndMap; // Close metadata map
547 serializer << YAML::EndMap; // Close asset map
548 }
549 serializer << YAML::EndSeq; // Close asset sequence
550 serializer << YAML::EndMap; // Close registry map
551
552 Utility::FileSystem::CreateNewDirectory(registryPath.parent_path());
553
554 std::ofstream fout(registryPath);
555 fout << serializer.c_str();
556 }

558 void DeserializeAssetRegistry()
559 {
560 // Clear current registry
561 m_AssetRegistry.clear();
562
563 // Get on-disk registry path
564 KG_ASSERT(Projects::ProjectService::GetActive(), "There is no currently loaded project when attempting to serialize an asset");
565 const std::filesystem::path registryPath = Projects::ProjectService::GetActiveIntermediateDirectory() / m_RegistryLocation;
566
567 if (!Utility::FileSystem::PathExists(registryPath))
568 {
569 KG_WARN("No .kgreg file found at provided registry path {}. Creating a new one.", registryPath.string());
570 SerializeAssetRegistry();
571 }
572 YAML::Node data;
573 try
574 {
575 data = YAML::LoadFile(registryPath.string());
576 }
577 catch (YAML::ParserException e)
578 {
579 KG_WARN("Failed to load {} file {}\n {}", m_FileExtension , registryPath.string(), e.what());
580 return;
581 }
582 // Opening registry node
583 if (!data["Registry"])
584 {
585 KG_WARN("Could not validate initial registry node for the file: {}", registryPath.string());
586 return;
587 }
588
589 std::string registryName = data["Registry"].as<std::string>();
590 KG_INFO("Deserializing {} Registry", registryName);
591
592 // Open registry specific data
593 DeserializeRegistrySpecificData(data);
594
595 // Opening all assets
596 auto assets = data["Assets"];
597 if (assets)
598 {
599 for (auto asset : assets)
600 {
601 Assets::AssetInfo newAsset{};
602 newAsset.m_Handle = asset["AssetHandle"].as<uint64_t>();
603
604 // Retrieving metadata for asset
605 auto metadata = asset["MetaData"];
606 newAsset.Data.CheckSum = metadata["CheckSum"].as<std::string>();
607 newAsset.Data.Type = Utility::StringToAssetType(metadata["AssetType"].as<std::string>());
608 if (m_Flags.test(AssetManagerOptions::HasFileLocation))
609 {
610 newAsset.Data.FileLocation = metadata["FileLocation"].as<std::string>();
611 }
612 if (m_Flags.test(AssetManagerOptions::HasIntermediateLocation))
613 {
614 newAsset.Data.IntermediateLocation = metadata["IntermediateLocation"].as<std::string>();
615 }
616
617 // Open registry specific metadata
618 DeserializeAssetSpecificMetadata(metadata, newAsset);
619
620 // Add asset to in memory registry
621 m_AssetRegistry.insert({ newAsset.m_Handle, newAsset });
622
623 }
624 }

629 {
630 // Validate asset exists
631 if (!m_AssetRegistry.contains(handle))
632 {
633 KG_WARN("Could not locate provided asset when attempting to its update file location");
634 return false;
635 }
636
637 // Ensure file path is not absolute
638 if (newFileLocation.is_absolute())
639 {
640 KG_WARN("Attempt to update an asset's file location with a path that is absolute. The filepath should be relative.");
641 return false;
642 }
643
644 // Ensure path provided is a file
645 if (!Utility::FileSystem::HasFileExtension(newFileLocation))
646 {
647 KG_WARN("Attempt to update an asset's file location with a path that does not seem to be a file");
648 return false;
649 }
650
651 // Get asset
652 Assets::AssetInfo& currentAsset = m_AssetRegistry.at(handle);
653
654 std::filesystem::path existingAssetExtension = currentAsset.Data.FileLocation.extension();
655 std::filesystem::path existingAssetName = currentAsset.Data.FileLocation.stem();
656 std::filesystem::path newFileLocationExtension = newFileLocation.extension();
657 std::filesystem::path newFileLocationName = newFileLocation.stem();
658
659 // Ensure extension does not change
660 if (newFileLocationExtension != existingAssetExtension)
661 {
662 KG_WARN("Attempt to update an asset's file location with a path whose extension does not match the asset type");
663 return false;
664 }
665
666 // Ensure file name does not change
667 if (newFileLocationName != existingAssetName)
668 {
669 KG_WARN("Attempt to update an asset's file location with a path whose name does not match the original");
670 return false;
671 }
672
673 // Update file location
674 currentAsset.Data.FileLocation = Utility::FileSystem::ConvertToUnixStylePath(newFileLocation);
675
676 // Save changes to disk
677 SerializeAssetRegistry();
678
679 // Throw update asset event
680 Ref<Events::ManageAsset> event = CreateRef<Events::ManageAsset>
