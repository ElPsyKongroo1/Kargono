1150 // Get Previous Buffer and Previous Shader
1151 Buffer oldBuffer = component.ShaderData;
1152 Ref<Rendering::Shader> oldShader = component.Shader;
1153 // Get New Shader
1154 auto [newShaderAssetHandle, newShader] = Assets::AssetManager::GetShader(component.ShaderSpecification);
1155 // Assign New Shader to Component
1156 component.ShaderHandle = newShaderAssetHandle;
1157 component.Shader = newShader;
1158 // Create New Buffer with New ShaderLayout Size and Set Entire Buffer to Zero
1159 Buffer newBuffer(newShader->GetInputLayout().GetStride() * sizeof(uint8_t));
1160 newBuffer.SetDataToByte(0);
1161
1162 // Transfer Data from Old Buffer to New Buffer if applicable
1163 // This for loop checks if each element in the old shader exists in the new shader.
1164 // If an element does exist in both shaders, the data from the old buffer is
1165 // transferred to the new buffer!
1166 for (const auto& element : oldShader->GetInputLayout().GetElements())
1167 {
1168 if (newShader->GetInputLayout().FindElementByName(element.Name))
1169 {
1170 // Get Location of Old Data Pointer
1171 std::size_t oldLocation = element.Offset;
1172 uint8_t* oldLocationPointer = oldBuffer.As<uint8_t>(oldLocation);
1173
1174 // Get Location of New Data Pointer

1176
1177 // Get Size of Data to Transfer
1178 std::size_t size = element.Size;
1179
1180 // Final Memory Copy
1181 memcpy_s(newLocationPointer, size, oldLocationPointer, size);
1182 }
1183 }
