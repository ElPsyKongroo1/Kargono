11 // Window Resize Event Class
12 //============================================================
13 // This event class represents a Window Resize event that is thrown by
14 // the current windowing system. This event simply contains the new
15 // Width and Height which can be used to resize the viewport among
16 // other use-cases.
17 class WindowResizeEvent : public Event
18 {
19 public:
20 //==============================
21 // Constructors and Destructors
22 //==============================
23 // This event is initialized with the new width and height of the window.
24 WindowResizeEvent(unsigned int width, unsigned int height)
25 : m_Width(width), m_Height(height) {}
26
27 //==============================
28 // Getters/Setters
29 //==============================
30
31 unsigned int GetWidth() const { return m_Width; }
32 unsigned int GetHeight() const { return m_Height; }
33
34 static EventType GetStaticType() { return EventType::WindowResize; }
35 virtual EventType GetEventType() const override { return GetStaticType(); }
36 virtual const char* GetName() const override { return "WindowResizeEvent"; }
37 virtual int GetCategoryFlags() const override { return EventCategory::Application; }
38
39 std::string ToString() const override
40 {
41 std::stringstream ss;
42 ss << "WindowResizeEvent: " << m_Width << ", " << m_Height;
43 return ss.str();
44 }
45 private:
46 // m_Width and m_Height represent the new width and height of the window
47 // after the resize has occurred on the Window.
48 unsigned int m_Width, m_Height;
49 };
