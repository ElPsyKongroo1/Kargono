38 bool TCPClientConnection::Connect(const asio::ip::tcp::resolver::results_type& endpoints)
39 {
40 asio::error_code ec;
41 //asio::connect(m_TCPSocket, endpoints, ec);
42
43 asio::ip::tcp::resolver::iterator iter = endpoints.begin();
44 asio::ip::tcp::resolver::iterator end; // End marker.
45
46 ec.clear();
47 bool connectionSuccessful = false;
48 while (iter != end)
49 {
50 m_TCPSocket.connect(*iter, ec);
51 if (!ec)
52 {
53 connectionSuccessful = true;
54 break;
55 }
56
57 ec.clear();
58 ++iter;
59 }
60 if (!connectionSuccessful) { return false; }
61
62 if (!ec)
63 {
64 ReadValidation();
65 m_UDPLocalEndpoint = asio::ip::udp::endpoint(m_TCPSocket.local_endpoint().address(),
66 m_TCPSocket.local_endpoint().port());
67 m_UDPRemoteSendEndpoint = asio::ip::udp::endpoint(m_TCPSocket.remote_endpoint().address(),
68 m_TCPSocket.remote_endpoint().port());
69 m_UDPRemoteReceiveEndpoint = asio::ip::udp::endpoint(m_TCPSocket.remote_endpoint().address(),
70 m_TCPSocket.remote_endpoint().port());
71 // Connected Successfully
72 return true;
73 }
74
75 // Failed to connect
76 return false;

144 bool Client::Connect(const std::string& host, const uint16_t port, bool remote)
145 {
146 try
147 {
148 // Create TCP Socket for this new connection
149 asio::ip::tcp::socket socket{m_context};
150
151 if (remote)
152 {
153 // Get local TCP address and port
154 asio::ip::tcp::resolver resolver(m_context);
155 asio::ip::tcp::resolver::query query(asio::ip::host_name(), "");
156 asio::ip::tcp::resolver::iterator iter = resolver.resolve(query);
157 while (iter != asio::ip::tcp::resolver::iterator())
158 {
159 if (iter->endpoint().address().is_v4()) { break; }
160 ++iter;
161 }
162
163 // Create local endpoint and bind to new socket
164 KG_ASSERT(port > 100, "Port must be greater than 100. If not, we have some issuessss!");
165 asio::ip::tcp::endpoint localTCP = asio::ip::tcp::endpoint(iter->endpoint().address(), port - 100);
166 socket.open(localTCP.protocol());
167 socket.bind(localTCP);
168 }
169
170 // Resolve provided host/port into a TCP endpoint list
171 asio::ip::tcp::resolver resolver(m_context);
172 asio::ip::tcp::resolver::results_type endpoints = resolver.resolve(host, std::to_string(port));
173
174 // Create a new connection object
175 m_connection = CreateRef<TCPClientConnection>(
176 m_context,
177 std::move(socket),
178 m_qMessagesIn,
179 m_BlockThreadCV,
180 m_BlockThreadMx);
181
182 // Initialize connection to remote endpoint
183 KG_INFO("Attempting to TCP connection to server");
184 if (!m_connection->Connect(endpoints)) { return false; }
185 KG_INFO("Successful TCP connection to server");
186
187 KG_INFO("Attempting UDP connection to server");
188
189 KG_TRACE_INFO("The local endpoint is {} {}", m_connection->GetUDPLocalEndpoint().address().to_string(),
190 m_connection->GetUDPLocalEndpoint().port());
191 KG_TRACE_INFO("The remote send endpoint is {} {}", m_connection->GetUDPRemoteSendEndpoint().address().to_string(),
192 m_connection->GetUDPRemoteSendEndpoint().port());
193
194 // Create Local UDP Socket
195 asio::ip::udp::socket serverUDPSocket {m_context};
196 // Get Local UDP Endpoint created from Connect Function
197 // and bind it to the socket
198 asio::ip::udp::endpoint& localEndpoint = m_connection->GetUDPLocalEndpoint();
199 serverUDPSocket.open(localEndpoint.protocol());
200 serverUDPSocket.bind(localEndpoint);
201
202 m_UDPClientConnection = CreateRef<UDPClientConnection>(m_context, std::move(serverUDPSocket), m_qMessagesIn,
203 m_BlockThreadCV, m_BlockThreadMx, m_connection);
204
205 m_UDPClientConnection->Start();
206
207 // Start Context Thread
208 thrContext = std::thread([this]() { m_context.run(); });
209
210 Utility::AsyncBusyTimer::CreateRecurringTimer(0.3f, 10, [&]()
211 {
212 ClientService::GetActiveClient()->SubmitToFunctionQueue([&]()
213 {
214 KG_TRACE_INFO("Submitted a UDP INIT BITCH");
215 // Do checks for UDP connection
216 Message newMessage{};
217 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UDPInit);
218 SendUDP(newMessage);
219 });
220 WakeUpNetworkThread();
221 }, [&]()
222 {
223 ClientService::GetActiveClient()->SubmitToFunctionQueue([&]()
224 {
225 if (!m_UDPConnectionSuccessful)
226 {
227 Disconnect();
228 return;
229 }
230 KG_INFO("UDP Connection Successful!");
231 });
232 WakeUpNetworkThread();
233 });
234
235
236 }
237 catch (std::exception& e)
238 {
239 KG_ERROR("Client Exception: {}", e.what());
240 return false;
241 }
242
243 return true;
244 }

442 void Client::RunClient()
443 {
444 bool remoteConnection = false;
445
446 if (Projects::ProjectService::GetActiveServerLocation() != "LocalMachine")
447 {
448 remoteConnection = true;
449 }
450
451 if (!Connect(Projects::ProjectService::GetActiveServerIP(), Projects::ProjectService::GetActiveServerPort(), remoteConnection)) { m_Quit = true; }
452
453 if (m_UDPClientConnection && IsConnected())
454 {
455 // Start timer for keep alive packets
456 EngineService::SubmitToEventQueue(CreateRef<Events::AddTickGeneratorUsage>(m_UDPClientConnection->GetKeepAliveDelay()));
457 }
458
459 while (!m_Quit)
460 {
461 if (IsConnected())
462 {
463 ProcessEventQueue();
464 ExecuteFunctionQueue();
465 Update(5);
466 }
467 else
468 {
469 // Client is disconnected
470 KG_WARN("Server Down");
471 m_Quit = true;
472 }
473 }
474 Disconnect();
475
476 if (m_UDPClientConnection)
477 {
478 EngineService::SubmitToEventQueue(CreateRef<Events::RemoveTickGeneratorUsage>(m_UDPClientConnection->GetKeepAliveDelay()));
479 }
480 }

482 void Client::OnMessage(Kargono::Network::Message& msg)
483 {
484 switch (static_cast<MessageType>(msg.Header.ID))
485 {
486 case MessageType::AcceptConnection:
487 {
488 KG_INFO("[SERVER]: Connection has been accepted!");
489 uint32_t userCount{};
490 msg >> userCount;
491 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateOnlineUsers>(userCount));
492 break;
493 }
494
495 case MessageType::UpdateUserCount:
496 {
497 uint32_t userCount{};
498 msg >> userCount;
499 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateOnlineUsers>(userCount));
500 break;
501 }
502
503 case MessageType::ServerChat:
504 {
505 uint32_t clientID{};
506 std::vector<uint8_t> data{};
507 uint64_t dataSize{};
508 msg >> clientID;
509 msg >> dataSize;
510 data = msg.GetBuffer(dataSize);
511 std::string text((char*)(data.data()), dataSize);
512 KG_INFO("[{}]: {}", clientID, text);
513 break;
514 }
515
516 case MessageType::ApproveJoinSession:
517 {
518 KG_INFO("[SERVER]: Approved joining session");
519 uint16_t userSlot{};
520 msg >> userSlot;
521 m_SessionSlot = userSlot;
522 EngineService::SubmitToEventQueue(CreateRef<Events::ApproveJoinSession>(userSlot));
523 break;
524 }
525
526 case MessageType::UpdateSessionUserSlot:
527 {
528 uint16_t userSlot{};
529 msg >> userSlot;
530 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateSessionUserSlot>(userSlot));
531 KG_INFO("[SERVER]: Updated User Slot {}", userSlot);
532 break;
533 }
534
535 case MessageType::UserLeftSession:
536 {
537 KG_INFO("[SERVER]: A User Left the Current Session");
538 uint16_t userSlot{};
539 msg >> userSlot;
540 EngineService::SubmitToEventQueue(CreateRef<Events::UserLeftSession>(userSlot));
541 break;
542 }
543
544 case MessageType::DenyJoinSession:
545 {
546 KG_INFO("[SERVER]: Denied joining session");
547 break;
548 }
549
550 case MessageType::CurrentSessionInit:
551 {
552 KG_INFO("[SERVER]: Active Session is initializing...");
553 EngineService::SubmitToEventQueue(CreateRef<Events::CurrentSessionInit>());
554 break;
555 }
556
557 case MessageType::InitSyncPing:
558 {
559 Kargono::Network::Message newMessage;
560 newMessage.Header.ID = static_cast<uint32_t>(MessageType::InitSyncPing);
561 Send(newMessage);
562 break;
563 }
564
565 case MessageType::StartSession:
566 {
567 float waitTime{};
568 msg >> waitTime;
569 Utility::AsyncBusyTimer::CreateTimer(waitTime, [&]()
570 {
571 // Note Starting Update
572 Network::ClientService::GetActiveClient()->SubmitToEventQueue(CreateRef<Events::StartSession>());
573
574 // Open Gameplay
575 EngineService::SubmitToEventQueue(CreateRef<Events::StartSession>());
576 });
577
578 break;
579 }
580
581 case MessageType::SessionReadyCheckConfirm:
582 {
583 float waitTime{};
584 msg >> waitTime;
585 Utility::AsyncBusyTimer::CreateTimer(waitTime, [&]()
586 {
587 // Open Gameplay
588 EngineService::SubmitToEventQueue(CreateRef<Events::SessionReadyCheckConfirm>());
589 });
590
591 break;
592 }
593
594 case MessageType::UpdateEntityLocation:
595 {
596 uint64_t id;
597 float x, y, z;
598 msg >> z;
599 msg >> y;
600 msg >> x;
601 msg >> id;
602 Math::vec3 trans{x, y, z};
603 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateEntityLocation>(id, trans));
604
605 break;
606 }
607
608 case MessageType::UpdateEntityPhysics:
609 {
610 uint64_t id;
611 float x, y, z, linx, liny;
612
613 msg >> liny;
614 msg >> linx;
615 msg >> z;
616 msg >> y;
617 msg >> x;
618 msg >> id;
619 Math::vec3 trans{x, y, z};
620 Math::vec2 linearV{linx, liny};
621 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateEntityPhysics>(id, trans, linearV));
622
623 break;
624 }
625
626 case MessageType::ReceiveSignal:
627 {
628 uint16_t signal{};
629 msg >> signal;
630 EngineService::SubmitToEventQueue(CreateRef<Events::ReceiveSignal>(signal));
631 break;
632 }
633
634 case MessageType::KeepAlive:
635 {
636 break;
637 }
638
639 case MessageType::UDPInit:
640 {
641 KG_TRACE_INFO("Received UDP Init Packet");
642 m_UDPConnectionSuccessful = true;
643 break;
644 }
645
646 default:
647 {
648 KG_ERROR("Invalid message type sent to client!");
649 break;
650 }
651 }
