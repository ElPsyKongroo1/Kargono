33 bool ClientTCPConnection::Connect(const asio::ip::tcp::resolver::results_type& endpoints)
34 {
35 std::atomic<bool> connectionSuccessful = false;
36
37 // Start async connection
38 asio::async_connect(m_TCPSocket, endpoints, [&](std::error_code ec, const asio::ip::tcp::endpoint& endpoint)
39 {
40 if (ec)
41 {
42 KG_WARN("Error while attempting to connect to server. Error Code: [{}] Message: {}", ec.value(), ec.message());
43 Disconnect();
44 }
45 else
46 {
47 // Start validation sequence
48 ReadValidationAsync();
49 // Wake up network thread
50 connectionSuccessful = true;
51 NetworkThreadWakeUp();
52 }
53 });
54
55 // Start Context Thread
56 m_NetworkContextPtr->AsioThread = std::thread([this]() { m_NetworkContextPtr->AsioContext.run(); });
57
58 while (!m_NetworkContextPtr->Quit && !connectionSuccessful)
59 {
60 NetworkThreadSleep();
61 }
62
63 // Check for failure to connect
64 if (!connectionSuccessful || m_NetworkContextPtr->Quit)
65 {
66 return false;
67 }
68
69 // Initialize UDP endpoints for new UDP connnection
70 m_UDPLocalEndpoint = asio::ip::udp::endpoint(m_TCPSocket.local_endpoint().address(),
71 m_TCPSocket.local_endpoint().port());
72 m_UDPRemoteSendEndpoint = asio::ip::udp::endpoint(m_TCPSocket.remote_endpoint().address(),
73 m_TCPSocket.remote_endpoint().port());
74 m_UDPRemoteReceiveEndpoint = asio::ip::udp::endpoint(m_TCPSocket.remote_endpoint().address(),
75 m_TCPSocket.remote_endpoint().port());
76
77 // TCP Connection successful!
78 return true;
79 }

139 bool Client::ConnectToServer(const std::string& serverIP, const uint16_t serverPort, bool remote)
140 {
141 try
142 {
143 // Create TCP Socket for this new connection
144 asio::ip::tcp::socket localSocket{m_NetworkContext.AsioContext};
145
146 if (remote)
147 {
148 // Get local TCP address and port from localhost
149 asio::ip::tcp::resolver resolver(m_NetworkContext.AsioContext);
150 asio::ip::tcp::resolver::query query(asio::ip::host_name(), "");
151 asio::ip::tcp::resolver::iterator iter = resolver.resolve(query);
152
153 // Search for IPV4 address
154 while (iter != asio::ip::tcp::resolver::iterator())
155 {
156 if (iter->endpoint().address().is_v4())
157 {
158 break;
159 }
160 ++iter;
161 }
162
163 // Create local endpoint, bind IPV4 address to new socket, and assign the local endpoint with the serverPort - 100
164 KG_ASSERT(serverPort > 100, "Port must be greater than 100. If not, we have some issuessss!");
165 asio::ip::tcp::endpoint localTCPEndpoint = asio::ip::tcp::endpoint(iter->endpoint().address(), serverPort - 100);
166 localSocket.open(localTCPEndpoint.protocol());
167 localSocket.bind(localTCPEndpoint);
168 }
169
170 // Resolve provided host/port into a TCP endpoint list
171 asio::ip::tcp::resolver resolver(m_NetworkContext.AsioContext);
172 asio::ip::tcp::resolver::results_type endpoints = resolver.resolve(serverIP, std::to_string(serverPort));
173
174 // Create a new connection object
175 m_ClientTCPConnection = CreateRef<ClientTCPConnection>(&m_NetworkContext, std::move(localSocket));
176
177 // Initialize connection to remote endpoint
178 KG_INFO("Attempting TCP connection to server");
179 if (!m_ClientTCPConnection->Connect(endpoints))
180 {
181 KG_INFO("Failed to initiate TCP connection to server");
182 return false;
183 }
184 KG_INFO("Successful TCP connection to server");
185
186
187 /*KG_TRACE_INFO("The local endpoint is {} {}", m_ClientTCPConnection->GetUDPLocalEndpoint().address().to_string(),
188 m_ClientTCPConnection->GetUDPLocalEndpoint().port());
189 KG_TRACE_INFO("The remote send endpoint is {} {}", m_ClientTCPConnection->GetUDPRemoteSendEndpoint().address().to_string(),
190 m_ClientTCPConnection->GetUDPRemoteSendEndpoint().port());*/
191
192 // Create Local UDP Socket
193 KG_INFO("Attempting UDP connection to server");
194 asio::ip::udp::socket serverUDPSocket {m_NetworkContext.AsioContext};
195 // Get Local UDP Endpoint created from Connect Function
196 // and bind it to the socket
197 asio::ip::udp::endpoint& localEndpoint = m_ClientTCPConnection->GetUDPLocalEndpoint();
198 serverUDPSocket.open(localEndpoint.protocol());
199 serverUDPSocket.bind(localEndpoint);
200
201 m_ClientUDPConnection = CreateRef<ClientUDPConnection>(&m_NetworkContext, std::move(serverUDPSocket), m_ClientTCPConnection);
202
203 m_ClientUDPConnection->Start();
204
205
206
207 Utility::AsyncBusyTimer::CreateRecurringTimer(0.3f, 10, [&]()
208 {
209 ClientService::SubmitToFunctionQueue([&]()
210 {
211 // Do checks for UDP connection
212 Message newMessage{};
213 newMessage.Header.ID = static_cast<uint32_t>(MessageType::UDPInit);
214 SendUDP(newMessage);
215 });
216 NetworkThreadWakeUp();
217 }, [&]()
218 {
219 ClientService::SubmitToFunctionQueue([&]()
220 {
221 if (!m_UDPConnectionSuccessful)
222 {
223 KG_WARN("UDP Connection Unsuccessful!");
224 DisconnectFromServer();
225 return;
226 }
227 KG_INFO("UDP Connection Successful!");
228 });
229 NetworkThreadWakeUp();
230 });
231
232
233 }
234 catch (std::exception& e)
235 {
236 KG_ERROR("Client Exception: {}", e.what());
237 return false;
238 }
239
240 return true;
241 }

270 void Client::CheckForMessages(size_t nMaxMessages)
271 {
272 if (m_NetworkContext.IncomingMessagesQueue.IsEmpty())
273 {
274 NetworkThreadSleep();
275 }
276
277 size_t messageCount = 0;
278 while (messageCount < nMaxMessages && !m_NetworkContext.IncomingMessagesQueue.IsEmpty())
279 {
280 // Grab the front message
281 auto msg = m_NetworkContext.IncomingMessagesQueue.PopFront();
282
283 // Pass to message handler
284 OpenMessageFromServer(msg.msg);
285 messageCount++;

326 void Client::OpenMessageFromServer(Kargono::Network::Message& msg)
327 {
328 // Process message from server based on type
329 switch (static_cast<MessageType>(msg.Header.ID))
330 {
331 case MessageType::AcceptConnection:
332 {
333 KG_INFO("Connection to server has been accepted!");
334 uint32_t userCount{};
335 msg >> userCount;
336 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateOnlineUsers>(userCount));
337 break;
338 }
339
340 case MessageType::UpdateUserCount:
341 {
342 uint32_t userCount{};
343 msg >> userCount;
344 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateOnlineUsers>(userCount));
345 break;
346 }
347
348 case MessageType::ServerChat:
349 {
350 uint32_t clientID{};
351 std::vector<uint8_t> data{};
352 uint64_t dataSize{};
353 msg >> clientID;
354 msg >> dataSize;
355 data = msg.GetPayloadCopy(dataSize);
356 std::string text((char*)(data.data()), dataSize);
357 KG_INFO("[{}]: {}", clientID, text);
358 break;
359 }
360
361 case MessageType::ApproveJoinSession:
362 {
363 KG_INFO("Approved joining session");
364 uint16_t userSlot{};
365 msg >> userSlot;
366 m_SessionSlot = userSlot;
367 EngineService::SubmitToEventQueue(CreateRef<Events::ApproveJoinSession>(userSlot));
368 break;
369 }
370
371 case MessageType::UpdateSessionUserSlot:
372 {
373 uint16_t userSlot{};
374 msg >> userSlot;
375 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateSessionUserSlot>(userSlot));
376 break;
377 }
378
379 case MessageType::UserLeftSession:
380 {
381 KG_INFO("A User Left the Current Session");
382 uint16_t userSlot{};
383 msg >> userSlot;
384 EngineService::SubmitToEventQueue(CreateRef<Events::UserLeftSession>(userSlot));
385 break;
386 }
387
388 case MessageType::DenyJoinSession:
389 {
390 KG_INFO("Denied joining session");
391 break;
392 }
393
394 case MessageType::CurrentSessionInit:
395 {
396 KG_INFO("Active Session is initializing");
397 EngineService::SubmitToEventQueue(CreateRef<Events::CurrentSessionInit>());
398 break;
399 }
400
401 case MessageType::InitSyncPing:
402 {
403 Kargono::Network::Message newMessage;
404 newMessage.Header.ID = static_cast<uint32_t>(MessageType::InitSyncPing);
405 SendTCP(newMessage);
406 break;
407 }
408
409 case MessageType::StartSession:
410 {
411 KG_INFO("Session Started");
412 float waitTime{};
413 msg >> waitTime;
414 Utility::AsyncBusyTimer::CreateTimer(waitTime, [&]()
415 {
416 // Note Starting Update
417 Network::ClientService::SubmitToEventQueue(CreateRef<Events::StartSession>());
418
419 // Open Gameplay
420 EngineService::SubmitToEventQueue(CreateRef<Events::StartSession>());
421 });
422
423 break;
424 }
425
426 case MessageType::SessionReadyCheckConfirm:
427 {
428 float waitTime{};
429 msg >> waitTime;
430 Utility::AsyncBusyTimer::CreateTimer(waitTime, [&]()
431 {
432 // Open Gameplay
433 EngineService::SubmitToEventQueue(CreateRef<Events::SessionReadyCheckConfirm>());
434 });
435
436 break;
437 }
438
439 case MessageType::UpdateEntityLocation:
440 {
441 uint64_t id;
442 float x, y, z;
443 msg >> z;
444 msg >> y;
445 msg >> x;
446 msg >> id;
447 Math::vec3 trans{x, y, z};
448 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateEntityLocation>(id, trans));
449
450 break;
451 }
452
453 case MessageType::UpdateEntityPhysics:
454 {
455 uint64_t id;
456 float x, y, z, linx, liny;
457
458 msg >> liny;
459 msg >> linx;
460 msg >> z;
461 msg >> y;
462 msg >> x;
463 msg >> id;
464 Math::vec3 trans{x, y, z};
465 Math::vec2 linearV{linx, liny};
466 EngineService::SubmitToEventQueue(CreateRef<Events::UpdateEntityPhysics>(id, trans, linearV));
467
468 break;
469 }
470
471 case MessageType::ReceiveSignal:
472 {
473 uint16_t signal{};
474 msg >> signal;
475 EngineService::SubmitToEventQueue(CreateRef<Events::ReceiveSignal>(signal));
476 break;
477 }
478
479 case MessageType::KeepAlive:
480 {
481 break;
482 }
483
484 case MessageType::UDPInit:
485 {
486 m_UDPConnectionSuccessful = true;
487 break;
488 }
489
490 default:
491 {
492 KG_ERROR("Invalid message type sent to client!");
493 break;
494 }
495 }
496 }

517 void ClientService::Run()
518 {
519 // Decide whether to connect using the local network or through the internet
520 bool remoteConnection = false;
521 if (Projects::ProjectService::GetActiveServerLocation() != "LocalMachine")
522 {
523 remoteConnection = true;
524 }
525
526 // Start connection to server
527 if (!s_Client->ConnectToServer(Projects::ProjectService::GetActiveServerIP(), Projects::ProjectService::GetActiveServerPort(), remoteConnection))
528 {
529 s_Client->m_NetworkContext.Quit = true;
530 }
531
532 // Ensure UDP connection is valid
533 if (s_Client->m_ClientUDPConnection && s_Client->IsConnected())
534 {
535 // Start timer for keep alive packets
536 EngineService::SubmitToEventQueue(CreateRef<Events::AddTickGeneratorUsage>(k_KeepAliveDelay));
537 }
538
539 KG_INFO("Started client networking thread");
540
541 // Main NetworkContext::NetworkThread loop
542 while (!s_Client->m_NetworkContext.Quit)
543 {
544 if (s_Client->IsConnected())
545 {
546 ClientService::ProcessEventQueue();
547 ClientService::ProcessFunctionQueue();
548 s_Client->CheckForMessages(5);
549 }
550 else
551 {
552 // Client is disconnect
553 s_Client->m_NetworkContext.Quit = true;
554 }
555 }
556
557 // Close connection to server
558 s_Client->DisconnectFromServer();
559 if (s_Client->m_ClientUDPConnection)
560 {
561 EngineService::SubmitToEventQueue(CreateRef<Events::RemoveTickGeneratorUsage>(k_KeepAliveDelay));
562 }
563 KG_INFO("Closed client networking thread");
564 }

567 {
568 // Set run loop to close and alert network thread
569 s_Client->m_NetworkContext.Quit = true;
570 s_Client->NetworkThreadWakeUp();
571 KG_INFO("Notified client network thread to close");
572 }
