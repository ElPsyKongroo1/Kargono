517  void ClientService::Run()
518  {
519  // Decide whether to connect using the local network or through the internet
520  bool remoteConnection = false;
521  if (Projects::ProjectService::GetActiveServerLocation() != "LocalMachine")
522  {
523  remoteConnection = true;
524  }
525 
526  // Start connection to server
527  if (!s_Client->ConnectToServer(Projects::ProjectService::GetActiveServerIP(), Projects::ProjectService::GetActiveServerPort(), remoteConnection))
528  {
529  s_Client->m_NetworkContext.Quit = true;
530  }
531 
532  // Ensure UDP connection is valid
533  if (s_Client->m_ClientUDPConnection && s_Client->IsConnected())
534  {
535  // Start timer for keep alive packets
536  EngineService::SubmitToEventQueue(CreateRef<Events::AddTickGeneratorUsage>(k_KeepAliveDelay));
537  }
538 
539  KG_INFO("Started client networking thread");
540 
541  // Main NetworkContext::NetworkThread loop
542  while (!s_Client->m_NetworkContext.Quit)
543  {
544  if (s_Client->IsConnected())
545  {
546  ClientService::ProcessEventQueue();
547  ClientService::ProcessFunctionQueue();
548  s_Client->CheckForMessages(5);
549  }
550  else
551  {
552  // Client is disconnect
553  s_Client->m_NetworkContext.Quit = true;
554  }
555  }
556 
557  // Close connection to server
558  s_Client->DisconnectFromServer();
559  if (s_Client->m_ClientUDPConnection)
560  {
561  EngineService::SubmitToEventQueue(CreateRef<Events::RemoveTickGeneratorUsage>(k_KeepAliveDelay));
562  }
563  KG_INFO("Closed client networking thread");
