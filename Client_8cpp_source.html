144  void Client::RunClient()
145  {
146  auto currentProject = Projects::Project::GetActive();
147 
148  bool remoteConnection = false;
149 
150  if (currentProject->GetServerLocation() != "LocalMachine")
151  {
152  remoteConnection = true;
153  }
154 
155  if (!Connect(currentProject->GetServerIP(), currentProject->GetServerPort(), remoteConnection)) { m_Quit = true; }
156 
157  if (m_UDPClient && IsConnected())
158  {
159  // Start timer for keep alive packets
160  EngineCore::GetCurrentEngineCore().SubmitToEventQueue(CreateRef<Events::AddTickGeneratorUsage>(m_UDPClient->GetKeepAliveDelay()));
161  }
162 
163  while (!m_Quit)
164  {
165  if (IsConnected())
166  {
167  ProcessEventQueue();
168  ExecuteFunctionQueue();
169  Update(5);
170  }
171  else
172  {
173  // Client is disconnected

175  m_Quit = true;
176  }
177  }
178  Disconnect();
179 
180  if (m_UDPClient)

182  EngineCore::GetCurrentEngineCore().SubmitToEventQueue(CreateRef<Events::RemoveTickGeneratorUsage>(m_UDPClient->GetKeepAliveDelay()));
183  }
