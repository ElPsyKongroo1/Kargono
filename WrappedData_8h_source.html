532  inline bool FillWrappedVarWithStringBuffer(Ref<WrappedVariable> variable, Buffer buffer)
533  {
534  // Func Resources
535  static std::string valueString {};
536 
537  bool success{ false };
538  switch (variable->Type())
539  {
540  case WrappedVarType::Integer32:
541  {
542  success = Conversions::CharBufferToVariable(buffer,
543  variable->GetWrappedValue<int32_t>());
544  break;
545  }
546  case WrappedVarType::UInteger16:
547  {
548  success = Conversions::CharBufferToVariable(buffer,
549  variable->GetWrappedValue<uint16_t>());
550  break;
551  }
552  case WrappedVarType::UInteger32:
553  {
554  success = Conversions::CharBufferToVariable(buffer,
555  variable->GetWrappedValue<uint32_t>());
556  break;
557  }
558  case WrappedVarType::UInteger64:
559  {
560  success = Conversions::CharBufferToVariable(buffer,
561  variable->GetWrappedValue<uint64_t>());
562  break;
563  }
564  case WrappedVarType::Vector3:
565  {
566  // TODO Figure out what happens to the vector
567  success = Conversions::CharBufferToVariable(buffer,
568  variable->GetWrappedValue<Math::vec3>());
569  break;
570  }
571  case WrappedVarType::String:
572  {
573  valueString = buffer.GetString();
574  variable->SetWrappedValue<std::string>(valueString);
575  valueString = {};
576  success = true;
577  break;
578  }
579  case WrappedVarType::Bool:
580  {
581  valueString = buffer.GetString();
582  variable->SetWrappedValue<bool>(Utility::Conversions::StringToBool(valueString));
583  success = true;
584  break;
585  }
586  case WrappedVarType::Float:
587  {
588  success = Conversions::CharBufferToVariable(buffer,
589  variable->GetWrappedValue<float>());
590  break;
591  }
592  case WrappedVarType::Void:
593  case WrappedVarType::None:
594  {
595  KG_WARN("Invalid WrappedVarType provided to FillStringBuffer Function");
596  return false;
597  }
598  }
599  return success;
600  }
