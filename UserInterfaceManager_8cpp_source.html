362 AssetHandle AssetManager::CreateNewUserInterface(const std::string& userInterfaceName)
363 {
364 // Create Checksum
365 const std::string currentCheckSum = Utility::FileSystem::ChecksumFromString(userInterfaceName);
366
367 if (currentCheckSum.empty())
368 {
369 KG_ERROR("Failed to generate checksum from file!");
370 return {};
371 }
372
373 // Compare currentChecksum to registered assets
374 for (const auto& [handle, asset] : s_UserInterfaceRegistry)
375 {
376 if (asset.Data.CheckSum == currentCheckSum)
377 {
378 KG_INFO("Attempt to instantiate duplicate font asset");
379 return handle;
380 }
381 }
382
383 // Create New Asset/Handle
384 AssetHandle newHandle{};
385 Assets::Asset newAsset{};
386 newAsset.Handle = newHandle;
387
388 // Create File
389 CreateUserInterfaceFile(userInterfaceName, newAsset);
390 newAsset.Data.CheckSum = currentCheckSum;
391
392 // Register New Asset and return handle.
393 s_UserInterfaceRegistry.insert({ newHandle, newAsset }); // Update Registry Map in-memory
394 SerializeUserInterfaceRegistry(); // Update Registry File on Disk
395
396 return newHandle;
397 }

434 {
435 KG_ASSERT(Projects::ProjectService::GetActive(), "Attempt to use Project Field without active project!");
436 std::filesystem::path userInterfacePath = filepath;
437
438 if (filepath.is_absolute())
439 {
440 userInterfacePath = Utility::FileSystem::GetRelativePath(Projects::ProjectService::GetActiveAssetDirectory(), filepath);
441 }
442
443 for (auto& [assetHandle, asset] : s_UserInterfaceRegistry)
444 {
445 if (asset.Data.IntermediateLocation.compare(userInterfacePath) == 0)

449 }
450 // Return empty userInterface if userInterface does not exist
