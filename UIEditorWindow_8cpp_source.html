237 void UIEditorWindow::OpenAssetInEditor(std::filesystem::path& assetLocation)
238 {
239 // Ensure provided path is within the active asset directory
240 std::filesystem::path activeAssetDirectory = Projects::ProjectService::GetActiveAssetDirectory();
241 if (!Utility::FileSystem::DoesPathContainSubPath(activeAssetDirectory, assetLocation))
242 {
243 KG_WARN("Could not open asset in editor. Provided path does not exist within active asset directory");
244 return;
245 }
246
247 // Look for asset in registry using the file location
248 std::filesystem::path relativePath{ Utility::FileSystem::GetRelativePath(activeAssetDirectory, assetLocation) };
249 Assets::AssetHandle assetHandle = Assets::AssetService::GetUserInterfaceHandleFromFileLocation(relativePath);
250
251 // Validate resulting handle
252 if (!assetHandle)
253 {
254 KG_WARN("Could not open asset in editor. Provided path does not result in an asset inside the registry.");
255 return;
256 }
257
258 // Early out if asset is already open
259 if (m_EditorUIHandle == assetHandle)
260 {
261 // Open the user interface window in the editor
262 EngineService::SubmitToMainThread([]()
263 {
264 s_EditorApp->SetActiveEditorWindow(ActiveEditorUIWindow::UIEditorWindow);
265 });
266 return;
267 }
268
269 // Check if panel is already occupied by an asset
270 if (!m_EditorUI)
271 {
272 // Open user interface in editor
273 m_TreePanel->OnOpenUI(assetHandle);
274
275 // Open the user interface window in the editor
276 EngineService::SubmitToMainThread([]()
277 {
278 s_EditorApp->SetActiveEditorWindow(ActiveEditorUIWindow::UIEditorWindow);
279 });
280 }
281 else
282 {
283 // Add warning to close active AI state before opening a new AIState
284 s_MainWindow->OpenWarningMessage("An user interface is already active inside the editor. Please close the current user interface before opening a new one.");
285 }
286 }
