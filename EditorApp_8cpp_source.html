655  {
656  // Create/Overwrite scene registry using the new archetype of field data
657  currentScene->ClearProjectComponentRegistry(event.GetAssetID());
658  return;
659  }
660 
661  // Generate and store new entity component data
662  std::unordered_map<UUID, std::vector<uint8_t>> entityToNewDataMap{};
663  for (auto& [entityID, enttID] : currentScene->m_EntityRegistry.m_EntityMap)
664  {
665  ECS::Entity entity = currentScene->GetEntityByEnttID(enttID);
666  if (entity.HasProjectComponentData(event.GetAssetID()))
667  {
668  // Get old data buffer reference and generate new data buffer
669  uint8_t* oldComponentData = (uint8_t*)entity.GetProjectComponentData(event.GetAssetID());
670  std::vector<uint8_t> newComponentData;
671  newComponentData.resize(reallocationInstructions.m_NewDataSize);
672 
673  // Transfer old data into new buffer and initialize indicated data
674  for (size_t iteration{ 0 }; iteration < reallocationInstructions.m_FieldTransferDirections.size(); iteration++)
675  {
676  if (reallocationInstructions.m_FieldTransferDirections.at(iteration) == Assets::k_NewAllocationIndex)
677  {
678  // Initialize data
679  Utility::InitializeDataForWrappedVarBuffer(reallocationInstructions.m_NewDataTypes.at(iteration), newComponentData.data() + reallocationInstructions.m_NewDataLocations.at(iteration));
680  }
681  else
682  {
683  // Transfer data
684  Utility::TransferDataForWrappedVarBuffer(reallocationInstructions.m_NewDataTypes.at(iteration),
685  oldComponentData + reallocationInstructions.m_OldDataLocations.at(reallocationInstructions.m_FieldTransferDirections.at(iteration)),
686  newComponentData.data() + reallocationInstructions.m_NewDataLocations.at(iteration));
687  }
688  }
689 
690  // Store new data into buffer map
691  entityToNewDataMap.insert_or_assign(entityID, newComponentData);
692  }
693  }
694 
695  // Clear old scene registry data for this project component type
696  currentScene->ClearProjectComponentRegistry(event.GetAssetID());
697 
698  // Create/Overwrite scene registry using the new archetype of field data
699  currentScene->AddProjectComponentRegistry(event.GetAssetID());
700 
701  // Write stored entity component data into new registry
702  for (auto& [entityID, dataBuffer] : entityToNewDataMap)
703  {
704  ECS::Entity currentEntity = currentScene->GetEntityByUUID(entityID);
705  if (!currentEntity.HasProjectComponentData(event.GetAssetID()))
706  {
707  currentEntity.AddProjectComponentData(event.GetAssetID());
708  }
709  uint8_t* currentData = (uint8_t*)currentEntity.GetProjectComponentData(event.GetAssetID());
710  memcpy(currentData, dataBuffer.data(), dataBuffer.size()); // Note that the data buffer is a vector of bytes, which means size == byte size
711  }
