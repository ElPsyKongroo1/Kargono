331  bool EditorApp::OnKeyPressedEditor(Events::KeyPressedEvent event)
332  {
333  if (event.IsRepeat()) { return false; }
334 
335  std::string focusedWindow = EditorUI::EditorUIService::GetFocusedWindowName();
336  if (m_PanelToKeyboardInput.contains(focusedWindow))
337  {
338  if (m_PanelToKeyboardInput.at(focusedWindow)(event))
339  {
340  return true;
341  }
342  }
343 
344  bool control = Input::InputPolling::IsKeyPressed(Key::LeftControl) || Input::InputPolling::IsKeyPressed(Key::RightControl);
345  bool shift = Input::InputPolling::IsKeyPressed(Key::LeftShift) || Input::InputPolling::IsKeyPressed(Key::RightShift);
346  bool alt = Input::InputPolling::IsKeyPressed(Key::LeftAlt) || Input::InputPolling::IsKeyPressed(Key::RightAlt);
347 
348  switch (event.GetKeyCode())
349  {
350 
351  case Key::N:
352  {
353  if (control) { NewScene(); }
354  break;
355  }
356  case Key::O:
357  {
358  if (control) { OpenProject(); }
359  break;
360  }
361  case Key::S:
362  {
363  if (control)
364  {
365  SaveScene();
366  SaveProject();
367  }
368  break;
369  }
370  case Key::F11:
371  {
372  EngineCore::GetActiveWindow().ToggleMaximized();
373  break;
374  }
375 
376  // Scene Commands
377 
378  case Key::D:
379  {
380  if (control) { OnDuplicateEntity(); }
381  break;
382  }
383 
384  // Gizmos
385  case Key::R:
386  {
387  if (!ImGuizmo::IsUsing()) { m_ViewportPanel->m_GizmoType = ImGuizmo::OPERATION::SCALE; }
388  break;
389  }
390  case Key::Delete:
391  {
392  if (EditorUI::EditorUIService::GetActiveWidgetID() == 0)
393  {
394  Scenes::Entity selectedEntity = *Scenes::Scene::GetActiveScene()->GetSelectedEntity();
395  if (selectedEntity)
396  {
397  m_EditorScene->DestroyEntity(selectedEntity);
398  m_SceneHierarchyPanel->SetSelectedEntity({});
399  }
400  }
401  break;
402 
403  }
404  default:
405  {
406  break;
407  }
408  }
409  return false;
410  }
