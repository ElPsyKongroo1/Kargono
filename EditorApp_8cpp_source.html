214 bool EditorApp::OnPhysicsCollisionStart(Events::PhysicsCollisionStart event)
215 {
216 Ref<Scenes::Scene> activeScene = Scenes::SceneService::GetActiveScene();
217 UUID entityOneID = event.GetEntityOne();
218 ECS::Entity entityOne = activeScene->GetEntityByUUID(entityOneID);
219 UUID entityTwoID = event.GetEntityTwo();
220 ECS::Entity entityTwo = activeScene->GetEntityByUUID(entityTwoID);
221
222 KG_ASSERT(entityOne);
223 KG_ASSERT(entityTwo);
224
225 bool collisionHandled = false;
226 if (entityOne.HasComponent<ECS::Rigidbody2DComponent>())
227 {
228 ECS::Rigidbody2DComponent& component = entityOne.GetComponent<ECS::Rigidbody2DComponent>();
229 Assets::AssetHandle scriptHandle = component.OnCollisionStartScriptHandle;
230 Scripting::Script* script = component.OnCollisionStartScript.get();
231 if (scriptHandle != Assets::EmptyHandle)
232 {
233 collisionHandled = Utility::CallWrappedBoolEntityEntity(script->m_Function, entityOneID, entityTwoID);
234 }
235 }
236
237 if (!collisionHandled && entityTwo.HasComponent<ECS::Rigidbody2DComponent>())
238 {
239 ECS::Rigidbody2DComponent& component = entityTwo.GetComponent<ECS::Rigidbody2DComponent>();
240 Assets::AssetHandle scriptHandle = component.OnCollisionStartScriptHandle;
241 Scripting::Script* script = component.OnCollisionStartScript.get();
242 if (scriptHandle != Assets::EmptyHandle)
243 {
244 collisionHandled = Utility::CallWrappedBoolEntityEntity(script->m_Function, entityTwoID, entityOneID);
245 }
246 }
247 return false;
248 }

317 {
318 if (Projects::ProjectService::OpenProject(path))

320 if (!EngineService::GetActiveWindow().GetNativeWindow())
321 {
322 EngineService::GetActiveWindow().Init();
