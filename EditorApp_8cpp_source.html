365  bool EditorApp::OnKeyPressedEditor(Events::KeyPressedEvent event)
366  {
367  if (event.IsRepeat()) { return false; }
368 
369  std::string focusedWindow = EditorUI::EditorUIService::GetFocusedWindowName();
370  if (m_PanelToKeyboardInput.contains(focusedWindow))
371  {
372  if (m_PanelToKeyboardInput.at(focusedWindow)(event))
373  {
374  return true;
375  }
376  }
377 
378  bool control = Input::InputService::IsKeyPressed(Key::LeftControl) || Input::InputService::IsKeyPressed(Key::RightControl);
379  bool shift = Input::InputService::IsKeyPressed(Key::LeftShift) || Input::InputService::IsKeyPressed(Key::RightShift);
380  bool alt = Input::InputService::IsKeyPressed(Key::LeftAlt) || Input::InputService::IsKeyPressed(Key::RightAlt);
381 
382  switch (event.GetKeyCode())
383  {
384 
385  case Key::N:
386  {
387  if (control) { NewScene(); }
388  break;
389  }
390  case Key::O:
391  {
392  if (control) { OpenProject(); }
393  break;
394  }
395  case Key::S:
396  {
397  if (control)
398  {
399  SaveScene();
400  SaveProject();
401  }
402  break;
403  }
404  case Key::F11:
405  {
406  EngineService::GetActiveWindow().ToggleMaximized();
407  break;
408  }
409 
410  // Scene Commands
411 
412  case Key::D:
413  {
414  if (control) { OnDuplicateEntity(); }
415  break;
416  }
417 
418  // Gizmos
419  case Key::R:
420  {
421  if (!ImGuizmo::IsUsing()) { m_ViewportPanel->m_GizmoType = ImGuizmo::OPERATION::SCALE; }
422  break;
423  }
424  case Key::Delete:
425  {
426  if (EditorUI::EditorUIService::GetActiveWidgetID() == 0)
427  {
428  Scenes::Entity selectedEntity = *Scenes::SceneService::GetActiveScene()->GetSelectedEntity();
429  if (selectedEntity)
430  {
431  m_EditorScene->DestroyEntity(selectedEntity);
432  m_SceneHierarchyPanel->SetSelectedEntity({});
433  }
434  }
435  break;
436 
437  }
438  default:
439  {
440  break;
441  }
442  }
443  return false;
444  }
