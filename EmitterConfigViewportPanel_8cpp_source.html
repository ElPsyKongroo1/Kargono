46 void EmitterConfigViewportPanel::OnUpdate(Timestep ts)
47 {
48 KG_PROFILE_FUNCTION();
49
50 // Handle editor camera movement
51 FixedString32 focusedWindow{ EditorUI::EditorUIService::GetFocusedWindowName() };
52 if (focusedWindow == m_PanelName)
53 {
54 m_EditorCamera.OnUpdate(ts);
55 }
56
57 // Adjust framebuffer & camera viewport size if necessary
58 Window& currentWindow = EngineService::GetActiveWindow();
59 currentWindow.SetActiveViewport(&m_ViewportData);
60 if (Rendering::FramebufferSpecification spec = m_ViewportFramebuffer->GetSpecification();
61 (float)m_ViewportData.m_Width > 0.0f && (float)m_ViewportData.m_Height > 0.0f &&
62 (spec.Width != m_ViewportData.m_Width || spec.Height != m_ViewportData.m_Height))
63 {
64 // Update framebuffer and camera viewport size
65 m_ViewportFramebuffer->Resize(m_ViewportData.m_Width, m_ViewportData.m_Height);
66 m_EditorCamera.SetViewportSize((float)m_ViewportData.m_Width, (float)m_ViewportData.m_Height);
67 }
68
69 // Prepare for rendering
70 Rendering::RenderingService::ResetStats();
71 m_ViewportFramebuffer->Bind();
72 Rendering::RendererAPI::SetClearColor({ 0.1f, 0.1f, 0.1f, 1.0f });
73 Rendering::RendererAPI::Clear();
74
75 // Clear mouse picking attachment value
76 m_ViewportFramebuffer->SetAttachment(1, -1);
77
78 // Draw gridlines
79 DrawGridLines();
80
81 // Draw particle bounds
82 if (m_DisplaySpawningBoundsOutline)
83 {
84 DrawParticleSpawningBounds();
85 }
86
87 // Draw emitters
88 Particles::ParticleService::OnUpdate(ts);
89 Particles::ParticleService::OnRender(m_EditorCamera.GetViewProjection());
90
91
92 HandleMouseHovering();
93
94 // Complete rendering
95 m_ViewportFramebuffer->Unbind();
96
97 }

478 void EmitterConfigViewportPanel::DrawGridLines()
479 {
480 Rendering::RenderingService::BeginScene(m_EditorCamera);
481
482 Math::vec3 cameraPosition = m_EditorCamera.GetPosition();
483 Math::vec3 fineGridStart =
484 {
485 /*Utility::Operations::RoundDown((int32_t)(cameraPosition.x - (m_LargeGridSpacing / 2)), (int32_t)m_FineGridSpacing),
486 Utility::Operations::RoundDown((int32_t)(cameraPosition.y - (m_LargeGridSpacing / 2)), (int32_t)m_FineGridSpacing),
487 Utility::Operations::RoundDown((int32_t)(cameraPosition.z - (m_LargeGridSpacing / 2)), (int32_t)m_FineGridSpacing),*/
488 Utility::Operations::RoundDown((int32_t)cameraPosition.x, (int32_t)m_LargeGridSpacing),
489 Utility::Operations::RoundDown((int32_t)cameraPosition.y, (int32_t)m_LargeGridSpacing),
490 Utility::Operations::RoundDown((int32_t)cameraPosition.z, (int32_t)m_LargeGridSpacing)
491 };
492
493 // Set cameraFrustrumVertices 0 - 7 with vertices from camera frustum
494 Math::vec3 currentVertex;
495 Math::vec3 minimumValues{ std::numeric_limits<float>().max() };
496 Math::vec3 maximumValues{ -std::numeric_limits<float>().max() };
497 for (size_t i = 0; i < 8; i++)
498 {
499 Math::vec4 localSpaceCoordinates = glm::inverse(m_EditorCamera.GetProjection()) * s_DefaultFrustumVertexPositions[i];
500 localSpaceCoordinates = localSpaceCoordinates / localSpaceCoordinates.w; // Perspective Division
501 currentVertex = glm::inverse(m_EditorCamera.GetViewMatrix()) * localSpaceCoordinates;
502 if (currentVertex.x > maximumValues.x)
503 {
504 maximumValues.x = currentVertex.x;
505 }
506 if (currentVertex.y > maximumValues.y)
507 {
508 maximumValues.y = currentVertex.y;
509 }
510 if (currentVertex.z > maximumValues.z)
511 {
512 maximumValues.z = currentVertex.z;
513 }
514 if (currentVertex.x < minimumValues.x)
515 {
516 minimumValues.x = currentVertex.x;
517 }
518 if (currentVertex.y < minimumValues.y)
519 {
520 minimumValues.y = currentVertex.y;
521 }
522 if (currentVertex.z < minimumValues.z)
523 {
524 minimumValues.z = currentVertex.z;
525 }
526 }
527
528 // Start Grids
529 int32_t currentLine;
530 s_OutputVector->clear();
531 Rendering::Shader::SetDataAtInputLocation<Math::vec4>(Utility::ImVec4ToMathVec4(EditorUI::EditorUIService::s_GridMajor),
532 "a_Color", s_LineInputSpec.m_Buffer, s_LineInputSpec.m_Shader);
533 // X-Y Grid
534 if (m_DisplayXYMajorGrid)
535 {
536 // Create Y Lines
537 currentLine = Utility::Operations::RoundDown((int32_t)minimumValues.x, (int32_t)m_LargeGridSpacing);
538 while ((float)currentLine < maximumValues.x)
539 {
540 if (currentLine == 0)
541 {
542 currentLine += (int32_t)m_LargeGridSpacing;
543 continue;
544 }
545 s_OutputVector->clear();
546 s_OutputVector->push_back({ currentLine, minimumValues.y, 0.0f });
547 s_OutputVector->push_back({ currentLine, maximumValues.y, 0.0f });
548 s_LineInputSpec.m_ShapeComponent->Vertices = s_OutputVector;
549 Rendering::RenderingService::SubmitDataToRenderer(s_LineInputSpec);
550 currentLine += (int32_t)m_LargeGridSpacing;
551 }
552
553 // Create X Lines
554 currentLine = Utility::Operations::RoundDown((int32_t)minimumValues.y, (int32_t)m_LargeGridSpacing);
555 while ((float)currentLine < maximumValues.y)
556 {
557 if (currentLine == 0)
558 {
559 currentLine += (int32_t)m_LargeGridSpacing;
560 continue;
561 }
562 s_OutputVector->clear();
563 s_OutputVector->push_back({ minimumValues.x, currentLine, 0.0f });
564 s_OutputVector->push_back({ maximumValues.x, currentLine, 0.0f });
565 s_LineInputSpec.m_ShapeComponent->Vertices = s_OutputVector;
566 Rendering::RenderingService::SubmitDataToRenderer(s_LineInputSpec);
567 currentLine += (int32_t)m_LargeGridSpacing;
568 }
569 }
570
571 // Y-Z Grid
572 if (m_DisplayYZMajorGrid)
573 {
574 // Create Y Lines
575 currentLine = Utility::Operations::RoundDown((int32_t)minimumValues.y, (int32_t)m_LargeGridSpacing);
576 while ((float)currentLine < maximumValues.y)
577 {
578 if (currentLine == 0)
579 {
580 currentLine += (int32_t)m_LargeGridSpacing;
581 continue;
582 }
583 s_OutputVector->clear();
584 s_OutputVector->push_back({ 0.0f, currentLine, minimumValues.z });
585 s_OutputVector->push_back({ 0.0f, currentLine, maximumValues.z });
586 s_LineInputSpec.m_ShapeComponent->Vertices = s_OutputVector;
587 Rendering::RenderingService::SubmitDataToRenderer(s_LineInputSpec);
588 currentLine += (int32_t)m_LargeGridSpacing;
589 }
590
591 // Create Z Lines
592 currentLine = Utility::Operations::RoundDown((int32_t)minimumValues.z, (int32_t)m_LargeGridSpacing);
593 while ((float)currentLine < maximumValues.z)
594 {
595 if (currentLine == 0)
596 {
597 currentLine += (int32_t)m_LargeGridSpacing;
598 continue;
599 }
600 s_OutputVector->clear();
601 s_OutputVector->push_back({ 0.0f, minimumValues.y, currentLine });
602 s_OutputVector->push_back({ 0.0f, maximumValues.y, currentLine });
603 s_LineInputSpec.m_ShapeComponent->Vertices = s_OutputVector;
604 Rendering::RenderingService::SubmitDataToRenderer(s_LineInputSpec);
605 currentLine += (int32_t)m_LargeGridSpacing;
606 }
607 }
608
609 // X-Z Grid
610 if (m_DisplayXZMajorGrid)
