132 template<typename DataType>
133 Message& operator<<(Message& msg, const DataType& data)
134 {
135 // Check that the type of the data being pushed is trivially copyable
136 static_assert(std::is_standard_layout_v<DataType>, "Data is too complex");
137
138 // Cache current size of vector, as this will be the point we insert the data
139 size_t currentBufferSize = msg.Body.size();
140
141 // Resize the vector to contain the new data being added
142 msg.Body.resize(msg.Body.size() + sizeof(DataType));
143
144 // Copy the data over
145 std::memcpy(msg.Body.data() + currentBufferSize, &data, sizeof(DataType));
146
147 // Update message header size
148 msg.Header.PayloadSize = static_cast<uint32_t>(msg.Body.size());
149
150 return msg;
151 }

153 template<typename DataType>
154 Message& operator>>(Message& msg, const DataType& data)
155 {
156 // Check that the type of the data being pushed is trivially copyable
157 static_assert(std::is_standard_layout<DataType>::value, "Data is too complex");
158
159 // Cache current size of vector, as this will be the point we insert the data
160 size_t bufferSizeAfterRemoval = msg.Body.size() - sizeof(DataType);
161
162 // Copy the data over
163 std::memcpy((void*)&data, msg.Body.data() + bufferSizeAfterRemoval, sizeof(DataType));
164
165 // Shrink the vector to remove the read bytes
166 msg.Body.resize(bufferSizeAfterRemoval);
167
168 // Update message header size
169 msg.Header.PayloadSize = static_cast<uint32_t>(msg.Size());
