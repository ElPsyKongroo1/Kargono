41 //==============================
42 // Metadata Struct
43 //==============================
44 // This is the generic metadata struct that is directly held inside an asset.
45 // This struct holds generic information about an asset that is universally
46 // needed. The IntermediateLocation provides a relative path from the selected
47 // project's asset directory to the intermediate. As an aside, an intermediate
48 // is file that has been preprocessed and is directly ready to be loaded into
49 // the engine and used without the use of external tools. The checksum is generated
50 // differently based on the file type, but it uniquely identifies the file itself.
51 // The checksum is generally used to ensure duplicate files of the same type are
52 // not being stored. The AssetType allows an easy method to identify the specific
53 // asset type in a generic sense. The SpecificFileData holds a reference to data
54 // that is only needed by this specific type of asset. Ex: A texture might need
55 // to know its width and height, while an audio file might need to know its sample
56 // rate.
57 struct Metadata
58 {
59 public:
60 std::filesystem::path IntermediateLocation;
61 std::string CheckSum;
62 Assets::AssetType Type = Assets::AssetType::None;
63 Ref<void> SpecificFileData = nullptr;
64 public:
65 template <typename T>
66 T* GetSpecificFileData()
67 {
68 return static_cast<T*>(SpecificFileData.get());
69 }
70 };

79 //==============================
80 // Texture MetaData Struct
81 //==============================
82 // This metadata struct mostly holds loading information for the intermediate to load correctly.
83 // When the file is loaded from binary (Intermediate), it needs to know the
84 // image's width, height, and number of channels to interpret the binary
85 // correctly. The InitialFileLocation field is for debugging.
86 struct TextureMetaData
87 {
88 int32_t Width, Height, Channels;
89 std::filesystem::path InitialFileLocation;
90 };
91
92 //==============================
93 // Audio MetaData Struct
94 //==============================
95 // This metadata struct mostly holds loading information for the intermediate to load correctly.
96 // The binary form of Audio is held in a PCM format. The channels, sample rate, TotalPcmFrameCount,
97 // and the TotalSize are used for loading the file correctly. The InitialFileLocation field is for debugging.
98 struct AudioMetaData
99 {
100 uint32_t Channels, SampleRate;
101 uint64_t TotalPcmFrameCount, TotalSize;
102 std::filesystem::path InitialFileLocation;
103 };
104
105 //==============================
106 // Shader MetaData Struct
107 //==============================
108 // This metadata struct holds information to correctly interpret the shader binary when creating the final Shader object
109 // in the selected rendering API. The ShaderSpec field holds the specification that was originally used to generate
110 // the shader through the shader builder. The UniformList holds the list of uniforms needed for this shader to
111 // run correctly. The InputBufferLayout is generated when a shader is created in the shader builder (the uniform list too)
112 // and holds the format of per vertex input. Ex: vec3(vertex position), vec3(world position), vec4(color), etc...
113 struct ShaderMetaData
114 {
115 Rendering::ShaderSpecification ShaderSpec{};
116 Rendering::UniformBufferList UniformList{};
117 Rendering::InputBufferLayout InputLayout{};
118 };
119
120 //==============================
121 // Font MetaData Struct
122 //==============================
123 // This metadata struct holds the information needed to correctly load a selected font
124 // from its intermediate(binary) form. Fonts are originally loaded from a .ttf file which
125 // holds most of this metadata itself. When the font is converted into an intermediate form,
126 // the information that defines each character is saved in the Characters vector. This
127 // information includes the character's height, width, etc... The BufferLocations describe
128 // the exact location in the intermediate where each character image is stored. The index of the
129 // Characters and BufferLocations vectors correspond with each other. The Character object
130 // holds enough information to get the size of the image inside of the intermediate.
131 // The AverageWidth and AverageHeight fields are generated when loading the .ttf file and are
132 // used for text positioning in the user interface. The InitialFileLocation field is for
133 // debugging.
134 struct FontMetaData
135 {
136 float AtlasWidth;
137 float AtlasHeight;
138 float LineHeight{};
139 std::vector<std::pair<unsigned char, RuntimeUI::Character>> Characters{};
140 std::filesystem::path InitialFileLocation;
141 };

211 inline std::string AssetTypeToString(Assets::AssetType type)
212 {
213 switch (type)
214 {
215 case Assets::AssetType::Texture: return "Texture";
216 case Assets::AssetType::Shader: return "Shader";
217 case Assets::AssetType::Audio: return "Audio";
218 case Assets::AssetType::Font: return "Font";
219 case Assets::AssetType::Scene: return "Scene";
220 case Assets::AssetType::UserInterface: return "UserInterface";
221 case Assets::AssetType::InputMode: return "InputMode";
222 case Assets::AssetType::Script: return "Script";
223 case Assets::AssetType::GameState: return "GameState";
224 case Assets::AssetType::EntityClass: return "EntityClass";
225 case Assets::AssetType::None: return "None";
226 }
227 KG_ERROR("Unknown Type of AssetType.");
228 return "";
229 }
