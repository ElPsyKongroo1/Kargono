43 //==============================
44 // Metadata Struct
45 //==============================
46 // This is the generic metadata struct that is directly held inside an asset.
47 // This struct holds generic information about an asset that is universally
48 // needed. The IntermediateLocation provides a relative path from the selected
49 // project's asset directory to the intermediate. As an aside, an intermediate
50 // is file that has been preprocessed and is directly ready to be loaded into
51 // the engine and used without the use of external tools. The checksum is generated
52 // differently based on the file type, but it uniquely identifies the file itself.
53 // The checksum is generally used to ensure duplicate files of the same type are
54 // not being stored. The AssetType allows an easy method to identify the specific
55 // asset type in a generic sense. The SpecificFileData holds a reference to data
56 // that is only needed by this specific type of asset. Ex: A texture might need
57 // to know its width and height, while an audio file might need to know its sample
58 // rate.
59 struct Metadata
60 {
61 public:
62 std::filesystem::path FileLocation;
63 std::filesystem::path IntermediateLocation;
64 std::string CheckSum;
65 Assets::AssetType Type = Assets::AssetType::None;
66 Ref<void> SpecificFileData { nullptr };
67 public:
68 template <typename T>
69 T* GetSpecificMetaData()
70 {
71 return static_cast<T*>(SpecificFileData.get());
72 }
73 };

82 //==============================
83 // Texture MetaData Struct
84 //==============================
85 // This metadata struct mostly holds loading information for the intermediate to load correctly.
86 // When the file is loaded from binary (Intermediate), it needs to know the
87 // image's width, height, and number of channels to interpret the binary
88 // correctly. The InitialFileLocation field is for debugging.
89 struct TextureMetaData
90 {
91 int32_t Width, Height, Channels;
92 };
93
94 //==============================
95 // Audio MetaData Struct
96 //==============================
97 // This metadata struct mostly holds loading information for the intermediate to load correctly.
98 // The binary form of Audio is held in a PCM format. The channels, sample rate, TotalPcmFrameCount,
99 // and the TotalSize are used for loading the file correctly. The InitialFileLocation field is for debugging.
100 struct AudioMetaData
101 {
102 uint32_t Channels, SampleRate;
103 uint64_t TotalPcmFrameCount, TotalSize;
104 };
105
106 //==============================
107 // Shader MetaData Struct
108 //==============================
109 // This metadata struct holds information to correctly interpret the shader binary when creating the final Shader object
110 // in the selected rendering API. The ShaderSpec field holds the specification that was originally used to generate
111 // the shader through the shader builder. The UniformList holds the list of uniforms needed for this shader to
112 // run correctly. The InputBufferLayout is generated when a shader is created in the shader builder (the uniform list too)
113 // and holds the format of per vertex input. Ex: vec3(vertex position), vec3(world position), vec4(color), etc...
114 struct ShaderMetaData
115 {
116 Rendering::ShaderSpecification ShaderSpec{};
117 Rendering::UniformBufferList UniformList{};
118 Rendering::InputBufferLayout InputLayout{};
119 };
120
121 //==============================
122 // Font MetaData Struct
123 //==============================
124 // This metadata struct holds the information needed to correctly load a selected font
125 // from its intermediate(binary) form. Fonts are originally loaded from a .ttf file which
126 // holds most of this metadata itself. When the font is converted into an intermediate form,
127 // the information that defines each character is saved in the Characters vector. This
128 // information includes the character's height, width, etc... The BufferLocations describe
129 // the exact location in the intermediate where each character image is stored. The index of the
130 // Characters and BufferLocations vectors correspond with each other. The Character object
131 // holds enough information to get the size of the image inside of the intermediate.
132 // The AverageWidth and AverageHeight fields are generated when loading the .ttf file and are
133 // used for text positioning in the user interface. The InitialFileLocation field is for
134 // debugging.
135 struct FontMetaData
136 {
137 float AtlasWidth{ 0.0f };
138 float AtlasHeight{ 0.0f };
139 float LineHeight{ 0.0f };
140 float Ascender{ 0.0f };
141 float Descender{ 0.0f };
142 std::vector<std::pair<unsigned char, RuntimeUI::Character>> Characters{};
143 };
