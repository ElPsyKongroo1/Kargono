580 bool ScriptEditorPanel::OnAssetEvent(Events::Event* event)
581 {
582 Events::ManageAsset& manageEvent = *(Events::ManageAsset*)event;
583
584 // Validate asset is a script
585 if (manageEvent.GetAssetType() != Assets::AssetType::Script)
586 {
587 return false;
588 }
589
590 // Handle deletion event
591 if (manageEvent.GetAction() == Events::ManageAssetAction::PreDelete)
592 {
593 // Search table for deleted asset
594 std::size_t assetLocation = m_AllScriptsList.SearchEntries([&](const EditorUI::ListEntry& currentEntry)
595 {
596 // Check if handle matches
597 if (currentEntry.m_Handle != manageEvent.GetAssetID())
598 {
599 return false;
600 }
601
602 // Index has been found
603 return true;
604
605 });
606
607 // Validate that table search was successful
608 KG_ASSERT(assetLocation != EditorUI::k_ListSearchIndex, "Asset being deleted was not found in asset table");
609
610 // Delete entry and validate deletion
611 bool deletionSuccess = m_AllScriptsList.RemoveEntry(assetLocation);
612 KG_ASSERT(deletionSuccess, "Unable to delete asset inside script editor panel");
613 }
614
615 // Handle all othe event types
616 else
617 {
618 ResetPanelResources();
619 }
620
621 return true;
