563 void UIEditorTreePanel::RecalculateTreeIndexData()
564 {
565 // Recalculate the handle for each window entry in the user interface tree
566 m_UITree.EditDepth([&](EditorUI::TreeEntry& entry)
567 {
568 // Get tree path from entry reference
569 EditorUI::TreePath entryPath = m_UITree.GetPathFromEntryReference(&entry);
570
571 // Ensure the depth is 1, since this is the expected depth for the user interface tree
572 if (entryPath.GetDepth() != 2)
573 {
574 KG_WARN("Invalid depth length when revalidating window index information");
575 return;
576 }
577
578 // Get the window index from the path
579 uint16_t currentWindow = entryPath.GetBack();
580
581 // Update the handle for the window entry
582 entry.m_Handle = (uint64_t)currentWindow;
583
584 }, 1);
585
586
587 // Recalculate the handle and provided data for each widget entry in the user interface tree
588 m_UITree.EditDepth([&](EditorUI::TreeEntry& entry)
589 {
590 // Get tree path from entry reference
591 EditorUI::TreePath entryPath = m_UITree.GetPathFromEntryReference(&entry);
592
593 // Ensure the depth is 2, since this is the expected depth for the user interface tree
594 if (entryPath.GetDepth() != 3)
595 {
596 KG_WARN("Invalid depth length when revalidating widget index information");
597 return;
598 }
599
600 // Get the current widget and window index from the path
601 uint16_t currentWidget = entryPath.GetBack();
602 entryPath.PopBack();
603 uint16_t currentWindow = entryPath.GetBack();
604
605 // Update the handle and provided data for the widget entry
606 entry.m_Handle = (uint64_t)currentWidget;
607 entry.m_ProvidedData = CreateRef<uint32_t>((uint32_t)currentWindow);
608
609 }, 2);

611 void UIEditorTreePanel::SelectTreeNode(uint16_t windowID, uint16_t widgetID)
612 {
613 if (windowID == RuntimeUI::k_InvalidWindowID)
614 {
615 return;
616 }
617
618 EditorUI::TreePath path;
619 bool success{ false };
620
621 // Add UI node
622 path.AddNode(0);
623
624 // Add window node
625 path.AddNode(windowID);
626
627 if (widgetID == RuntimeUI::k_InvalidWidgetID)
628 {
629 m_UITree.ExpandFirstLayer();
630 success = m_UITree.SelectEntry(path);
631 }
632 else
633 {
634 m_UITree.ExpandFirstLayer();
635 m_UITree.ExpandNodePath(path);
636 path.AddNode(widgetID);
637 success = m_UITree.SelectEntry(path);
638 }
639
640 if (!success)
641 {
642 KG_WARN("Failed to select window/widget with ID {} and {}", windowID, widgetID);
643 }
