12 std::tuple<bool, std::string> ScriptOutputGenerator::GenerateOutput(ScriptAST&& ast)
13 {
14 m_AST = std::move(ast);
15 m_OutputText = {};
16
17 // Get Program Node
18 if (!m_AST.m_ProgramNode)
19 {
20 return { false, {} };
21 }
22
23 FunctionNode& funcNode = m_AST.m_ProgramNode.FuncNode;
24 // Emit Function Signature
25 m_OutputText << funcNode.ReturnType.Value << " " << funcNode.Name.Value << '(';
26 uint32_t iteration{ 0 };
27 for (auto& [allTypes, identifier] : funcNode.Parameters)
28 {
29 PrimitiveType primitiveType = ScriptCompilerService::s_ActiveLanguageDefinition.GetPrimitiveTypeFromName(allTypes.at(0).Value);
30 if (primitiveType.Name == "")
31 {
32 return { false, {} };
33 }
34 m_OutputText << primitiveType.EmittedParameter << ' ' << identifier.Value;
35 if (iteration + 1 < funcNode.Parameters.size())
36 {
37 m_OutputText << ", ";
38 }
39 iteration++;
40 }
41 m_OutputText << ")\n";
42 m_OutputText << "{\n";
43 for (auto statement : funcNode.Statements)
44 {
45 GenerateStatement(statement);
46 }
47 m_OutputText << "}\n";
48
49 return { true, m_OutputText.str() };
50 }

52 void ScriptOutputGenerator::GenerateStatement(Ref<Statement> statement)
53 {
54 if (StatementEmpty* emptyStatement = std::get_if<StatementEmpty>(&statement->Value))
55 {
56 // Do not display anything for an empty statement
57 }
58 else if (StatementExpression* expressionStatement = std::get_if<StatementExpression>(&statement->Value))
59 {
60 AddIndentation();
61 GenerateExpression(expressionStatement->Value);
62 m_OutputText << ";\n";
63 }
64 else if (StatementReturn* returnStatement = std::get_if<StatementReturn>(&statement->Value))
65 {
66 AddIndentation();
67
68 if (returnStatement->ReturnValue)
69 {
70 m_OutputText << "return ";
71 GenerateExpression(returnStatement->ReturnValue);
72 m_OutputText << ";\n";
73 }
74 else
75 {
76
77 m_OutputText << "return;\n";
78 }
79
80 }
81 else if (StatementDeclaration* declarationStatement = std::get_if<StatementDeclaration>(&statement->Value))
82 {
83 AddIndentation();
84 PrimitiveType typeValue = ScriptCompilerService::s_ActiveLanguageDefinition.GetPrimitiveTypeFromName(declarationStatement->Type.Value);
85 if (typeValue.Name == "")
86 {
87 return;
88 }
89 m_OutputText << typeValue.EmittedDeclaration << " " << declarationStatement->Name.Value << ";\n";
90
91 }
92 else if (StatementAssignment* assignmentStatement = std::get_if<StatementAssignment>(&statement->Value))
93 {
94 // Check if an OnGenerateSetter override is appropriate
95 if (MemberNode* memberNode = std::get_if<MemberNode>(&assignmentStatement->Name->Value))
96 {
97 // Get terminal node of MemberNode list
98 Ref<MemberNode> terminalNode = memberNode->ChildMemberNode;
99 {
100 while (terminalNode->ChildMemberNode)
101 {
102 terminalNode = terminalNode->ChildMemberNode;
103 }
104 }
105
106 // Check if terminal node uses an OnGenerateSetter override
107 if (DataMember* currentDataMember = std::get_if<DataMember>(&terminalNode->m_MemberType->Value))
108 {
109 if (currentDataMember->OnGenerateSetter)
110 {
111 AddIndentation();
112 currentDataMember->OnGenerateSetter(*this, *assignmentStatement);
113 m_OutputText << ";\n";
114 return;
115 }
116 }
117 }
118
119 // If not, simply output the assignment statement normally
120 AddIndentation();
121 GenerateExpression(assignmentStatement->Name);
122 m_OutputText << " = ";
123 GenerateExpression(assignmentStatement->Value);
124 m_OutputText << ";\n";
125
126 }
127 else if (StatementDeclarationAssignment* declarationAssignmentStatement = std::get_if<StatementDeclarationAssignment>(&statement->Value))
128 {
129 AddIndentation();
130 PrimitiveType typeValue = ScriptCompilerService::s_ActiveLanguageDefinition.GetPrimitiveTypeFromName(declarationAssignmentStatement->Type.Value);
131 if (typeValue.Name == "")
132 {
133 return;
134 }
135 m_OutputText << typeValue.EmittedDeclaration << " " << declarationAssignmentStatement->Name.Value << " = ";
136 GenerateExpression(declarationAssignmentStatement->Value);
137 m_OutputText << ";\n";
138 }
139
140 else if (StatementConditional* conditionalStatement = std::get_if<StatementConditional>(&statement->Value))
141 {
142 AddIndentation();
143 switch (conditionalStatement->Type)
144 {
145 case ConditionalType::IF:
146 m_OutputText << "if (";
147 break;
148 case ConditionalType::ELSEIF:
149 m_OutputText << "else if (";
150 break;
151 case ConditionalType::ELSE:
152 m_OutputText << "else\n";
153 break;
154 case ConditionalType::None:
155 default:
156 KG_WARN("Invalid conditional type provided to ScriptOutputGenerator");
157 return;
158 }
159
160 if (conditionalStatement->Type == ConditionalType::IF || conditionalStatement->Type == ConditionalType::ELSEIF)
161 {
162 GenerateExpression(conditionalStatement->ConditionExpression);
163 m_OutputText << ")\n";
164 }
165 AddIndentation();
166 m_OutputText << "{\n";
167
168 m_IndentLevel++;
169 for (auto& statement : conditionalStatement->BodyStatements)
170 {
171 GenerateStatement(statement);
172 }
173 m_IndentLevel--;
174 AddIndentation();
175 m_OutputText << "}\n";
176
177 for (auto& statement : conditionalStatement->ChainedConditionals)
178 {
179 GenerateStatement(statement);
180 }
181
182 }
183
184 else if (StatementWhileLoop* whileLoopStatement = std::get_if<StatementWhileLoop>(&statement->Value))
185 {
186 AddIndentation();
187
188 m_OutputText << "while (";
189 GenerateExpression(whileLoopStatement->ConditionExpression);
190 m_OutputText << ")\n";
191
192 AddIndentation();
193 m_OutputText << "{\n";
194
195 m_IndentLevel++;
196 for (auto& statement : whileLoopStatement->BodyStatements)
197 {
198 GenerateStatement(statement);
199 }
200 m_IndentLevel--;
201 AddIndentation();
202 m_OutputText << "}\n";
203 }
204
205 else if (StatementBreak* breakStatement = std::get_if<StatementBreak>(&statement->Value))
206 {
207 AddIndentation();
208 m_OutputText << "break;";
209 }
210
211 else if (StatementContinue* continueStatement = std::get_if<StatementContinue>(&statement->Value))
212 {
213 AddIndentation();
214 m_OutputText << "continue;";
215 }

219 void ScriptOutputGenerator::GenerateExpression(Ref<Expression> expression)
220 {
221 if (expression->GenerationAffixes)
222 {
223 m_OutputText << expression->GenerationAffixes->Prefix;
224 }
225
226 if (TokenExpressionNode* token = std::get_if<TokenExpressionNode>(&expression->Value))
227 {
228 KG_ASSERT(token->Value.Type != ScriptTokenType::CustomLiteral);
229
230 if (token->Value.Type == ScriptTokenType::MessageTypeLiteral)
231 {
232 m_OutputText << Utility::FileSystem::CRCFromString(token->Value.Value.c_str());
233 }
234 else
235 {
236 m_OutputText << token->Value.Value;
237 }
238 }
239 else if (FunctionCallNode* funcNode = std::get_if<FunctionCallNode>(&expression->Value))
240 {
241 if (!ScriptCompilerService::s_ActiveLanguageDefinition.FunctionDefinitions.contains(funcNode->Identifier.Value))
242 {
243 KG_WARN("Invalid function definition name provided when generating function call C++ code");
244 return;
245 }
246 std::function<void(ScriptOutputGenerator& generator, FunctionCallNode&)> onGenerateFunc =
247 ScriptCompilerService::s_ActiveLanguageDefinition.FunctionDefinitions.at(funcNode->Identifier.Value).OnGenerateFunction;
248
249 if (onGenerateFunc)
250 {
251 onGenerateFunc(*this, *funcNode);
252 }
253
254 if (funcNode->Namespace)
255 {
256 m_OutputText << funcNode->Namespace.Value << "::";
257 }
258 m_OutputText << funcNode->Identifier.Value << '(';
259 uint32_t iteration{ 0 };
260 for (auto& argument : funcNode->Arguments)
261 {
262 GenerateExpression(argument);
263 if (iteration + 1 < funcNode->Arguments.size())
264 {
265 m_OutputText << ", ";
266 }
267 iteration++;
268 }
269 m_OutputText << ')';
270 }
271
272 else if (CustomLiteralNode* customLiteralNode = std::get_if<CustomLiteralNode>(&expression->Value))
273 {
274 // Output ID of the customLiteral
275 m_OutputText << customLiteralNode->OutputValue;
276 }
277
278 else if (InitializationListNode* initListNode = std::get_if<InitializationListNode>(&expression->Value))
279 {
280 m_OutputText << '{';
281 uint32_t iteration{ 0 };
282 for (Ref<Expression> argument : initListNode->Arguments)
283 {
284 GenerateExpression(argument);
285 if (iteration + 1 < initListNode->Arguments.size())
286 {
287 m_OutputText << ", ";
288 }
289 iteration++;
290 }
291 m_OutputText << '}';
292 }
293 else if (UnaryOperationNode* unaryOperationNode = std::get_if<UnaryOperationNode>(&expression->Value))
294 {
295 m_OutputText << unaryOperationNode->Operator.Value << unaryOperationNode->Operand.Value;
296 }
297 else if (BinaryOperationNode* binaryOperationNode = std::get_if<BinaryOperationNode>(&expression->Value))
298 {
299 GenerateExpression(binaryOperationNode->LeftOperand);
300 m_OutputText << " " << binaryOperationNode->Operator.Value << " ";
301 GenerateExpression(binaryOperationNode->RightOperand);
302 }
303 else if (TernaryOperationNode* ternaryOperationNode = std::get_if<TernaryOperationNode>(&expression->Value))
304 {
305 GenerateExpression(ternaryOperationNode->Conditional);
306 m_OutputText << " ? ";
307 GenerateExpression(ternaryOperationNode->OptionOne);
308 m_OutputText << " : ";
309 GenerateExpression(ternaryOperationNode->OptionTwo);
310 }
311 else if (MemberNode* memberNode = std::get_if<MemberNode>(&expression->Value))
312 {
313 // Find terminal node
314 Ref<MemberNode> currentNode = memberNode->ChildMemberNode;
315 {
316 while (currentNode->ChildMemberNode)
317 {
318 currentNode = currentNode->ChildMemberNode;
319 }
320 }
321
322 // Check if member node uses an OnGenerateGetter function
323 bool useOnGenerate = false;
324 if (FunctionCallNode* funcCallNode = std::get_if<FunctionCallNode>(&currentNode->CurrentNodeExpression->Value))
325 {
326 if (funcCallNode->m_FunctionNode && funcCallNode->m_FunctionNode->OnGenerateGetter)
327 {
328 funcCallNode->m_FunctionNode->OnGenerateGetter(*this, *memberNode);
329 useOnGenerate = true;
330 }
331 }
332 else if (TokenExpressionNode* tokenNode = std::get_if<TokenExpressionNode>(&currentNode->CurrentNodeExpression->Value))
333 {
334 std::variant<FunctionNode, DataMember> Value {};
335 DataMember* currentDataMember = std::get_if<DataMember>(&currentNode->m_MemberType->Value);
336 KG_ASSERT(currentDataMember);
337
338 if (tokenNode && currentDataMember->OnGenerateGetter)
339 {
340 currentDataMember->OnGenerateGetter(*this, *memberNode);
341 useOnGenerate = true;
342 }
343 }
344
345 // If on generate is not available, then generate the output as it appears (Ex: entity.Transform.GetTranslation())
346 if (!useOnGenerate)
347 {
348 // Generate text for member node
349 GenerateExpression(memberNode->CurrentNodeExpression);
350 currentNode = memberNode->ChildMemberNode;
351 while (currentNode)
352 {
353 // Generate expression code for each MemberNode in linkedlist
354 m_OutputText << '.';
355 if (TokenExpressionNode* expressionToken = std::get_if<TokenExpressionNode>(&currentNode->CurrentNodeExpression->Value))
356 {
357 GenerateExpression(currentNode->CurrentNodeExpression);
358 }
359 else if (FunctionCallNode* expressionFunctionCall = std::get_if<FunctionCallNode>(&currentNode->CurrentNodeExpression->Value))
360 {
361 m_OutputText << expressionFunctionCall->Identifier.Value << '(';
362 uint32_t iteration{ 0 };
363 for (auto argument : expressionFunctionCall->Arguments)
364 {
365 GenerateExpression(argument);
366 if (iteration + 1 < expressionFunctionCall->Arguments.size())
367 {
368 m_OutputText << ", ";
369 }
370 iteration++;
371 }
372 m_OutputText << ')';
373 }
374 else
375 {
376 KG_WARN("Invalid current expression inside MemberNode when generating output");
377 }
378 currentNode = currentNode->ChildMemberNode;
379 }
380 }
381
382
383 }
384
385 if (expression->GenerationAffixes)
386 {
387 m_OutputText << expression->GenerationAffixes->Postfix;
388 }
