1151 static void TransferSceneData(Events::ManageAsset& event, Ref<Scenes::Scene> currentScene)
1152 {
1153 // Get reallocation instructions
1154 KG_ASSERT(event.GetProvidedData());
1155 Assets::FieldReallocationInstructions& reallocationInstructions = *(Assets::FieldReallocationInstructions*)event.GetProvidedData().get();
1156
1157 // Check if original data source is already empty. If it is empty, all there is to do is create a new registry
1158 if (reallocationInstructions.m_OldDataLocations.size() == 0)
1159 {
1160 // Create/Overwrite scene registry using the new archetype of field data
1161 currentScene->AddProjectComponentRegistry(event.GetAssetID());
1162 return;
1163 }
1164
1165 // Check if new data layout is empty. Simply remove the current registry.
1166 if (reallocationInstructions.m_NewDataLocations.size() == 0)
1167 {
1168 // Create/Overwrite scene registry using the new archetype of field data
1169 currentScene->ClearProjectComponentRegistry(event.GetAssetID());
1170 return;
1171 }
1172
1173 // Generate and store new entity component data
1174 std::unordered_map<UUID, std::vector<uint8_t>> entityToNewDataMap{};
1175 for (auto& [entityID, enttID] : currentScene->m_EntityRegistry.m_EntityMap)
1176 {
1177 ECS::Entity entity = currentScene->GetEntityByEnttID(enttID);
1178 if (entity.HasProjectComponentData(event.GetAssetID()))
1179 {
1180 // Get old data buffer reference and generate new data buffer
1181 uint8_t* oldComponentData = (uint8_t*)entity.GetProjectComponentData(event.GetAssetID());
1182 std::vector<uint8_t> newComponentData;
1183 newComponentData.resize(reallocationInstructions.m_NewDataSize);
1184
1185 // Transfer old data into new buffer and initialize indicated data
1186 for (size_t iteration{ 0 }; iteration < reallocationInstructions.m_FieldTransferDirections.size(); iteration++)
1187 {
1188 if (reallocationInstructions.m_FieldTransferDirections.at(iteration) == Assets::k_NewAllocationIndex)
1189 {
1190 // Initialize data
1191 Utility::InitializeDataForWrappedVarBuffer(reallocationInstructions.m_NewDataTypes.at(iteration), newComponentData.data() + reallocationInstructions.m_NewDataLocations.at(iteration));
1192 }
1193 else
1194 {
1195 // Transfer data
1196 Utility::TransferDataForWrappedVarBuffer(reallocationInstructions.m_NewDataTypes.at(iteration),
1197 oldComponentData + reallocationInstructions.m_OldDataLocations.at(reallocationInstructions.m_FieldTransferDirections.at(iteration)),
1198 newComponentData.data() + reallocationInstructions.m_NewDataLocations.at(iteration));
1199 }
1200 }
1201
1202 // Store new data into buffer map
1203 entityToNewDataMap.insert_or_assign(entityID, newComponentData);
1204 }
1205 }
1206
1207 // Clear old scene registry data for this project component type
1208 currentScene->ClearProjectComponentRegistry(event.GetAssetID());
1209
1210 // Create/Overwrite scene registry using the new archetype of field data
1211 currentScene->AddProjectComponentRegistry(event.GetAssetID());
1212
1213 // Write stored entity component data into new registry
1214 for (auto& [entityID, dataBuffer] : entityToNewDataMap)
1215 {
1216 ECS::Entity currentEntity = currentScene->GetEntityByUUID(entityID);
1217 if (!currentEntity.HasProjectComponentData(event.GetAssetID()))
1218 {
1219 currentEntity.AddProjectComponentData(event.GetAssetID());
1220 }
1221 uint8_t* currentData = (uint8_t*)currentEntity.GetProjectComponentData(event.GetAssetID());
1222 memcpy(currentData, dataBuffer.data(), dataBuffer.size()); // Note that the data buffer is a vector of bytes, which means size == byte size
1223 }
1224 }
